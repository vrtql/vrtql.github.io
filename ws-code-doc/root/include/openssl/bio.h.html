202/*
203 * The next three are used in conjunction with the BIO_should_io_special()
204 * condition. After this returns true, BIO *BIO_get_retry_BIO(BIO *bio, int

206 * and the offending BIO. Given a BIO, BIO_get_retry_reason(bio) will return
207 * the code.

226/*
227 * The callback is called before and after the underling operation, The
228 * BIO_CB_RETURN flag indicates if it is after the call
229 */
230# define BIO_CB_RETURN 0x80
231# define BIO_CB_return(a) ((a)|BIO_CB_RETURN)
232# define BIO_cb_pre(a) (!((a)&BIO_CB_RETURN))
233# define BIO_cb_post(a) ((a)&BIO_CB_RETURN)
