7/**
8 * @struct vws_http_msg
9 * @brief Structure representing an HTTP request
10 */
11typedef struct vws_http_msg
12{
13 /**< The parser */
14 http_parser* parser;
15
16 /**< The parser settings */
17 http_parser_settings* settings;
18
19 /**< A map storing header fields. */
20 struct sc_map_str headers;
21
22 /**< The url */
23 vws_buffer* url;
24
25 /**< The body */
26 vws_buffer* body;
27
28 /**< Placeholder for header field */
29 vws_buffer* field;
30
31 /**< Placeholder for header value */
32 vws_buffer* value;
33
34 /** Flag indicates a complete message has been parsed. */
35 bool complete;
36
37} vws_http_msg;
38
39/**
40 * @brief Creates a new instance of vws_http_msg.
41 * @param mode Must be one of HTTP_REQUEST or HTTP_RESPONSE depending on the
42 * message to parse.
43 * @return The newly created vws_http_msg instance.
44 */
45vws_http_msg* vws_http_msg_new(int mode);
46
47/**
48 * @brief Parses the provided data as an HTTP request.
49 * @param req The vws_http_msg instance.
50 * @param data The data to parse.
51 * @param size The size of the data.
52 * @return The number of bytes parsed, or a negative value on error.
53 */
54int vws_http_msg_parse(vws_http_msg* req, cstr data, size_t size);
55
56/**
57 * @brief Frees the resources associated with the vws_http_msg instance.
58 * @param req The vws_http_msg instance to free.
59 */
60void vws_http_msg_free(vws_http_msg* req);

62#endif /* VWS_HTTP_MSG_DECLARE */
