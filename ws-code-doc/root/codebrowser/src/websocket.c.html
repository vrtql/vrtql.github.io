20/** @brief Defines the various states of a WebSocket connection */
21typedef enum
22{
23 /** The connection with the client is closed. */
24 CNX_CLOSED = 0,
25
26 /** The connection with the client is established and open. */
27 CNX_CONNECTED = (1 << 1),
28
29 /** The connection with the client is in the process of being closed. */
30 CNX_CLOSING = (1 << 2),
31
32 /** The connection with the client is in the initial SSL handshake phase. */
33 CNX_SSL_INIT = (1 << 3),
34
35 /** The connection is in server mode. */
36 CNX_SERVER = (1 << 4)
37
38} cnx_flags_t;

40/** @brief Defines WebSocket close reason codes for CLOSE frames */
41typedef enum
42{
43 /** Normal Closure. This means that the purpose for which the connection was
44 * established has been fulfilled. */
45 WS_CLOSE_NORMAL = 1000,
46
47 /** Going Away. A server is going down, a browser has navigated away from a
48 * page, etc. */
49 WS_CLOSE_GOING_AWAY = 1001,
50
51 /** Protocol Error. The endpoint is terminating the connection due to a
52 * protocol error. */
53 WS_CLOSE_PROTOCOL_ERROR = 1002,
54
55 /** Unsupported Data. The connection is being terminated because an endpoint
56 * received a type of data it cannot accept. */
57 WS_CLOSE_UNSUPPORTED = 1003,
58
59 /** Reserved. The specific meaning might be defined in the future. */
60 WS_CLOSE_RESERVED = 1004,
61
62 /** No Status Received. Reserved value. The connection is closed with no
63 * status code. */
64 WS_CLOSE_NO_STATUS = 1005,
65
66 /** Abnormal Closure. Reserved value. The connection is closed with no
67 * status code. */
68 WS_CLOSE_ABNORMAL = 1006,
69
70 /** Invalid frame payload data. The endpoint is terminating the connection
71 * because a message was received that contains inconsistent data. */
72 WS_CLOSE_INVALID_PAYLOAD = 1007,
73
74 /** Policy Violation. The endpoint is terminating the connection because it
75 * received a message that violates its policy. */
76 WS_CLOSE_POLICY_VIOLATION = 1008,
77
78 /** Message Too Big. The endpoint is terminating the connection because a
79 * data frame was received that is too large. */
80 WS_CLOSE_TOO_BIG = 1009,
81
82 /** Missing Extension. The client is terminating the connection because it
83 * wanted the server to negotiate one or more extension, but the server
84 * didn't. */
85 WS_CLOSE_MISSING_EXTENSION = 1010,
86
87 /** Internal Error. The server is terminating the connection because it
88 * encountered an unexpected condition that prevented it from fulfilling the
89 * request. */
90 WS_CLOSE_INTERNAL_ERROR = 1011,
91
92 /** Service Restart. The server is terminating the connection because it is
93 * restarting. */
94 WS_CLOSE_SERVICE_RESTART = 1012,
95
96 /** Try Again Later. The server is terminating the connection due to a
97 * temporary condition, e.g. it is overloaded and is casting off some of its
98 * clients. */
99 WS_CLOSE_TRY_AGAIN_LATER = 1013,
100
101 /** Bad Gateway. The server was acting as a gateway or proxy and received an
102 * invalid response from the upstream server. This is similar to 502 HTTP
103 * Status Code. */
104 WS_CLOSE_BAD_GATEWAY = 1014,
105
106 /** TLS handshake. Reserved value. The connection is closed due to a failure
107 * to perform a TLS handshake. */
108 WS_CLOSE_TLS_HANDSHAKE = 1015
109
110} ws_close_code_t;

272bool cnx_connect(vws_cnx* c)
273{
274 if (c->url->host == NULL)
275 {
276 vws.error(VE_MEM, "Invalid or missing host");
277 return false;
278 }
279
280 // Connect to the server
281 cstr default_port = strcmp(c->url->protocol, "wss") == 0 ? "443" : "80";
282 cstr port = c->url->port != NULL ? c->url->port : default_port;
283
284 bool ssl = false;
285 if (strcmp(c->url->protocol, "wss") == 0)
286 {
287 ssl = true;
288 }
289
290 return vws_socket_connect((vws_socket*)c, c->url->host, atoi(port), ssl);
291}

315{
316 if (c == NULL)
317 {
318 return;
319 }
320
321 vws_disconnect(c);
322
323 // Free receive queue contents
324 vws_frame* f;
325 sc_queue_foreach (&c->queue, f)
326 {
327 vws_frame_free(f);
328 }
329
330 // Free receive queue
331 sc_queue_term(&c->queue);
332
333 // Free URL
334 if (c->url != NULL)
335 {
336 url_free((url_data_t*)c->url);
337 c->url = NULL;
338 }
339
340 // Free websocket key
341 vws.free(c->key);
342
343 // Call base constructor
344 vws_socket_dtor((vws_socket*)c);
345}

397bool socket_handshake(vws_socket* s)
398{
399 vws_cnx* c = (vws_cnx*)s;
400
401 // Send the WebSocket handshake request
402 const char* rt =
403 "GET %s HTTP/1.1\r\n"
404 "Host: %s\r\n"
405 "Cache-Control: no-cache\r\n"
406 "Origin: %s\r\n"
407 "Upgrade: websocket\r\n"
408 "Connection: Upgrade\r\n"
409 "Sec-WebSocket-Key: %s\r\n"
410 "Sec-WebSocket-Version: 13\r\n"
411 "\r\n";
412
413 char req[MAX_BUFFER_SIZE];
414 snprintf(req, sizeof(req), rt, c->url->path, c->url->host, c->url->href, c->key);
415
416 ssize_t n;
417 size_t total = 0;
418 size_t size = strlen(req);
419
420 while (true)
421 {
422 n = vws_socket_write(s, (ucstr)req, size);
423
424 if (vws_cnx_is_connected(c) == false)
425 {
426 return false;
427 }
428
429 if (n > 0)
430 {
431 total += n;
432
433 if (total == size)
434 {
435 break;
436 }
437 }
438
439 if (n == 0)
440 {
441 if (vws.e.code == VE_TIMEOUT)
442 {
443 return false;
444 }
445 }
446
447 if (n < 0)
448 {
449 return false;
450 }
451 }
452
453 // Create HTTP response message
454 vws_http_msg* http = vws_http_msg_new(HTTP_RESPONSE);
455
456 // Read until full HTTP response is received
457 while (true)
458 {
459 n = vws_socket_read(s);
460
461 if (vws_cnx_is_connected(c) == false)
462 {
463 // Clear the socket buffer of anything that did arrive,
464 // otherwise it will possibly be in inconsistent state.
465 vws_buffer_clear(c->base.buffer);
466
467 // Free HTTP response
468 vws_http_msg_free(http);
469
470 return false;
471 }
472
473 // If there was a timeout
474 if (n == 0)
475 {
476 // Fail
477 if (vws.e.code == VE_TIMEOUT)
478 {
479 // Clear the socket buffer of anything that did arrive,
480 // otherwise it will possibly be in inconsistent state.
481 vws_buffer_clear(c->base.buffer);
482
483 // Free HTTP response
484 vws_http_msg_free(http);
485
486 return false;
487 }
488 }
489
490 if (n < 0)
491 {
492 // Clear the socket buffer of anything that did arrive,
493 // otherwise it will possibly be in inconsistent state.
494 vws_buffer_clear(c->base.buffer);
495
496 // Free HTTP response
497 vws_http_msg_free(http);
498
499 return false;
500 }
501
502 if (s->buffer->size > 0)
503 {
504 cstr data = (cstr)s->buffer->data;
505 size_t size = s->buffer->size;
506 ssize_t n = vws_http_msg_parse(http, data, size);
507
508 if (http->headers_complete == true)
509 {
510 // Drain HTTP request data from socket buffer
511 vws_buffer_drain(c->base.buffer, n);
512
513 break;
514 }
515 }
516 }
517
518 struct sc_map_str* headers = &http->headers;
519 cstr accept_key = vws_map_get(headers, "sec-websocket-accept");
520
521 if (accept_key == NULL)
522 {
523 vws.error(VE_SYS, "connect failed: no accept key returned");
524
525 return false;
526 }
527
528 if (verify_handshake(c->key, accept_key) == false)
529 {
530 vws.error(VE_RT, "Handshake verification failed");
531 vws.free(accept_key);
532 return false;
533 }
534
535 vws_http_msg_free(http);
536
537 return true;
538}

540void vws_disconnect(vws_cnx* c)
541{
542 vws_socket* s = (vws_socket*)c;
543
544 if (vws_cnx_is_connected(c) == false)
545 {
546 return;
547 }
548
549 // If disconnect callback is registered
550 if (c->disconnect != NULL)
551 {
552 // Call it
553 c->disconnect(c);
554 }
555
556 c->flags = CNX_CLOSED;
557
558 vws_buffer* buffer = vws_generate_close_frame();
559
560 for (size_t i = 0; i < buffer->size;)
561 {
562 int n = vws_socket_write(s, buffer->data + i, buffer->size - i);
563
564 if (n < 0)
565 {
566 break;
567 }
568
569 i += n;
570 }
571
572 vws_buffer_free(buffer);
573
574 vws_socket_disconnect(s);
575}

611ssize_t vws_frame_send(vws_cnx* c, vws_frame* frame)
612{
613 if (vws_cnx_is_connected(c) == false)
614 {
615 return -1;
616 }
617
618 vws_buffer* binary = vws_serialize(frame);
619
620 if (vws.tracelevel >= VT_PROTOCOL)
621 {
622 vws_trace_lock();
623 printf("\n\n");
624 printf("+----------------------------------------------------+\n");
625 printf("| Frame Sent |\n");
626 printf("+----------------------------------------------------+\n");
627
628 vws_dump_websocket_frame(binary->data, binary->size);
629 printf("------------------------------------------------------\n");
630 vws_trace_unlock();
631 }
632
633 ssize_t n = 0;
634
635 if (binary->data != NULL)
636 {
637 n = vws_socket_write((vws_socket*)c, binary->data, binary->size);
638 vws_buffer_free(binary);
639
640 if (vws_cnx_is_connected(c) == false)
641 {
642 return -1;
643 }
644 }
645
646 vws.success();
647
648 return n;
649}

744{

746 vws.success();

769vws_buffer* vws_serialize(vws_frame* f)
770{
771 if (f == NULL)
772 {
773 vws.error(VE_RT, "empty frame");
774
775 return NULL;
776 }
777
778 //> Section 1: Size calculation
779
780 // Calculate the frame size
781 size_t payload_length = f->size;
782
783 // Set the mask bit and payload length. Maximum frame size with extended
784 // payload length and masking key
785 unsigned char header[14];
786
787 // Minimum frame size
788 size_t header_size = 2;
789
790 // Set the FIN bit and opcode
791 header[0] = f->fin << 7 | f->opcode;
792
793 if (payload_length <= 125)
794 {
795 header[1] = payload_length;
796 }
797 else if (payload_length <= 65535)
798 {
799 header[1] = 126;
800 header[2] = (payload_length >> 8) & 0xFF;
801 header[3] = payload_length & 0xFF;
802
803 // Additional bytes for payload length
804 header_size += 2;
805 }
806 else
807 {
808 header[1] = 127;
809 header[2] = (payload_length >> 56) & 0xFF;
810 header[3] = (payload_length >> 48) & 0xFF;
811 header[4] = (payload_length >> 40) & 0xFF;
812 header[5] = (payload_length >> 32) & 0xFF;
813 header[6] = (payload_length >> 24) & 0xFF;
814 header[7] = (payload_length >> 16) & 0xFF;
815 header[8] = (payload_length >> 8) & 0xFF;
816 header[9] = payload_length & 0xFF;
817
818 // Additional bytes for payload length
819 header_size += 8;
820 }
821
822 //> Section 2: Frame allocation
823
824 size_t frame_size = header_size + payload_length;
825
826 if (f->mask)
827 {
828 // Set the masking bit
829 header[1] |= 0x80;
830
831 // Additional bytes for masking key
832 frame_size += 4;
833 }
834
835 // Allocate memory for the frame
836 unsigned char* frame_data = (unsigned char*)vws.malloc(frame_size);
837
838 // Copy the header to the frame
839 memcpy(frame_data, header, header_size);
840
841 //> Section 3: Masking
842
843 if (f->mask)
844 {
845 // Generate a random masking key
846
847 unsigned char masking_key[4];
848
849 if (RAND_bytes(masking_key, sizeof(masking_key)) != 1)
850 {
851 vws.error(VE_RT, "RAND_bytes() failed");
852 vws.free(frame_data);
853 return NULL;
854 }
855
856 // Copy the masking key to the frame
857 memcpy(frame_data + header_size, masking_key, 4);
858
859 // Apply masking to the payload data
860 size_t payload_start = header_size + 4;
861 for (size_t i = 0; i < payload_length; i++)
862 {
863 frame_data[payload_start + i] = f->data[i] ^ masking_key[i % 4];
864 }
865 }
866 else
867 {
868 // Copy the payload data without masking
869 memcpy(frame_data + header_size, f->data, payload_length);
870 }
871
872 //> Section 4: Finalizing
873
874 // Free the frame
875 vws_frame_free(f);
876
877 // Create the vws_buffer to hold the frame data
878 vws_buffer* buffer = vws_buffer_new();
879
880 // Have buffer take ownership of data
881 buffer->data = frame_data;
882 buffer->size = frame_size;
883
884 vws.success();
885
886 return buffer;
887}

889fs_t vws_deserialize(ucstr data, size_t size, vws_frame* f, size_t* consumed)
890{
891 // Check if the data contains the minimum required frame header bytes
892 if (size < 2)
893 {
894 return FRAME_INCOMPLETE;
895 }
896
897 // Read the first byte (FIN bit and opcode)
898 f->fin = (data[0] >> 7) & 0x01;
899 f->opcode = data[0] & 0x0F;
900
901 // Read the second byte (mask bit and payload length)
902 f->mask = (data[1] >> 7) & 0x01;
903 f->size = data[1] & 0x7F;
904
905 // Check if the payload length requires additional bytes
906 size_t size_bytes = 0;
907 if (f->size == 126)
908 {
909 size_bytes = 2;
910 }
911 else if (f->size == 127)
912 {
913 size_bytes = 8;
914 }
915
916 // Check if the data contains complete frame header and payload
917 size_t required_bytes = 2 + size_bytes;
918
919 if (size < required_bytes)
920 {
921 return FRAME_INCOMPLETE;
922 }
923
924 // Read the payload
925 if (size_bytes > 0)
926 {
927 f->size = 0;
928 for (size_t i = 0; i < size_bytes; i++)
929 {
930 f->size = (f->size << 8) | data[2 + i];
931 }
932 }
933
934 // Check if the frame has masking key and payload data
935 if (f->mask)
936 {
937 // Check if the data contains the masking key and payload data
938 required_bytes += 4 + f->size;
939
940 if (size < required_bytes)
941 {
942 return FRAME_INCOMPLETE;
943 }
944
945 // Store the payload offset
946 f->offset = 2 + size_bytes + 4;
947
948 // Allocate the frame data
949 f->data = vws.malloc(f->size);
950
951 // Create a temp variable for the masking key
952 unsigned char mask[4];
953 memcpy(mask, data + 2 + size_bytes, 4);
954
955 // Read the payload data and apply the masking
956 for (size_t i = 0; i < f->size; i++)
957 {
958 f->data[i] = data[f->offset + i] ^ mask[i % 4];
959 }
960 }
961 else
962 {
963 // Check if the data contains the payload data
964
965 required_bytes += f->size;
966
967 if (size < required_bytes)
968 {
969 return FRAME_INCOMPLETE;
970 }
971
972 // Store the payload offset
973 f->offset = 2 + size_bytes;
974
975 // Allocate the frame data
976 f->data = vws.malloc(f->size);
977
978 // Copy the payload data
979 memcpy(f->data, data + f->offset, f->size);
980 }
981
982 // Update the bytes consumed
983 *consumed = required_bytes;
984
985 return FRAME_COMPLETE;
986}

992void process_frame(vws_cnx* c, vws_frame* f)
993{
994 switch (f->opcode)
995 {
996 case CLOSE_FRAME:
997 {
998 // Set closing state
999 vws_set_flag(&c->flags, CNX_CLOSING);
1000
1001 // Build the response frame
1002 vws_buffer* buffer = vws_generate_close_frame();
1003
1004 // Send the response frame
1005 vws_socket_write((vws_socket*)c, buffer->data, buffer->size);
1006
1007 // Clean up
1008 vws_buffer_free(buffer);
1009 vws_frame_free(f);
1010
1011 break;
1012 }
1013
1014 case TEXT_FRAME:
1015 case BINARY_FRAME:
1016 case CONTINUATION_FRAME:
1017 {
1018 // Add to queue
1019 sc_queue_add_first(&c->queue, f);
1020
1021 break;
1022 }
1023
1024 case PING_FRAME:
1025 {
1026 // Generate the PONG response
1027 vws_buffer* buffer = vws_generate_pong_frame(f->data, f->size);
1028
1029 // Send the PONG response
1030 vws_socket_write((vws_socket*)c, buffer->data, buffer->size);
1031
1032 // Clean up
1033 vws_buffer_free(buffer);
1034 vws_frame_free(f);
1035
1036 break;
1037 }
1038
1039 case PONG_FRAME:
1040 {
1041 // No need to send a response
1042
1043 vws_frame_free(f);
1044
1045 break;
1046 }
1047
1048 default:
1049 {
1050 // Invalid frame type
1051 vws_frame_free(f);
1052 }
1053 }
1054
1055 vws.success();
1056}

1078cstr vws_accept_key(cstr key)
1079{
1080 // Concatenate the key and WebSocket GUID
1081 const char* websocket_guid = "258EAFA5-E914-47DA-95CA-C5AB0DC85B11";
1082 size_t key_length = strlen(key);
1083 size_t guid_length = strlen(websocket_guid);
1084 size_t input_length = key_length + guid_length;
1085 char* input = (char*)vws.malloc(input_length + 1);
1086
1087 strncpy(input, key, key_length);
1088 strncpy(input + key_length, websocket_guid, guid_length);
1089 input[input_length] = '\0';
1090
1091 // Compute the SHA-1 hash of the concatenated value
1092 unsigned char hash[SHA_DIGEST_LENGTH];
1093 SHA1((const unsigned char*)input, input_length, hash);
1094
1095 // Base64-encode the hash
1096 char* encoded_hash = vws_base64_encode(hash, sizeof(hash));
1097
1098 vws.free(input);
1099
1100 return encoded_hash;
1101}

1112ssize_t socket_wait_for_frame(vws_cnx* c)
1113{
1114 // Default success unless error
1115 vws.success();
1116
1117 if (vws_cnx_is_connected(c) == false)
1118 {
1119 return -1;
1120 }
1121
1122 ssize_t n;
1123 unsigned char buf[1024];
1124
1125 while (true)
1126 {
1127 memset(buf, 0, 1024);
1128 n = vws_socket_read((vws_socket*)c);
1129
1130 if (n <= 0)
1131 {
1132 // Error already set
1133 return n;
1134 }
1135
1136 if (vws_cnx_ingress(c) > 0)
1137 {
1138 break;
1139 }
1140 }
1141
1142 return n;
1143}

1145ssize_t vws_cnx_ingress(vws_cnx* c)
1146{
1147 size_t total_consumed = 0;
1148
1149 // Process as many frames as possible
1150 while (true)
1151 {
1152 // If there is no more data in socket buffer
1153 if (c->base.buffer->size == 0)
1154 {
1155 break;
1156 }
1157
1158 // Attempt to parse complete frame
1159 size_t consumed = 0;
1160 vws_buffer* b = c->base.buffer;
1161 vws_frame* frame = vws_frame_new(NULL, 0, TEXT_FRAME);
1162
1163 if (vws.tracelevel >= VT_PROTOCOL)
1164 {
1165 vws_trace_lock();
1166 printf("\n+----------------------------------------------------+\n");
1167 printf("| Frame Received |\n");
1168 printf("+----------------------------------------------------+\n");
1169 vws_dump_websocket_frame(b->data, b->size);
1170 printf("------------------------------------------------------\n");
1171 vws_trace_unlock();
1172 }
1173
1174 fs_t rc = vws_deserialize(b->data, b->size, frame, &consumed);
1175
1176 if (rc == FRAME_ERROR)
1177 {
1178 vws.error(VE_WARN, "FRAME_ERROR");
1179 vws_frame_free(frame);
1180
1181 return 0;
1182 }
1183
1184 if (rc == FRAME_INCOMPLETE)
1185 {
1186 // No complete frame in socket buffer
1187 vws_frame_free(frame);
1188
1189 return 0;
1190 }
1191
1192 // Update
1193 total_consumed += consumed;
1194
1195 // We have a frame. Process it.
1196 c->process(c, frame);
1197
1198 // Drain the consumed frame data from buffer
1199 vws_buffer_drain(c->base.buffer, consumed);
1200 }
1201
1202 vws.success();
1203
1204 return total_consumed;
1205}

1230vws_msg* vws_msg_pop(vws_cnx* c)
1231{
1232 if (has_complete_message(c) == false)
1233 {
1234 return NULL;
1235 }
1236
1237 // Create new message
1238 vws_msg* m = vws_msg_new();
1239
1240 // Set to sentinel value to detect first frame
1241 m->opcode = 100;
1242
1243 do
1244 {
1245 vws_frame* f = sc_queue_del_last(&c->queue);
1246
1247 // If this is first frame, opcode is sentinel value. We take the opcode
1248 // from the first frame only
1249 if (m->opcode == 100)
1250 {
1251 m->opcode = f->opcode;
1252 }
1253
1254 // Copy frame data into message buffer
1255 vws_buffer_append(m->data, f->data, f->size);
1256
1257 // Is this the completion frame?
1258 bool complete = (f->fin == 1);
1259
1260 // Free frame
1261 vws_frame_free(f);
1262
1263 if (complete)
1264 {
1265 break;
1266 }
1267 }
1268 while (true);
1269
1270 return m;
1271}

1296void vws_dump_websocket_frame(const uint8_t* frame, size_t size)
1297{
1298 if (size < 2)
1299 {
1300 printf("Invalid WebSocket frame\n");
1301 return;
1302 }
1303
1304 ws_header header;
1305 size_t header_size = 2;
1306 header.fin = (frame[0] & 0x80) >> 7;
1307 header.opcode = frame[0] & 0x0F;
1308 header.mask = (frame[1] & 0x80) >> 7;
1309 header.payload_len = frame[1] & 0x7F;
1310
1311 if (header.payload_len == 126)
1312 {
1313 if (size < 4)
1314 {
1315 printf("Invalid WebSocket frame\n");
1316 return;
1317 }
1318
1319 header_size += 2;
1320 header.payload_len = ((uint64_t)frame[2] << 8) | frame[3];
1321 }
1322 else if (header.payload_len == 127)
1323 {
1324 if (size < 10)
1325 {
1326 printf("Invalid WebSocket frame\n");
1327 return;
1328 }
1329
1330 header_size += 8;
1331 header.payload_len =
1332 ((uint64_t)frame[2] << 56) |
1333 ((uint64_t)frame[3] << 48) |
1334 ((uint64_t)frame[4] << 40) |
1335 ((uint64_t)frame[5] << 32) |
1336 ((uint64_t)frame[6] << 24) |
1337 ((uint64_t)frame[7] << 16) |
1338 ((uint64_t)frame[8] << 8) |
1339 frame[9];
1340 }
1341
1342 if (header.mask)
1343 {
1344 if (size < header_size + 4)
1345 {
1346 printf("Invalid WebSocket frame\n");
1347 return;
1348 }
1349 header.masking_key =
1350 ((uint32_t)frame[header_size] << 24) |
1351 ((uint32_t)frame[header_size + 1] << 16) |
1352 ((uint32_t)frame[header_size + 2] << 8) |
1353 frame[header_size + 3];
1354 header_size += 4;
1355 }
1356 else
1357 {
1358 header.masking_key = 0;
1359 }
1360
1361 printf(" header: %zu bytes\n", header_size);
1362 dump_websocket_header(&header);
1363
1364 if (size > header_size)
1365 {
1366 for (size_t i = header_size; i < size; ++i)
1367 {
1368 printf("%02x ", frame[i]);
1369 if ((i - header_size + 1) % 16 == 0)
1370 {
1371 printf("\n");
1372 }
1373 }
1374
1375 printf("\n");
1376 }
1377}
