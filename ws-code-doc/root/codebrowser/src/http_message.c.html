75/**
76 * @brief Converts a string to lowercase.
77 * @param s The string to convert.
78 * @return The converted lowercase string.
79 */
80ucstr lcase(char* s)
81{
82 for(int i = 0; s[i]; i++)
83 {
84 s[i] = tolower(s[i]);
85 }
86
87 return (ucstr)s;
88}

90static void process_header(vws_http_msg* req)
91{
92 if (req->value->size != 0)
93 {
94 // We've got a complete field-value pair.
95
96 // Null-terminate
97 vws_buffer_append(req->field, (ucstr)"\0", 1);
98 vws_buffer_append(req->value, (ucstr)"\0", 1);
99
100 ucstr field = lcase((cstr)req->field->data);
101 ucstr data = req->value->data;
102 vws_map_set(&req->headers, (cstr)field, (cstr)data);
103
104 // Reset for the next field-value pair.
105 vws_buffer_clear(req->field);
106 vws_buffer_clear(req->value);
107 }
108}

110vws_http_msg* vws_http_msg_new(int mode)
111{
112 vws_http_msg* req = vws.malloc(sizeof(vws_http_msg));
113 req->parser = vws.malloc(sizeof(http_parser));
114 req->settings = vws.malloc(sizeof(http_parser_settings));
115 req->parser->data = req;
116 req->url = vws_buffer_new();
117 req->body = vws_buffer_new();
118 req->field = vws_buffer_new();
119 req->value = vws_buffer_new();
120 req->complete = false;
121
122 http_parser_init(req->parser, mode);
123 http_parser_settings_init(req->settings);
124
125 http_parser_settings* s = req->settings;
126 s->on_message_begin = on_message_begin;
127 s->on_headers_complete = on_headers_complete;
128 s->on_message_complete = on_message_complete;
129 s->on_url = on_url;
130 s->on_header_field = on_header_field;
131 s->on_header_value = on_header_value;
132 s->on_body = on_body;
133
134 sc_map_init_str(&req->headers, 0, 0);
135
136 return req;
137}

139void vws_http_msg_free(vws_http_msg* req)
140{
141 if (req == NULL)
142 {
143 return;
144 }
145
146 cstr key; cstr value;
147 sc_map_foreach(&req->headers, key, value)
148 {
149 vws.free(key);
150 vws.free(value);
151 }
152
153 sc_map_term_str(&req->headers);
154 vws_buffer_free(req->url);
155 vws_buffer_free(req->body);
156 vws_buffer_free(req->field);
157 vws_buffer_free(req->value);
158 vws.free(req->parser);
159 vws.free(req->settings);
160 vws.free(req);
161}
