73vws_buffer* vrtql_msg_serialize(vrtql_msg* msg)
74{
75 if (msg == NULL)
76 {
77 return false;
78 }
79
80 if (msg->format == VM_MPACK_FORMAT)
81 {
82 // Serialize MessagePack
83
84 // Buffer to hold data
85 vws_buffer* buffer = vws_buffer_new();
86
87 // Initialize writer
88
89 cstr key; cstr value;
90 mpack_writer_t writer;
91 mpack_writer_init_growable(&writer, (char**)&buffer->data, &buffer->size);
92
93 // Binary is an array of 3 elements: routing, headers, content.
94 mpack_start_array(&writer, 3);
95
96 // Generate routing
97
98 mpack_build_map(&writer);
99 sc_map_foreach(&msg->routing, key, value)
100 {
101 mpack_write_cstr(&writer, key);
102 mpack_write_cstr(&writer, value);
103 }
104 mpack_complete_map(&writer);
105
106 // Generate headers
107
108 mpack_build_map(&writer);
109 sc_map_foreach(&msg->headers, key, value)
110 {
111 mpack_write_cstr(&writer, key);
112 mpack_write_cstr(&writer, value);
113 }
114 mpack_complete_map(&writer);
115
116 // Create content
117
118 int size = msg->content->size;
119 cstr data = (cstr)msg->content->data;
120 mpack_write_bin(&writer, data, size);
121
122 // Close array
123 mpack_finish_array(&writer);
124
125 // Cleanup
126
127 mpack_error_t rc = mpack_writer_destroy(&writer);
128
129 if (rc != mpack_ok)
130 {
131 char buf[256];
132 cstr text = mpack_error_to_string(rc);
133 snprintf(buf, sizeof(buf), "Encoding errror: %s", text);
134 vws.error(VE_RT, buf);
135
136 vws_buffer_free(buffer);
137 return NULL;
138 }
139
140 return buffer;
141 }
142
143 if (msg->format == VM_JSON_FORMAT)
144 {
145 // Serialize JSON
146
147 // Buffer to hold data
148 vws_buffer* buffer = vws_buffer_new();
149
150 // Create a mutable doc
151 yyjson_mut_doc* doc = yyjson_mut_doc_new(NULL);
152
153 // We're generating an array as root.
154 yyjson_mut_val* root = yyjson_mut_arr(doc);
155
156 // Set root of the doc
157 yyjson_mut_doc_set_root(doc, root);
158
159 // Generate routing
160 yyjson_mut_val* routing = yyjson_mut_arr_add_obj(doc, root);
161 cstr key; cstr value;
162 sc_map_foreach(&msg->routing, key, value)
163 {
164 yyjson_mut_obj_add_str(doc, routing, key, value);
165 }
166
167 // Generate headers
168 yyjson_mut_val* headers = yyjson_mut_arr_add_obj(doc, root);
169 sc_map_foreach(&msg->headers, key, value)
170 {
171 yyjson_mut_obj_add_str(doc, headers, key, value);
172 }
173
174 // Add content
175 int size = msg->content->size;
176 cstr data = (cstr)msg->content->data;
177
178 if (size > 0)
179 {
180 yyjson_mut_arr_add_strncpy(doc, root, data, size);
181 }
182
183 // To string, minified
184 cstr json = yyjson_mut_write(doc, 0, NULL);
185
186 if (json)
187 {
188 // Assuming vws_buffer_write takes a char* and size, writes the
189 // data to the buffer
190 vws_buffer_append(buffer, (ucstr)json, strlen(json));
191 vws.free((void*)json);
192 }
193
194 // Free the doc
195 yyjson_mut_doc_free(doc);
196
197 return buffer;
198 }
199
200 return NULL;
201}

203bool vrtql_msg_deserialize(vrtql_msg* msg, ucstr data, size_t length)
204{
205 if ((data == NULL) || (length == 0))
206 {
207 return false;
208 }
209
210 // Parse message based on exptected format
211 //
212 // A message can be serialized into both JSON and MessagePack on the wire
213 // within the same connection. That is, the protocol supports auto-detection
214 // of JSON or MessagePack. The binary MessagePack format conveniently starts
215 // with an invalid UTF-8 character which works as a magic number to identify
216 // the binary message in the stream. This way a single connection can use
217 // both JSON and binary MessageMack messages in same stream. The presence of
218 // magic number signifies binary format and is therefore deserialized using
219 // MessagePack. Absent that it parses UTF-8/JSON until encountering a NULL
220 // terminator.
221 //
222 // The magic number comes from MessagePack binary format which is based on
223 // the number of arguments (n) to be serialized. We are a format consisting
224 // of an array with three elements: routing, headers, content. This means
225 // the value fori n is 3. Therefore the MessagePack format for this object
226 // would start with the value (0x90u | 3). This is our magic number which we
227 // use to check if the data is MessagePack. Everything else is assumed to be
228 // JSON.
229
230 unsigned char magic_number = (unsigned char)(0x90u | 3);
231 unsigned char first_byte = (unsigned char)data[0];
232
233 if (first_byte == magic_number)
234 {
235 // Deserialize MessagePack
236
237 // Initialize reader
238 mpack_reader_t reader;
239 mpack_reader_init_data(&reader, (cstr)data, length);
240
241 // Expect an array of size 3
242 if (mpack_expect_array(&reader) != 3)
243 {
244 vws.error(VE_RT, "Invalid MessagePack format");
245 return false;
246 }
247
248 // Parse routing map
249
250 if (msg_parse_map(&reader, &msg->routing) == false)
251 {
252 return false;
253 }
254
255 // Parse header map
256
257 if (msg_parse_map(&reader, &msg->headers) == false)
258 {
259 return false;
260 }
261
262 // Parse content
263
264 ssize_t n = msg_parse_content(&reader, msg->content);
265
266 if (n < 0)
267 {
268 return false;
269 }
270 else
271 {
272 msg->content->size = n;
273 }
274
275 // Finish reading the array
276 mpack_done_array(&reader);
277
278 // Cleanup
279
280 mpack_error_t rc = mpack_reader_destroy(&reader);
281
282 if (rc != mpack_ok)
283 {
284 char buf[256];
285 cstr text = mpack_error_to_string(rc);
286 snprintf(buf, sizeof(buf), "Decoding errror: %s", text);
287 vws.error(VE_RT, buf);
288
289 return false;
290 }
291
292 // Record format
293 msg->format = VM_MPACK_FORMAT;
294 }
295 else
296 {
297 // Deserialize JSON
298
299 yyjson_read_flag flags = YYJSON_READ_NOFLAG;
300 yyjson_doc* doc = yyjson_read((cstr)data, length, flags);
301 yyjson_val* root = yyjson_doc_get_root(doc);
302
303 if (!yyjson_is_arr(root) || yyjson_arr_size(root) != 3)
304 {
305 vws.error(VE_RT, "Invalid JSON: Root is not an array of size 3");
306 yyjson_doc_free(doc);
307
308 return false;
309 }
310
311 yyjson_val* routing = yyjson_arr_get(root, 0);
312 yyjson_val* headers = yyjson_arr_get(root, 1);
313 yyjson_val* content = yyjson_arr_get(root, 2);
314
315 yyjson_val* key;
316 yyjson_val* value;
317 yyjson_obj_iter iter;
318
319 if (routing && yyjson_is_obj(routing))
320 {
321 yyjson_obj_iter_init(routing, &iter);
322
323 while ((key = yyjson_obj_iter_next(&iter)))
324 {
325 value = yyjson_obj_iter_get_val(key);
326 vws_map_set( &msg->routing,
327 yyjson_get_str(key),
328 yyjson_get_str(value) );
329 }
330 }
331 else
332 {
333 vws.error(VE_RT, "Invalid JSON: routing not JSON object");
334 yyjson_doc_free(doc);
335
336 return false;
337 }
338
339 if (headers && yyjson_is_obj(headers))
340 {
341 yyjson_obj_iter_init(headers, &iter);
342 while ((key = yyjson_obj_iter_next(&iter)))
343 {
344 value = yyjson_obj_iter_get_val(key);
345 vws_map_set( &msg->headers,
346 yyjson_get_str(key),
347 yyjson_get_str(value) );
348 }
349 }
350 else
351 {
352 vws.error(VE_RT, "Invalid JSON: headers is not JSON object");
353 yyjson_doc_free(doc);
354
355 return false;
356 }
357
358 if (content && yyjson_is_str(content))
359 {
360 vrtql_msg_set_content(msg, yyjson_get_str(content));
361 }
362 else
363 {
364 vws.error(VE_RT, "Invalid JSON: content is not a string");
365 yyjson_doc_free(doc);
366
367 return false;
368 }
369
370 yyjson_doc_free(doc);
371
372 // Record format
373 msg->format = VM_JSON_FORMAT;
374 }
375
376 return true;
377}

506

508 vrtql_msg* m = vrtql_msg_new();
509 ucstr data = wsm->data->data;

511 if (vrtql_msg_deserialize(m, data, size) == false)

514 vws_msg_free(wsm);

517 }
518
519 vws_msg_free(wsm);
520
521 return m;
522}

528static bool msg_parse_map(mpack_reader_t* reader, struct sc_map_str* map)
529{
530 // Clear contents
531 vws_map_clear(map);
532
533 mpack_tag_t tag = mpack_read_tag(reader);
534
535 if (mpack_reader_error(reader) != mpack_ok)
536 {
537 return false;
538 }
539
540 if (mpack_tag_type(&tag) != mpack_type_map)
541 {
542 return false;
543 }
544
545 char* key;
546 char* value;
547
548 uint32_t count = mpack_tag_map_count(&tag);
549
550 while (count-- > 0)
551 {
552 uint32_t length;
553 cstr data;
554 mpack_tag_t tag;
555
556 //> Get key
557
558 tag = mpack_read_tag(reader);
559
560 if (mpack_tag_type(&tag) != mpack_type_str)
561 {
562 printf("ERROR: key must be string\n");
563 return false;
564 }
565
566 length = mpack_tag_str_length(&tag);
567 data = mpack_read_bytes_inplace(reader, length);
568 key = vws.malloc(length + 1);
569
570 memcpy(key, data, length);
571 key[length] = 0;
572 mpack_done_str(reader);
573
574 //> Get value
575
576 tag = mpack_read_tag(reader);
577
578 if (mpack_tag_type(&tag) != mpack_type_str)
579 {
580 printf("ERROR: value must be string\n");
581 return false;
582 }
583
584 length = mpack_tag_str_length(&tag);
585 data = mpack_read_bytes_inplace(reader, length);
586 value = vws.malloc(length + 1);
587
588 memcpy(value, data, length);
589 value[length] = 0;
590 mpack_done_str(reader);
591
592 sc_map_put_str(map, key, value);
593
594 if (mpack_reader_error(reader) != mpack_ok)
595 {
596 return false;
597 }
598 }
599
600 mpack_done_map(reader);
601
602 return true;
603}

605int32_t msg_parse_content(mpack_reader_t* reader, vws_buffer* buffer)
606{
607 mpack_tag_t tag = mpack_read_tag(reader);
608
609 if (mpack_reader_error(reader) != mpack_ok)
610 {
611 return -1;
612 }
613
614 mpack_type_t type = mpack_tag_type(&tag);
615
616 if ((type != mpack_type_bin) && (type != mpack_type_str))
617 {
618 return -1;
619 }
620
621 uint32_t length = mpack_tag_str_length(&tag);
622
623 if (length == 0)
624 {
625 return 0;
626 }
627
628 cstr data = mpack_read_bytes_inplace(reader, length);
629
630 vws_buffer_clear(buffer);
631 vws_buffer_append(buffer, (ucstr)data, length);
632 mpack_done_str(reader);
633
634 return length;
635}
