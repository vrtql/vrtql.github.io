19/**
20 * @brief Represents a message with routing, headers, and content.
21 */
22typedef struct vrtql_msg
23{
24 struct sc_map_str routing; /**< A map storing routing information. */
25 struct sc_map_str headers; /**< A map storing header fields. */
26 vws_buffer* content; /**< Buffer for the message content. */
27 uint64_t flags; /**< Message state flags */
28 vrtql_msg_format_t format; /**< Message format */
