14/**
15 * @brief Struct representing a RPC environment

18{
19 /**< The data */
20 char* data;
21
22} vrtql_rpc_env;
23
24/**
25 * @brief Callback for RPC call
26 * @param e The RPC environment
27 * @param m The incoming message to process
28 */
29typedef vrtql_msg* (*vrtql_rpc_call)(vrtql_rpc_env* e, vrtql_msg* m);
30
31/** Abbreviation for generic RPC map This is used to store registry of modules
32 * for the module system and registry of RPC calls for each module. */
33typedef struct sc_map_sv vrtql_rpc_map;
34
35typedef struct vrtql_rpc_module
36{
37 /**< Module name. */
38 cstr name;
39
40 /**< Map of RPC calls. Key is call name. Value is vrtql_rpc_call. */
41 vrtql_rpc_map calls;
42
43} vrtql_rpc_module;
44
45typedef struct vrtql_rpc_system
46{
47 /**< Map of RPC modules. Key is module name. Value is module instance. */
48 vrtql_rpc_map modules;
49
50} vrtql_rpc_system;
