19/**
20 * @brief Callback for out-of-band messages received during RPC call
21 * @param e The RPC environment
22 * @param m The incoming message to process
23 */
24typedef void (*vrtql_rpc_ob)(struct vrtql_rpc* rpc, vrtql_msg* m);
25
26/**
27 * @brief Callback for reconnect. This is called after successful websocket
28 * reconnection.
29 * @param e The RPC environment
30 * @return Returns true on successful reconnect, false otherwise.
31 */
32typedef bool (*vrtql_rpc_reconnect)(struct vrtql_rpc* rpc);
33
34/**
35 * @brief Struct representing a RPC environment
36 */
37typedef struct vrtql_rpc
38{
39 /**< WebSocket connection */
40 vws_cnx* cnx;
41
42 /**< The number of retries of timeout occurs (default 5) */
43 uint8_t retries;
44
45 /**> User-defined handler for out-of-band messages (default delete) */
46 vrtql_rpc_ob out_of_band;
47
48 /**> User-defined handler for reconnect */
49 vrtql_rpc_reconnect reconnect;
50
51 /**< Data from last response */
52 vws_buffer* val;
53
54 /**> User-defined data*/
55 void* data;
56
57} vrtql_rpc;
58
59/**
60 * @brief Creates a new RPC module.
61 *
62 * @param cnx The WebSocket connection to use
63 * @return A new thread RPC module.
64 */
65vrtql_rpc* vrtql_rpc_new(vws_cnx* cnx);
66
67/**
68 * @brief Frees the resources allocated to a RPC module
69 *
70 * @param rpc The RPC instance
71 */
72void vrtql_rpc_free(vrtql_rpc* rpc);
73
74/**
75 * @brief Low-level RPC call invocation. This takes a message as input, sends
76 * and waits for a response.
77 *
78 * @param rpc The RPC instance
79 * @param req The message to send
80 * @return The response message on success, NULL otherwise. The caller must free
81 * message with vrtql_msg_free(). If connection fails, error will have
82 * VE_SOCKET along with either VE_SEND if failure was on send or VE_RECV
83 * if failure was on receive. Caller can check by using
84 * vws_is_flag(&vws.e, VE_SEND) or vws_is_flag(&vws.e, VE_RECV).
85 */
86vrtql_msg* vrtql_rpc_exec(vrtql_rpc* rpc, vrtql_msg* req);
87
88/**
89 * @brief RPC call invocation. This takes a message as input, sends and waits
90 * for a response. It translates the response's "rc" and "msg" attributes to the
91 * vws.e.code and vws.e.text values respectively. The message content, if any,
92 * is copied into the rpc->val member.
93 *
94 * @param rpc The RPC instance
95 * @param req The message to send. This is automatically freed. Caller should
96 * NOT use this message again.
97 * @return True if the RPC call received a response. False otherwise. Call
98 * information is copies into vws.e. If connection fails, error will
99 * have VE_SOCKET along with either VE_SEND if failure was on send or
100 * VE_RECV if failure was on receive. Caller can check by using
101 * vws_is_flag(&vws.e, VE_SEND) or vws_is_flag(&vws.e, VE_RECV).
102 */
103bool vrtql_rpc_invoke(vrtql_rpc* rpc, vrtql_msg* req);
