17/**
18 * @brief Callback for handshake on connect. This is called after socket
19 * connection but before set to non-blocking. This provides a way to do basic
20 * socket handshake before going into poll() mode.
21
22 * @param s The socket
23 * @return Returns true if handshake succeeded, false otherwise
24 */
25typedef bool (*vws_socket_hs)(struct vws_socket* s);
26
27/**
28 * @brief A socket
29 */
30typedef struct vws_socket
31{
32 /**< The socket file descriptor. */
33 int sockfd;
34
35 /**< The SSL context for the connection. */
36 SSL_CTX* ssl_ctx;
37
38 /**< The SSL connection instance. */
39 SSL* ssl;
40
41 /**< Socket receive buffer. */
42 vrtql_buffer* buffer;
43
44 /**< Socket timeout in milliseconds. */
45 int timeout;
46
47 /**< Enable tracing. */
48 uint8_t trace;
49
50 /**< User-defined data associated with the connection */
51 char* data;
52
53 /**< User-defined handshake function to be called on connect */
54 vws_socket_hs hs;
55
56} vws_socket;

65/**
66 * @brief Allocates a new socket connection.
67 *
68 * @return A pointer to the new connection instance.
69 *
70 * @ingroup SocketFunctions

72vws_socket* vws_socket_new();

74/**
75 * @brief Deallocates a socket connection.
76 *
77 * @param c The socket connection.
78 */
79void vws_socket_free(vws_socket* s);
80
81/**
82 * @brief Server instance constructor
83 *
84 * Constructs a new server instance. This takes a new, empty vws_socket instance
85 * and initializes all of its members.
86 *
87 * @param s The socket instance to be initialized
88 * @return The initialized socket instance
89 *
90 * @ingroup SocketFunctions
91 */
92vws_socket* vws_socket_ctor(vws_socket* s);
93
94/**
95 * @brief Socket instance destructor
96 *
97 * Destructs an initialized socket instance.
98 *
99 * @param s The socket instance to be destructed
100 *
101 * @ingroup SocketFunctions
102 */
103void vws_socket_dtor(vws_socket* s);
104
105/**
106 * @brief Connects to a specified host URL.
107 *
108 * @param s The socket instance to be destructed
109 * @param host The host to connect to
110 * @param port The port to connect to
111 * @param ssl Flag to enable/disable SSL
112 * @return Returns true if the connection is successful, false otherwise.
113 *
114 * @ingroup SocketFunctions
115 */
116bool vws_socket_connect(vws_socket* s, cstr host, int port, bool ssl);
117
118/**
119 * @brief Sets a timeout on a socket read/write operations. The default
120 * timeout is 10 seconds.
121 *
122 * @param s The socket instance to be destructed
123 * @param sec The timeout value in seconds.
124 * @return True if successful, false otherwise.
125 *
126 * @ingroup SocketFunctions
127 */
128bool vws_socket_set_timeout(vws_socket* s, int sec);
129
130/**
131 * @brief Closes the connection to the host.
132 *
133 * @param c The socket connection.
134 *
135 * @ingroup SocketFunctions
136 */
137void vws_socket_disconnect(vws_socket* s);

139/**
140 * @brief Checks if a connection is established.

143 * @return Returns true if the connection is established, false otherwise.
144 *
145 * @ingroup SocketFunctions
146 */
147bool vws_socket_is_connected(vws_socket* s);

149/**
150 * @brief Closes a socket

152 * @param c The vws_socket

158/**
159 * @brief Reads data from a socket connection into a buffer

164 * @return The number of bytes read, or an error code if an error occurred.
165 *
166 * @ingroup SocketFunctions

168ssize_t vws_socket_read(vws_socket* s);

170/**
171 * @brief Writes data from a buffer to a socket
172 *
173 * @param c The vws_cnx representing the socket
174 * @param data The buffer containing the data to write.
175 * @param size The size of the data to write.
176 * @return The number of bytes written, or an error code if an error occurred.
177 *
178 * @ingroup SocketFunctions
179 */
180ssize_t vws_socket_write(vws_socket* s, ucstr data, size_t size);
