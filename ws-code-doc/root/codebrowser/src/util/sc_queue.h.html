51#ifndef SC_QUEUE_MAX
52#define SC_QUEUE_MAX (SIZE_MAX)
53#endif
54
55#define sc_queue_def(T, name) \
56 struct sc_queue_##name { \
57 bool oom; \
58 size_t cap; \
59 size_t first; \
60 size_t last; \
61 /* NOLINTNEXTLINE */ \
62 T *elems; \
63 }
64
65#define sc_queue_expand(q) \
66 do { \
67 size_t _cap, _len, _off; \
68 size_t _pos = ((q)->last + 1) & ((q)->cap - 1); \
69 void *_dst, *_src; \
70 \
71 if (_pos == (q)->first) { \
72 if ((q)->cap > SC_QUEUE_MAX / 2ul) { \
73 (q)->oom = true; \
74 break; \
75 } \
76 _cap = (q)->cap * 2; \
77 _dst = sc_queue_calloc(_cap, sizeof(*((q)->elems))); \
78 if (_dst == NULL) { \
79 (q)->oom = true; \
80 break; \
81 } \
82 _len = ((q)->cap - (q)->first) * sizeof(*(q)->elems); \
83 _off = ((q)->first * sizeof(*((q)->elems))); \
84 _src = ((char *) (q)->elems) + _off; \
85 \
86 memcpy(_dst, _src, _len); \
87 memcpy(((char *) _dst) + _len, (q)->elems, _off); \
88 (q)->oom = false; \
89 (q)->last = (q)->cap - 1; \
90 (q)->first = 0; \
91 (q)->cap = _cap; \
92 sc_queue_free((q)->elems); \
93 (q)->elems = _dst; \
94 } \
95 } while (0)

97/**
98 * Init queue. Call sc_queue_oom(q) to see if memory allocation succeeded.
99 * @param q queue
100 */
101#define sc_queue_init(q) \
102 do { \
103 (q)->oom = false; \
104 (q)->cap = 8; \
105 (q)->first = 0; \
106 (q)->last = 0; \
107 (q)->elems = sc_queue_calloc(1, sizeof(*(q)->elems) * 8); \
108 if ((q)->elems == NULL) { \
109 (q)->oom = true; \
110 } \
111 } while (0)
112
113/**
114 * Destroy queue
115 * @param q queue
116 */
117#define sc_queue_term(q) \
118 do { \
119 sc_queue_free((q)->elems); \
120 (q)->elems = NULL; \
121 (q)->cap = 0; \
122 (q)->first = 0; \
123 (q)->last = 0; \
124 (q)->oom = false; \
125 } while (0)

127/**
128 * @param q queue
129 * @return true if last add operation failed, false otherwise.
130 */
131#define sc_queue_oom(q) ((q)->oom)
132
133/**
134 * @param q queue
135 * @return element count
136 */
137#define sc_queue_size(q) (((q)->last - (q)->first) & ((q)->cap - 1))
138
139/**
140 * Clear the queue without deallocating underlying memory.
141 * @param q queue
142 */
143#define sc_queue_clear(q) \
144 do { \
145 (q)->first = 0; \
146 (q)->last = 0; \
147 (q)->oom = false; \
148 } while (0)
149
150/**
151 * @param q queue
152 * @return true if queue is empty
153 */
154#define sc_queue_empty(q) (((q)->last == (q)->first))
155
156/**
157 * @param q queue
158 * @return index of the first element. If queue is empty, result is undefined.
159 */
160#define sc_queue_first(q) ((q)->first)
161
162/**
163 * @param q queue
164 * @return index of the last element. If queue is empty, result is undefined.
165 */
166#define sc_queue_last(q) ((q)->last)
167
168/**
169 * @param q queue
170 * @param i index
171 * @return index of the next element after i, if i is the last element,
172 * result is undefined.
173 */
174#define sc_queue_next(q, i) (((i) + 1) & ((q)->cap - 1))
175
176/**
177 * Returns element at index 'i', so regular loops are possible :
178 *
179 * for (size_t i = 0; i < sc_queue_size(q); i++) {
180 * printf("%d" \n, sc_queue_at(q, i));
181 * }
182 *
183 * @param q queue
184 * @return element at index i
185 */
186#define sc_queue_at(q, i) (q)->elems[(((q)->first) + (i)) & ((q)->cap - 1)]
187
188/**
189 * @param q queue
190 * @return peek first element, if queue is empty, result is undefined
191 */
192#define sc_queue_peek_first(q) ((q)->elems[(q)->first])
193
194/**
195 * @param q queue
196 * @return peek last element, if queue is empty, result is undefined
197 */
198#define sc_queue_peek_last(q) (q)->elems[((q)->last - 1) & ((q)->cap - 1)]
199
200/**
201 * Call sc_queue_oom(q) after this function to check out of memory condition.
202 *
203 * @param q queue
204 * @param elem elem to be added at the end of the list
205 */
206#define sc_queue_add_last(q, elem) \
207 do { \
208 sc_queue_expand(q); \
209 if ((q)->oom) { \
210 break; \
211 } \
212 (q)->oom = false; \
213 (q)->elems[(q)->last] = elem; \
214 (q)->last = ((q)->last + 1) & ((q)->cap - 1); \
215 } while (0)
216
217/**
218 * @param q queue
219 * @return delete the last element from the queue and return its value.
220 * If queue is empty, result is undefined.
221 */
222#define sc_queue_del_last(q) \
223 ((q)->elems[((q)->last = ((q)->last - 1) & ((q)->cap - 1))])
224
225/**
226 * Call sc_queue_oom(q) after this function to check out of memory condition.
227 *
228 * @param q queue.
229 * @param elem elem to be added at the head of the list.
230 */
231#define sc_queue_add_first(q, elem) \
232 do { \
233 sc_queue_expand(q); \
234 if ((q)->oom) { \
235 break; \
236 } \
237 (q)->oom = false; \
238 (q)->first = ((q)->first - 1) & ((q)->cap - 1); \
239 (q)->elems[(q)->first] = elem; \
240 } while (0)
241
242static inline size_t sc_queue_inc_first(size_t *first, size_t cap)
243{
244 size_t tmp = *first;
245
246 *first = (*first + 1) & (cap - 1);
247 return tmp;
248}

250/**
251 * @param q queue
252 * @return delete the first element from the queue and return its value.
253 * If queue is empty, result is undefined.
254 */
255#define sc_queue_del_first(q) \
256 (q)->elems[sc_queue_inc_first(&(q)->first, (q)->cap)]
257
258/**
259 * For each loop,
260 *
261 * struct sc_queue_int queue;
262 * sc_queue_init(&queue, 4);"
263 *
264 * int elem;
265 * sc_queue_foreach(&queue, elem) {
266 * printf("Elem : %d \n, elem);
267 * }
268 */
269#define sc_queue_foreach(q, elem) \
270 for (size_t _k = 1, _i = sc_queue_first(q); \
271 _k && _i != sc_queue_last(q); \
272 _k = !_k, _i = sc_queue_next(q, _i)) \
273 for ((elem) = (q)->elems[_i]; _k; _k = !_k)
