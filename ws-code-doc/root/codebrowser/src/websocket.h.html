21/** @brief Defines the states of a WebSocket frame. */
22typedef enum
23{
24 /** Not all of the frame data has been received or processed. */
25 FRAME_INCOMPLETE,
26
27 /** All of the frame data has been received and processed. */
28 FRAME_COMPLETE,
29
30 /** There was an error in processing the frame data. */
31 FRAME_ERROR
32
33} fs_t;
34
35/** @brief Defines the types of WebSocket frames */
36typedef enum
37{
38 /** A continuation frame, data message split across multiple frames. */
39 CONTINUATION_FRAME = 0x0,
40
41 /** A text data frame. */
42 TEXT_FRAME = 0x1,
43
44 /** A binary data frame. */
45 BINARY_FRAME = 0x2,
46
47 /** A close control frame. */
48 CLOSE_FRAME = 0x8,
49
50 /** A ping control frame. */
51 PING_FRAME = 0x9,
52
53 /** A pong control frame, in response to a ping. */
54 PONG_FRAME = 0xA
55
56} frame_type_t;
57
58/**
59 * @brief Represents a Websocket frame.
60 */
61typedef struct vws_frame
62{
63 /**< Final frame in the message (1) or not (0). */
64 unsigned char fin;
65
66 /**< Defines the interpretation of the payload data. */
67 unsigned char opcode;
68
69 /**< Defines whether the payload is masked. */
70 unsigned char mask;
71
72 /**< Position of the data in the frame. */
73 unsigned int offset;
74
75 /**< The size of the payload data. */
76 unsigned long long size;
77
78 /**< The payload data for the frame. */
79 unsigned char* data;
80
81} vws_frame;
82
83/**
84 * @brief Instantiates a new websocket frame given some data, size, and opcode.
85 *
86 * @param data The data to be contained in the frame.
87 * @param s The size of the data.
88 * @param oc The opcode for the frame.
89 * @return Returns a pointer to the new frame.
90 */
91vws_frame* vws_frame_new(ucstr data, size_t s, unsigned char oc);
92
93/**
94 * @brief Frees memory associated with a vws_frame object.
95 *
96 * @param frame The vws_frame object to free.
97 * @return void
98 *
99 * @ingroup FrameFunctions
100 */
101void vws_frame_free(vws_frame* frame);
102
103/**
104 * @brief Serializes a vws_frame into a buffer that can be sent over the
105 * network.
106 *
107 * @param f The vws_frame to serialize.
108 * @return A pointer to the serialized buffer.
109 *
110 * @ingroup FrameFunctions
111 */
112vrtql_buffer* vws_serialize(vws_frame* f);
113
114/**
115 * @brief Deserializes raw network data into a vws_frame, updating consumed
116 * with the number of bytes processed.
117 *
118 * @param data The raw network data.
119 * @param size The size of the data.
120 * @param f The vws_frame to deserialize into.
121 * @param consumed Pointer to the number of bytes consumed during
122 * deserialization.
123 * @return The status of the deserialization process, 0 if successful, an error
124 * code otherwise.
125 *
126 * @ingroup FrameFunctions
127 */
128fs_t vws_deserialize(ucstr data, size_t size, vws_frame* f, size_t* consumed);

130/**

196/**
197 * @brief Callback for frame processing
198 * @param s The server instance
199 * @param t The incoming request to process
200 */
201typedef void (*vws_process_frame)(struct vws_cnx* cnx, vws_frame* frame);
202
203/**
204 * @brief A WebSocket connection.
205 */
206typedef struct vws_cnx
207{
208 /**< Base class */
209 struct vws_socket base;
210
211 /**< Connection state flags. */
212 uint64_t flags;
213
214 /**< The URL of the websocket server. */
215 vrtql_url url;
216
217 /**< The WebSocket origin. */
218 char* origin;
219
220 /**< The websocket key. */
221 char* key;
222
223 /**< Queue for incoming frames. */
224 struct sc_queue_ptr queue;
225
226 /**< Frame processing callback. */
227 vws_process_frame process;
228
229 /**< User-defined data associated with the connection */
230 char* data;
231
232} vws_cnx;

280/**
281 * @brief Processes incoming data from a WebSocket connection.
282 *
283 * This function parses the data in the socket buffer and attempts to extract
284 * complete WebSocket frames. It processes as many frames as possible and
285 * calls the appropriate processing function for each frame. The consumed
286 * data is then drained from the buffer.
287 *
288 * @param c The WebSocket connection.
289 * @return The total number of bytes consumed from the socket buffer.
290 * If no complete frame is available or an error occurs, it returns 0.
291 */
292ssize_t vws_socket_ingress(vws_cnx* c);
