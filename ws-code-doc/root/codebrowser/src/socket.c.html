75vws_socket* vws_socket_ctor(vws_socket* s)
76{
77 s->buffer = vrtql_buffer_new();
78 s->ssl = NULL;
79 s->ssl_ctx = NULL;
80 s->timeout = 10000;

82 s->hs = NULL;

97void vws_socket_dtor(vws_socket* s)
98{
99 vws_socket_disconnect(s);
100
101 // Free receive buffer
102 vrtql_buffer_free(s->buffer);
103
104 if (s->sockfd >= 0)
105 {
106 close(s->sockfd);
107 }
108
109 // Free connection
110 free(s);
111}

117bool vws_socket_set_timeout(vws_socket* s, int sec)
118{
119 if (socket_set_timeout(s->sockfd, sec) == false)
120 {
121 return false;
122 }
123
124 // Set socket attribute, this will apply to poll().
125 s->timeout = 10000;
126
127 return true;
128}

130bool socket_set_timeout(int fd, int sec)
131{
132#if defined(__linux__) || defined(__bsd__) || defined(__sunos__)
133
134 if (fd < 0)
135 {
136 vrtql.error(VE_RT, "Not connected");
137 return false;
138 }
139
140 struct timeval tm;
141 tm.tv_sec = sec;
142 tm.tv_usec = 0;
143
144 // Set the send timeout
145 if (setsockopt(fd, SOL_SOCKET, SO_SNDTIMEO, (char *)&tm, sizeof(tm)) < 0)
146 {
147 vrtql.error(VE_SYS, "setsockopt failed");
148
149 return false;
150 }
151
152 // Set the receive timeout
153 if (setsockopt(fd, SOL_SOCKET, SO_RCVTIMEO, (char *)&tm, sizeof(tm)) < 0)
154 {
155 vrtql.error(VE_SYS, "setsockopt failed");
156
157 return false;
158 }
159
160#elif defined(__windows__)
161
162 if (fd == INVALID_SOCKET)
163 {
164 vrtql.error(VE_RT, "Not connected");
165 return false;
166 }
167
168 // Convert from sec to ms for Windows
169 DWORD tm = sec * 1000;
170
171 // Set the send timeout
172 if (setsockopt(fd, SOL_SOCKET, SO_SNDTIMEO, (cstr)&tm, sizeof(tm)) < 0)
173 {
174 vrtql.error(VE_SYS, "setsockopt failed");
175
176 return false;
177 }
178
179 // Set the receive timeout
180 if (setsockopt(fd, SOL_SOCKET, SO_RCVTIMEO, (cstr)&tm, sizeof(tm)) < 0)
181 {
182 vrtql.error(VE_SYS, "setsockopt failed");
183
184 return false;
185 }
186
187#else
188 #error Platform not supported
189#endif
190
191 vrtql.success();
192
193 return true;
194}

196bool socket_set_nonblocking(int sockfd)
197{
198#if defined(__linux__) || defined(__bsd__) || defined(__sunos__)
199
200 int flags = fcntl(sockfd, F_GETFL, 0);
201
202 if (flags == -1)
203 {
204 vrtql.error(VE_SYS, "fcntl(sockfd, F_GETFL, 0) failed");
205
206 return false;
207 }
208
209 if (fcntl(sockfd, F_SETFL, flags | O_NONBLOCK) == -1)
210 {
211 vrtql.error(VE_SYS, "fcntl(sockfd, F_SETFL, flags | O_NONBLOCK) failed");
212
213 return false;
214 }
215
216#elif defined(__windows__)
217
218 unsigned long arg = 1;
219 if (ioctlsocket(sockfd, FIONBIO, &arg) == SOCKET_ERROR)
220 {
221 vrtql.error(VE_SYS, "ioctlsocket(sockfd, FIONBIO, &arg)");
222
223 return false;
224 }
225
226#else
227 #error Platform not supported
228#endif
229
230 vrtql.success();
231
232 return true;
233}

245bool vws_socket_connect(vws_socket* c, cstr host, int port, bool ssl)
246{
247 if (c == NULL)
248 {
249 // Return early if failed to create a connection.
250 vrtql.error(VE_RT, "Invalid connection pointer()");
251 return false;
252 }
253
254 if (ssl == true)
255 {
256 if (vrtql_is_flag(&vrtql.state, CNX_SSL_INIT) == false)
257 {
258 SSL_library_init();
259 RAND_poll();
260 SSL_load_error_strings();
261
262 c->ssl_ctx = SSL_CTX_new(TLS_method());
263
264 if (c->ssl_ctx == NULL)
265 {
266 vrtql.error(VE_SYS, "Failed to create new SSL context");
267 return false;
268 }
269
270 SSL_CTX_set_options(c->ssl_ctx, SSL_OP_NO_SSLv2 | SSL_OP_NO_SSLv3);
271
272 c->ssl = SSL_new(c->ssl_ctx);
273
274 if (c->ssl == NULL)
275 {
276 vrtql.error(VE_SYS, "Failed to create new SSL object");
277 vws_socket_close(c);
278 return false;
279 }
280
281 vrtql_set_flag(&vrtql.state, CNX_SSL_INIT);
282 }
283 }
284
285 char port_str[20];
286 sprintf(port_str, "%d", port);
287 c->sockfd = connect_to_host(host, port_str);
288
289 if (c->sockfd < 0)
290 {
291 vrtql.error(VE_SYS, "Connection failed");
292 vws_socket_close(c);
293 return false;
294 }
295
296 // Set default timeout
297
298 if (socket_set_timeout(c->sockfd, c->timeout/1000) == false)
299 {
300 // Error already set
301 vws_socket_close(c);
302 return false;
303 }
304
305 if (c->ssl != NULL)
306 {
307 SSL_set_fd(c->ssl, c->sockfd);
308
309 if (SSL_connect(c->ssl) <= 0)
310 {
311 vrtql.error(VE_SYS, "SSL connection failed");
312 vws_socket_close(c);
313 return false;
314 }
315 }
316
317 // Check if handshake handler is registered
318 if (c->hs != NULL)
319 {
320 if (c->hs(c) == false)
321 {
322 vrtql.error(VE_SYS, "Handshake failed");
323 vws_socket_close(c);
324 return false;
325 }
326 }
327
328 // Go into non-blocking mode as we are using poll() for socket_read() and
329 // socket_write().
330 if (socket_set_nonblocking(c->sockfd) == false)
331 {
332 // Error already set
333 vws_socket_close(c);
334 return false;
335 }
336
337 vrtql.success();
338
339 return true;
340}

354ssize_t vws_socket_read(vws_socket* c)
355{
356 // Default success unless error
357 vrtql.success();
358
359 // Validate input parameters
360 if (c == NULL)
361 {
362 vrtql.error(VE_WARN, "Invalid parameters");
363 return -1;
364 }
365
366#if defined(__linux__) || defined(__bsd__) || defined(__sunos__)
367
368 struct pollfd fds;
369 fds.fd = c->sockfd;
370 fds.events = POLLIN;
371
372 int poll_result = poll(&fds, 1, c->timeout);
373
374#elif defined(__windows__)
375
376 WSAPOLLFD fds;
377 fds.fd = c->sockfd;
378 fds.events = POLLIN;
379
380 int poll_result = WSAPoll(&fds, 1, c->timeout);
381
382#else
383 #error Platform not supported
384#endif
385
386 if (poll_result == -1)
387 {
388 vrtql.error(VE_RT, "poll() failed");
389 return -1;
390 }
391
392 if (poll_result == 0)
393 {
394 vrtql.error(VE_WARN, "timeout");
395 return 0;
396 }
397
398 unsigned char data[1024];
399 int size = 1024;
400 ssize_t n = 0;
401
402 if (fds.revents & POLLIN)
403 {
404 if (c->ssl != NULL)
405 {
406 // SSL socket is readable, perform SSL_read() operation
407 n = SSL_read(c->ssl, data, size);
408 }
409 else
410 {
411 // Non-SSL socket is readable, perform recv() operation
412 n = recv(c->sockfd, data, size, 0);
413
414 if (n == -1)
415 {
416 vrtql.error(VE_WARN, "recv() failed");
417 }
418 }
419 }
420
421 if (n > 0)
422 {
423 vrtql_buffer_append(c->buffer, data, n);
424 }
425 else
426 {
427 vrtql.error(VE_WARN, "Unexpected event in poll()");
428 }
429
430 return n;
431}

433ssize_t vws_socket_write(vws_socket* c, const ucstr data, size_t size)
434{
435 // Default success unless error
436 vrtql.success();
437
438 if (vws_socket_is_connected(c) == false)
439 {
440 vrtql.error(VE_RT, "Not connected");
441 return -1;
442 }
443
444 // Validate input parameters
445 if (c == NULL || data == NULL || size == 0)
446 {
447 vrtql.error(VE_WARN, "Invalid parameters");
448 return -1;
449 }
450
451#if defined(__linux__) || defined(__bsd__) || defined(__sunos__)
452
453 struct pollfd fds;
454 fds.fd = c->sockfd;
455 fds.events = POLLOUT;
456
457 int poll_result = poll(&fds, 1, c->timeout);
458
459#elif defined(__windows__)
460
461 WSAPOLLFD fds;
462 fds.fd = c->sockfd;
463 fds.events = POLLOUT;
464
465 int poll_result = WSAPoll(&fds, 1, c->timeout);
466
467#else
468 #error Platform not supported
469#endif
470
471 if (poll_result == -1)
472 {
473 vrtql.error(VE_SYS, "poll() failed");
474 return -1;
475 }
476
477 if (poll_result == 0)
478 {
479 vrtql.error(VE_TIMEOUT, "timeout");
480 return 0;
481 }
482
483 if (fds.revents & POLLOUT)
484 {
485 if (c->ssl != NULL)
486 {
487 // SSL socket is writable, perform SSL_write() operation
488 return SSL_write(c->ssl, data, size);
489 }
490 else
491 {
492 // Non-SSL socket is writable, perform send() operation
493 ssize_t bytes_sent = send(c->sockfd, data, size, 0);
494 if (bytes_sent == -1)
495 {
496 vrtql.error(VE_SYS, "send() failed");
497 }
498
499 return bytes_sent;
500 }
501 }
502
503 vrtql.error(VE_SYS, "Unexpected event in poll()");
504
505 return -1;
506}

508void vws_socket_close(vws_socket* c)
509{
510 if (c->ssl != NULL)
511 {
512 int rc = SSL_shutdown(c->ssl);
513
514 if (rc == 0)
515 {
516 // the shutdown is not yet finished
517 rc = SSL_shutdown(c->ssl);
518 }
519
520 if (rc != 1)
521 {
522 vrtql.error(VE_WARN, "SSL_shutdown failed");
523 }
524
525 SSL_free(c->ssl);
526 c->ssl = NULL;
527 }
528
529 if (c->ssl_ctx != NULL)
530 {
531 SSL_CTX_free(c->ssl_ctx);
532 c->ssl_ctx = NULL;
533 }
534
535 if (c->sockfd > 0)
536 {
537#if defined(__windows__)
538 if (closesocket(c->sockfd) == SOCKET_ERROR)
539#else
540 if (close(c->sockfd) == -1)
541#endif
542 {
543 vrtql.error(VE_SYS, "Socket close failed");
544 }
545
546#if defined(__windows__)
547 WSACleanup();
548#endif
549
550 c->sockfd = -1;
551 }
552}

554int connect_to_host(const char* host, const char* port)
555{
556 int sockfd = -1;
557
558#if defined(__linux__) || defined(__bsd__) || defined(__sunos__)
559
560 // Resolve the host
561 struct addrinfo hints, *res, *res0;
562 int error;
563
564 memset(&hints, 0, sizeof(hints));
565 hints.ai_family = PF_UNSPEC; // Accept any family (IPv4 or IPv6)
566 hints.ai_socktype = SOCK_STREAM;
567
568 error = getaddrinfo(host, port, &hints, &res0);
569
570 if (error)
571 {
572 if (vrtql.trace > 0)
573 {
574 cstr msg = gai_strerror(error);
575 vrtql_trace(VL_ERROR, "getaddrinfo failed: %s: %s", host, msg);
576 }
577
578 vrtql.error(VE_SYS, "getaddrinfo() failed");
579
580 return -1;
581 }
582
583 for (res = res0; res; res = res->ai_next)
584 {
585 sockfd = socket(res->ai_family, res->ai_socktype, res->ai_protocol);
586
587 if (sockfd == -1)
588 {
589 vrtql.error(VE_SYS, "Failed to create socket");
590 continue;
591 }
592
593 if (connect(sockfd, res->ai_addr, res->ai_addrlen) == -1)
594 {
595 close(sockfd);
596 sockfd = -1;
597
598 vrtql.error(VE_SYS, "Failed to connect");
599 continue;
600 }
601
602 break; // If we get here, we must have connected successfully
603 }
604
605 freeaddrinfo(res0); // Free the addrinfo structure for this host
606
607#elif defined(__windows__)
608
609 // Windows specific implementation
610 // Please refer to Windows Socket programming guide
611
612 WSADATA wsaData;
613 struct addrinfo *result = NULL, *ptr = NULL, hints;
614 sockfd = INVALID_SOCKET;
615
616 // Initialize Winsock
617 if (WSAStartup(MAKEWORD(2,2), &wsaData) != 0)
618 {
619 vrtql.error(VE_SYS, "WSAStartup failed");
620 return -1;
621 }
622
623 ZeroMemory(&hints, sizeof(hints));
624 hints.ai_family = AF_UNSPEC;
625 hints.ai_socktype = SOCK_STREAM;
626 hints.ai_protocol = IPPROTO_TCP;
627
628 // Resolve the server address and port
629 if (getaddrinfo(host, port, &hints, &result) != 0)
630 {
631 vrtql.error(VE_SYS, "getaddrinfo failed\n");
632 return -1;
633 }
634
635 // Attempt to connect to an address until one succeeds
636 for (ptr = result; ptr != NULL; ptr =ptr->ai_next)
637 {
638 // Create a SOCKET for connecting to server
639 sockfd = socket(ptr->ai_family, ptr->ai_socktype, ptr->ai_protocol);
640
641 if (sockfd == INVALID_SOCKET)
642 {
643 char buf[256];
644 int e = WSAGetLastError();
645 snprintf(buf, sizeof(buf), "socket failed with error: %ld", e);
646 vrtql.error(VE_RT, buf);
647
648 WSACleanup();
649 return -1;
650 }
651
652 // Connect to server.
653 if (connect(sockfd, ptr->ai_addr, (int)ptr->ai_addrlen) == SOCKET_ERROR)
654 {
655 closesocket(sockfd);
656 sockfd = INVALID_SOCKET;
657 continue;
658 }
659
660 break;
661 }
662
663 freeaddrinfo(result);
664
665 if (sockfd == INVALID_SOCKET)
666 {
667 vrtql.error(VE_SYS, "Unable to connect to host");
668 WSACleanup();
669 return -1;
670 }
671
672#else
673 #error Platform not supported
674#endif
675
676 vrtql.success();
677
678 return sockfd;
679}
