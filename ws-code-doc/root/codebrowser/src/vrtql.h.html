21typedef enum
22{

26 VE_SYS = 10, /**< System call error */
27 VE_RT = 11, /**< Runtime error */
28 VE_MEM = 100, /**< Memory failure */
29 VE_FATAL = 200, /**< Fatal error */

61/**

105/**
106 * @brief Callback for error submission. Error submission function. Error
107 * submission takes care of recording the error in the vrtql.e member. The next
108 * step is the process the error.
109 */
110typedef int (*vrtql_error_submit_cb)(int code, cstr message, ...);
111
112/**
113 * @brief Callback for error processing. Error processing function. Error
114 * processing makes policy decisions, if any, on how to handle specific classes
115 * of errors, for example how to exit the process on fatal errors (VE_FATAL), or
116 * how to handle memory allocation errors (VE_MEM).
117 */
118typedef int (*vrtql_error_process_cb)(int code, cstr message);

215/**
216 * @brief Appends formatted data to a vrtql_buffer using printf() format.
217 *
218 * This function appends formatted data to the specified vrtql_buffer using a
219 * printf()-style format string and variable arguments. The formatted data is
220 * appended to the existing content of the buffer.
221 *
222 * @param buffer The vrtql_buffer to append the formatted data to.
223 * @param format The printf() format string specifying the format of the data to
224 * be appended.
225 * @param ... Variable arguments corresponding to the format specifier in the
226 * format string.
227 *
228 * @note The behavior of this function is similar to the standard printf()
229 * function, where the format string specifies the expected format of the
230 * data and the variable arguments provide the values to be formatted and
231 * appended to the buffer.
232 *
233 * @note The vrtql_buffer must be initialized and have sufficient capacity to
234 * hold the appended data. If the buffer capacity is exceeded, the
235 * behavior is undefined.
236 *
237 * @warning Take care to ensure the format string and variable arguments are
238 * consistent, as mismatches can lead to undefined behavior or security
239 * vulnerabilities (e.g., format string vulnerabilities).
240 *
241 * @see vrtql_buffer_init
242 * @see vrtql_buffer_append
243 */
244void vrtql_buffer_printf(vrtql_buffer* buffer, cstr format, ...);
