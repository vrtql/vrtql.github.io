88/**
89 * @brief Lock the log mutex to synchronize access to the logging functionality.
90 *
91 * This function is responsible for locking the log mutex, which ensures that
92 * only one thread can access the logging functionality at a time. It uses
93 * platform-specific synchronization mechanisms depending on the platform:
94 *
95 * - On Windows, it uses the Windows API function WaitForSingleObject() to
96 * lock the mutex.
97 * - On other platforms, it uses the pthread_mutex_lock() function to lock
98 * the mutex.
99 *
100 * If an error occurs while trying to lock the mutex, this function will
101 * submit an error message using vrtql_error_default_submit() with a
102 * corresponding error code.
103 *
104 * @see vrtql_trace_unlock()
105 */
106void vrtql_trace_lock();
107
108/**
109 * @brief Unlock the log mutex to release synchronization after accessing
110 * the logging functionality.
111 *
112 * This function is responsible for unlocking the log mutex, allowing other
113 * threads to access the logging functionality. It complements the
114 * vrtql_trace_lock() function and should be called after finishing the
115 * logging-related tasks to release the mutex.
116 *
117 * This function uses platform-specific synchronization mechanisms depending
118 * on the platform:
119 *
120 * - On Windows, it uses the Windows API function ReleaseMutex() to unlock
121 * the mutex.
122 * - On other platforms, it uses the pthread_mutex_unlock() function to
123 * unlock the mutex.
124 *
125 * If an error occurs while trying to unlock the mutex, this function will
126 * submit an error message using vrtql_error_default_submit() with a
127 * corresponding error code.
128 *
129 * @see vrtql_trace_lock()
130 */
131void vrtql_trace_unlock();

281 * @brief Appends formatted data to a vrtql_buffer using printf() format.
282 *
283 * This function appends formatted data to the specified vrtql_buffer using a
284 * printf()-style format string and variable arguments. The formatted data is
285 * appended to the existing content of the buffer.
286 *
287 * @param buffer The vrtql_buffer to append the formatted data to.
288 * @param format The printf() format string specifying the format of the data to
289 * be appended.
290 * @param ... Variable arguments corresponding to the format specifier in the
291 * format string.
292 *
293 * @note The behavior of this function is similar to the standard printf()
294 * function, where the format string specifies the expected format of the
295 * data and the variable arguments provide the values to be formatted and
296 * appended to the buffer.
297 *
298 * @note The vrtql_buffer must be initialized and have sufficient capacity to
299 * hold the appended data. If the buffer capacity is exceeded, the
300 * behavior is undefined.
301 *
302 * @warning Take care to ensure the format string and variable arguments are
303 * consistent, as mismatches can lead to undefined behavior or security
304 * vulnerabilities (e.g., format string vulnerabilities).
305 *
306 * @see vrtql_buffer_init
307 * @see vrtql_buffer_append
308 */
309void vrtql_buffer_printf(vrtql_buffer* buffer, cstr format, ...);
