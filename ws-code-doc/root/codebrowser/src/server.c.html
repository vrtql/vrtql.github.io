26/**
27 * @brief UV callback executed in response to (server->wakeup) async signal.
28 *
29 * This function handles asynchronous events in the main thread, specifically
30 * managing outgoing network I/O from worker threads back to clients. It runs
31 * within the main UV loop, within the main thread, also referred to as the
32 * networking thread.
33 *
34 * Worker threads pass data to it through a queue. The data is in the form of
35 * vws_svr_data instances. When a worker sends a vws_svr_data instance, it
36 * adds it to the queue (server->responses) and notifies (wakes up) the main UV
37 * loop (uv_run() in vrtql_svr_run()) by calling uv_async_send(server->wakeup)
38 * which in turn calls this function to check the server->responses queue. This
39 * function unloads all data in the queue and sends the data out to each
40 * respective client. It then returns control back to the main UV loop which
41 * resumes polling the network connections (blocking if there is no activity).
42 *
43 * @param handle A pointer to the uv_async_t handle that triggered the callback.
44 *
45 * @ingroup ThreadFunctions
46 */
47static void uv_thread(uv_async_t* handle);
48
49/**
50 * @brief The entry point for a worker thread.
51 *
52 * This function implements the worker thread pool. It is what each worker
53 * thread runs. It loops continuously, handling incoming data from clients,
54 * processing them and returning data back to them via the uv_thread(). It
55 * processes data by taking data (requests) from the server->requests queue,
56 * dispatching them to server->process(data) for processing, which in turn
57 * generates data (responses), sending them back to the client by putting them
58 * on the server->responses queue (processed by uv_thread()).
59 *
60 * @param arg A void pointer, typically to the server object,
61 * used to pass data into the thread.
62 *
63 * @ingroup ThreadFunctions
64 */
65static void worker_thread(void* arg);

654void worker_thread(void* arg)
655{
656 vws_tcp_svr* server = (vws_tcp_svr*)arg;
657
658 // Set thread tracing level to server.
659 vws.tracelevel = server->trace;
660
661 if (vws.tracelevel >= VT_THREAD)
662 {
663 vws.trace(VL_INFO, "worker_thread(): Starting");
664 }
665
666 while (true)
667 {
668 //> Wait for arrival
669
670 // This will put the thread to sleep on a condition variable until
671 // something arrives in queue.
672 vws_svr_data* request = queue_pop(&server->requests);
673
674 // If there's no request (null request), check the server's state
675 if (request == NULL)
676 {
677 // If server is in halting state, return
678 if (server->state == VS_HALTING)
679 {
680 if (vws.tracelevel >= VT_THREAD)
681 {
682 vws.trace(VL_INFO, "worker_thread(): Exiting");
683 }
684
685 return;
686 }
687 else
688 {
689 // If not halting, skip to the next iteration of the loop
690 continue;
691 }
692 }
693
694 server->on_data_in(request);
695 }
696}
697
698void uv_thread(uv_async_t* handle)
699{
700 vws_tcp_svr* server = (vws_tcp_svr*)handle->data;
701
702 if (server->state == VS_HALTING)
703 {
704 if (vws.tracelevel >= VT_THREAD)
705 {
706 vws.trace(VL_INFO, "uv_thread(): stop");
707 }
708
709 // Join worker threads. Worker threads must all exit before we can
710 // shutdown libuv as they must release their mutexes and condition
711 // variables first, otherwise uv_stop() will not actually stop the loop.
712 for (int i = 0; i < server->pool_size; i++)
713 {
714 uv_thread_join(&server->threads[i]);
715 }
716
717 svr_shutdown(server);
718
719 return;
720 }
721
722 while (queue_empty(&server->responses) == false)
723 {
724 vws_svr_data* data = queue_pop(&server->responses);
725
726 if ((data == NULL) || (server->responses.state != VS_RUNNING))
727 {
728 return;
729 }
730
731 if (vws_is_flag(&data->flags, VM_SVR_DATA_CLOSE))
732 {
733 // Close connection
734 svr_cnx_close(data->cnx);
735 }
736 else
737 {
738 server->on_data_out(data);
739 }
740 }
741}

812int vws_tcp_svr_run(vws_tcp_svr* server, cstr host, int port)
813{
814 if (vws.tracelevel >= VT_SERVICE)
815 {
816 vws.trace( VL_INFO,
817 "vws_tcp_svr_run(%p): Starting worker %i threads",
818 server,
819 server->pool_size );
820 }
821
822 for (int i = 0; i < server->pool_size; i++)
823 {
824 uv_thread_create(&server->threads[i], worker_thread, server);
825 }
826
827 //> Create listening socket
828
829 uv_tcp_t* socket = vws.malloc(sizeof(uv_tcp_t));
830
831 uv_tcp_init(server->loop, socket);
832 socket->data = server;
833
834 //> Bind to address
835
836 int rc;
837 struct sockaddr_in addr;
838 uv_ip4_addr(host, port, &addr);
839 rc = uv_tcp_bind(socket, (const struct sockaddr*)&addr, 0);
840
841 if (vws.tracelevel >= VT_SERVICE)
842 {
843 vws.trace( VL_INFO,
844 "vws_tcp_svr_run(%p): Bind %s:%lu",
845 server, host, port );
846 }
847
848 if (rc)
849 {
850 vws.error(VE_RT, "Bind error %s", uv_strerror(rc));
851 return -1;
852 }
853
854 //> Listen
855
856 rc = uv_listen((uv_stream_t*)socket, server->backlog, svr_on_connect);
857
858 if (rc)
859 {
860 vws.error(VE_RT, "Listen error %s", uv_strerror(rc));
861 return -1;
862 }
863
864 if (vws.tracelevel >= VT_SERVICE)
865 {
866 vws.trace( VL_INFO,
867 "vws_tcp_svr_run(%s): Listen %s:%lu",
868 server, host, port );
869 }
870
871 //> Start server
872
873 // Set state to running
874 server->state = VS_RUNNING;
875
876 if (vws.tracelevel >= VT_SERVICE)
877 {
878 vws.trace(VL_INFO, "vws_tcp_svr_run(%s): Starting uv_run()", server);
879 }
880
881 // Run UV loop. This runs indefinitely, passing network I/O and and out of
882 // system until server is shutdown by vws_tcp_svr_stop() (by external
883 // thread).
884 uv_run(server->loop, UV_RUN_DEFAULT);
885
886 //> Shutdown server
887
888 // Close the listening socket handle
889 uv_close((uv_handle_t*)socket, svr_on_close);
890
891 if (vws.tracelevel >= VT_SERVICE)
892 {
893 vws.trace(VL_INFO, "vws_tcp_svr_run(%p): Shutdown complete", server);
894 }
895
896 // Set state to halted
897 server->state = VS_HALTED;
898
899 return 0;
900}

902void vws_tcp_svr_stop(vws_tcp_svr* server)
903{
904 // Set shutdown flags
905 server->state = VS_HALTING;
906 server->requests.state = VS_HALTING;
907 server->responses.state = VS_HALTING;
908
909 // Wakeup all worker threads
910 if (vws.tracelevel >= VT_SERVICE)
911 {
912 vws.trace(VL_INFO, "vws_tcp_svr_stop(): stop worker threads");
913 }
914
915 uv_mutex_lock(&server->requests.mutex);
916 uv_cond_broadcast(&server->requests.cond);
917 uv_mutex_unlock(&server->requests.mutex);
918
919 // Wakeup the main event loop to shutdown main thread
920 if (vws.tracelevel >= VT_SERVICE)
921 {
922 vws.trace(VL_INFO, "vws_tcp_svr_stop(): stop main thread");
923 }
924
925 uv_async_send(server->wakeup);
926
927 while (server->state != VS_HALTED)
928 {
929 sleep(1);
930 }
931}

933int vws_tcp_svr_inetd_run(vws_tcp_svr* server, int sockfd)
934{
935 if (sockfd < 0)
936 {
937 vws.error(VE_RT, "Invalid server or socket descriptor provided");
938 return 1;
939 }
940
941 if (vws.tracelevel >= VT_SERVICE)
942 {
943 vws.trace( VL_INFO,
944 "vws_tcp_svr_run(%p): Starting worker %i threads",
945 server,
946 server->pool_size );
947 }
948
949 for (int i = 0; i < server->pool_size; i++)
950 {
951 uv_thread_create(&server->threads[i], worker_thread, server);
952 }
953
954 // Go into non-blocking mode as we are using poll() for socket_read() and
955 // socket_write().
956 if (vws_socket_set_nonblocking(sockfd) == false)
957 {
958 vws.error(VE_RT, "Failed to set socket to nonblocking");
959 return 1;
960 }
961
962 // Set to inetd mode
963 server->inetd_mode = 1;
964
965 // Initialize and adopt the existing socket descriptor.
966 uv_tcp_t* c = (uv_tcp_t*)vws.malloc(sizeof(uv_tcp_t));
967
968 if (uv_tcp_init(server->loop, c))
969 {
970 // Handle uv_tcp_init failure.
971 vws.error(VE_RT, "Failed to initialize new TCP handle");
972 vws.free(c);
973 return 1;
974 }
975
976 if (uv_tcp_open(c, sockfd))
977 {
978 // Handle uv_tcp_open failure.
979 vws.error(VE_RT, "Failed to adopt the socket descriptor.");
980 vws.free(c);
981 return 1;
982 }
983
984 // Associate server with this handle for callbacks.
985 c->data = server;
986
987 if (uv_read_start((uv_stream_t*)c, svr_on_realloc, svr_on_read) != 0)
988 {
989 vws.error(VE_RT, "Failed to start reading from client");
990 vws.free(c);
991 return 1;
992 }
993
994 //> Add connection to registry and initialize
995
996 vws_svr_cnx* cnx = svr_cnx_new(server, (uv_stream_t*)c);
997
998 if (svr_cnx_map_set(&server->cnxs, (uv_stream_t*)c, cnx) == false)
999 {
1000 vws.error(VE_FATAL, "Connection already registered");
1001 }
1002
1003 //> Call svr_on_connect() handler
1004
1005 server->on_connect(cnx);
1006
1007 // Now, the handle is associated with the socket and is ready to be used.
1008 // Start the libuv loop.
1009 uv_run(server->loop, UV_RUN_DEFAULT);
1010
1011 return 0;
1012}

1014void vws_tcp_svr_inetd_stop(vws_tcp_svr* server)
1015{
1016 // Set shutdown flags
1017 server->state = VS_HALTING;
1018 server->requests.state = VS_HALTING;
1019 server->responses.state = VS_HALTING;
1020
1021 // Stop the loop. We have not more I/O to deal with. We don't want the loop
1022 // to run any more for any reason.
1023 uv_stop(server->loop);
1024
1025 // Wakeup all worker threads
1026 if (vws.tracelevel >= VT_SERVICE)
1027 {
1028 vws.trace(VL_INFO, "vws_tcp_svr_inetd_stop(): stop worker threads");
1029 }
1030
1031 uv_mutex_lock(&server->requests.mutex);
1032 uv_cond_broadcast(&server->requests.cond);
1033 uv_mutex_unlock(&server->requests.mutex);
1034
1035 // Wakeup the main event loop to shutdown main thread
1036 if (vws.tracelevel >= VT_SERVICE)
1037 {
1038 vws.trace(VL_INFO, "vws_tcp_svr_inetd_stop(): stop main thread");
1039 }
1040
1041 // Wait for all threads to complete
1042 uv_thread(server->wakeup);
1043
1044 if (vws.tracelevel >= VT_SERVICE)
1045 {
1046 vws.trace(VL_INFO, "vws_tcp_svr_inetd_stop(): done");
1047 }
1048
1049 // Set state to halted
1050 server->state = VS_HALTED;
1051}

1121void tcp_svr_dtor(vws_tcp_svr* svr)
1122{
1123 //> Stop/shutdown server
1124
1125 if (svr->state == VS_RUNNING)
1126 {
1127 vws_tcp_svr_stop(svr);
1128 }
1129
1130 svr_shutdown(svr);
1131 vws.free(svr->threads);
1132
1133 // Close the server async handle
1134 uv_close((uv_handle_t*)svr->wakeup, svr_on_close);
1135
1136 //> Shutdown libuv
1137
1138 // Walk the loop to close everything
1139 uv_walk(svr->loop, on_uv_walk, NULL);
1140

1142 {
1143 // Run the loop until there are no more active handles
1144 while (uv_loop_alive(svr->loop))
1145 {
1146 uv_run(svr->loop, UV_RUN_DEFAULT);
1147 }
1148 }
1149
1150 // Free loop
1151 vws.free(svr->loop);
1152
1153 // Free connection map
1154 svr_cnx_map_clear(&svr->cnxs);
1155 sc_map_term_64v(&svr->cnxs);
1156}

1476
1477void queue_push(vws_svr_queue* queue, vws_svr_data* data)
1478{
1479 if (queue->state != VS_RUNNING)
1480 {
1481 vws.free(data);
1482 return;
1483 }
1484
1485 uv_mutex_lock(&queue->mutex);
1486
1487 while (queue->size == queue->capacity)
1488 {
1489 uv_cond_wait(&queue->cond, &queue->mutex);
1490 }
1491
1492 if (queue->state != VS_RUNNING)
1493 {
1494 uv_cond_signal(&queue->cond);
1495 uv_mutex_unlock(&queue->mutex);
1496 return;
1497 }
1498
1499 queue->buffer[queue->tail] = data;
1500 queue->tail = (queue->tail + 1) % queue->capacity;
1501 queue->size++;
1502
1503 // Signal condition variable
1504 uv_cond_signal(&queue->cond);
1505 uv_mutex_unlock(&queue->mutex);
1506}
1507
1508vws_svr_data* queue_pop(vws_svr_queue* queue)
1509{
1510 uv_mutex_lock(&queue->mutex);
1511
1512 while (queue->size == 0 && queue->state == VS_RUNNING)
1513 {
1514 uv_cond_wait(&queue->cond, &queue->mutex);
1515 }
1516
1517 if (queue->state == VS_HALTING)
1518 {
1519 uv_cond_broadcast(&queue->cond);
1520 uv_mutex_unlock(&queue->mutex);
1521
1522 return NULL;
1523 }
1524
1525 vws_svr_data* data = queue->buffer[queue->head];
1526 queue->head = (queue->head + 1) % queue->capacity;
1527 queue->size--;
1528
1529 uv_mutex_unlock(&queue->mutex);
1530
1531 return data;
1532}

1547void ws_svr_process_frame(vws_cnx* c, vws_frame* f)
1548{
1549 vws_svr_cnx* cnx = (vws_svr_cnx*)c->data;
1550
1551 switch (f->opcode)
1552 {
1553 case CLOSE_FRAME:
1554 {
1555 // Build the response frame
1556 vws_buffer* buffer = vws_generate_close_frame();
1557
1558 // Send back to cliane Send the PONG response
1559 vws_svr_data* response;
1560
1561 response = vws_svr_data_new(cnx, buffer);
1562 vws_tcp_svr_send(cnx->server, response);
1563
1564 // Free buffer
1565 vws_buffer_free(buffer);
1566
1567 // Free frame
1568 vws_frame_free(f);
1569
1570 break;
1571 }
1572
1573 case TEXT_FRAME:
1574 case BINARY_FRAME:
1575 case CONTINUATION_FRAME:
1576 {
1577 // Add to queue
1578 sc_queue_add_first(&c->queue, f);
1579
1580 break;
1581 }
1582
1583 case PING_FRAME:
1584 {
1585 // Generate the PONG response
1586 vws_buffer* buffer = vws_generate_pong_frame(f->data, f->size);
1587
1588 // Send back to cliane Send the PONG response
1589 vws_svr_data* response;
1590 response = vws_svr_data_new(cnx, buffer);
1591 vws_tcp_svr_send(cnx->server, response);
1592
1593 // Free buffer
1594 vws_buffer_free(buffer);
1595
1596 // Free frame
1597 vws_frame_free(f);
1598
1599 break;
1600 }
1601
1602 case PONG_FRAME:
1603 {
1604 // No need to send a response
1605
1606 vws_frame_free(f);
1607
1608 break;
1609 }
1610
1611 default:
1612 {
1613 // Invalid frame type
1614 vws_frame_free(f);
1615 }
1616 }
1617
1618 vws.success();
1619}

1662// Runs in uv_thread()
1663void ws_svr_client_read(vws_svr_cnx* cnx, ssize_t size, const uv_buf_t* buf)
1664{
1665 vws_tcp_svr* server = cnx->server;
1666 vws_cnx* c = (vws_cnx*)cnx->data;
1667
1668 // Add to client socket buffer
1669 vws_buffer_append(c->base.buffer, (ucstr)buf->base, size);
1670
1671 // Free libuv memory
1672 vws.free(buf->base);
1673
1674 // If we are in HTTP mode
1675 if (cnx->upgraded == false)
1676 {
1677 // Parse incoming data as HTTP request.
1678
1679 ucstr data = c->base.buffer->data;
1680 size_t size = c->base.buffer->size;
1681 ssize_t n = vws_http_msg_parse(cnx->http, (cstr)data, size);
1682
1683 // Did we get a complete request?
1684 if (cnx->http->headers_complete == true)
1685 {
1686 // Check for parsing errors
1687 enum llhttp_errno err = llhttp_get_errno(cnx->http->parser);
1688
1689 // If there was a parsing error, close connection
1690 if(err != HPE_PAUSED)
1691 {
1692 vws.error( VE_RT, "Error: %s (%s)",
1693 llhttp_errno_name(err),
1694 llhttp_get_error_reason(cnx->http->parser) );
1695
1696 // Close connection
1697 svr_cnx_close(cnx);
1698
1699 return;
1700 }
1701
1702 // Drain HTTP request data from cnx->data buffer
1703 vws_buffer_drain(c->base.buffer, n);
1704
1705 //> Generate HTTP response and send
1706
1707 vws_buffer* http = vws_buffer_new();
1708
1709 struct sc_map_str* headers = &cnx->http->headers;
1710 cstr key = vws_map_get(headers, "sec-websocket-key");
1711 cstr proto = vws_map_get(headers, "sec-websocket-protocol");
1712
1713 vws_buffer_printf(http, "HTTP/1.1 101 Switching Protocols\r\n");
1714 vws_buffer_printf(http, "Upgrade: websocket\r\n");
1715 vws_buffer_printf(http, "Connection: Upgrade\r\n");
1716
1717 cstr ac = vws_accept_key(key);
1718 vws_buffer_printf(http, "Sec-WebSocket-Accept: %s\r\n", ac);
1719 vws.free(ac);
1720
1721 vws_buffer_printf(http, "Sec-WebSocket-Version: 13\r\n");
1722 vws_buffer_printf(http, "Sec-WebSocket-Protocol: ");
1723
1724 if (proto != NULL)
1725 {
1726 vws_buffer_printf(http, "%s\r\n", proto);
1727 }
1728 else
1729 {
1730 vws_buffer_printf(http, "vrtql\r\n");
1731 }
1732
1733 vws_buffer_printf(http, "\r\n");
1734
1735 // Package up response
1736 vws_svr_data* reply;
1737 reply = vws_svr_data_new(cnx, http);
1738
1739 // Send directly out as we are in uv_thread()
1740 server->on_data_out(reply);
1741
1742 // Cleanup. Buffer data was passed to reply.
1743 vws_buffer_free(http);
1744
1745 //> Change state to WebSocket mode
1746
1747 // Set the flag that we are in WebSocket mode
1748 cnx->upgraded = true;
1749
1750 // Free HTTP request as we don't need it anymore
1751 vws_http_msg_free(cnx->http);
1752 cnx->http = NULL;
1753
1754 // Do we have any data in the socket after consuming the HTTP
1755 // request? We shouldn't but if so this is WebSocket data.
1756 if (c->base.buffer->size == 0)
1757 {
1758 // No more data in the socket buffer. Done for now.
1759 return;
1760 }
1761 }
1762 else
1763 {
1764 // No complete HTTP request yet
1765 return;
1766 }
1767
1768 // If we get here, we have a complete request and we have incoming
1769 // WebSocket data to process (c->base.buffer->size > 0)
1770 }
1771
1772 if (vws_cnx_ingress(c) > 0)
1773 {
1774 // Process as many messages as possible
1775 while (true)
1776 {
1777 // Check for a complete message
1778 vws_msg* wsm = vws_msg_pop(c);
1779
1780 if (wsm == NULL)
1781 {
1782 return;
1783 }
1784
1785 // Pass message pointer in block
1786 vws_svr_data* block;
1787 block = vws_svr_data_own(cnx, (ucstr)wsm, sizeof(vws_msg*));
1788 queue_push(&server->requests, block);
1789 }
1790 }
1791}
