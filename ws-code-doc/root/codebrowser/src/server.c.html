26/**
27 * @brief UV callback executed in response to (server->wakeup) async signal.
28 *
29 * This function handles asynchronous events in the main thread, specifically
30 * managing outgoing network I/O from worker threads back to clients. It runs
31 * within the main UV loop, within the main thread, also referred to as the
32 * networking thread.
33 *
34 * Worker threads pass data to it through a queue. The data is in the form of
35 * vrtql_svr_data instances. When a worker sends a vrtql_svr_data instance, it
36 * adds it to the queue (server->responses) and notifies (wakes up) the main UV
37 * loop (uv_run() in vrtql_svr_run()) by calling uv_async_send(server->wakeup)
38 * which in turn calls this function to check the server->responses queue. This
39 * function unloads all data in the queue and sends the data out to each
40 * respective client. It then returns control back to the main UV loop which
41 * resumes polling the network connections (blocking if there is no activity).
42 *
43 * @param handle A pointer to the uv_async_t handle that triggered the callback.
44 *
45 * @ingroup ThreadFunctions
46 */
47static void uv_thread(uv_async_t* handle);
48
49/**
50 * @brief The entry point for a worker thread.
51 *
52 * This function implements the worker thread pool. It is what each worker
53 * thread runs. It loops continuously, handling incoming data from clients,
54 * processing them and returning data back to them via the uv_thread(). It
55 * processes data by taking data (requests) from the server->requests queue,
56 * dispatching them to server->process(data) for processing, which in turn
57 * generates data (responses), sending them back to the client by putting them
58 * on the server->responses queue (processed by uv_thread()).
59 *
60 * @param arg A void pointer, typically to the server object,
61 * used to pass data into the thread.
62 *
63 * @ingroup ThreadFunctions
64 */
65static void worker_thread(void* arg);

570 * @brief Queue functions which bridge the network thread and workers
571 *
572 * The network thread and worker threads pass data via queues. These queues
573 * contain vrtql_svr_data instances. There is a requests queue which passes
574 * data from the network thread to workers for processing. There is a response

654void worker_thread(void* arg)
655{
656 vrtql_svr* server = (vrtql_svr*)arg;
657
658 // Set thread tracing level to server.
659 vrtql.tracelevel = server->trace;
660
661 if (vrtql.tracelevel >= VT_THREAD)
662 {
663 vrtql.trace(VL_INFO, "worker_thread(): Starting");
664 }
665
666 while (true)
667 {
668 //> Wait for arrival
669
670 // This will put the thread to sleep on a condition variable until
671 // something arrives in queue.
672 vrtql_svr_data* request = queue_pop(&server->requests);
673
674 // If there's no request (null request), check the server's state
675 if (request == NULL)
676 {
677 // If server is in halting state, return
678 if (server->state == VS_HALTING)
679 {
680 if (vrtql.tracelevel >= VT_THREAD)
681 {
682 vrtql.trace(VL_INFO, "worker_thread(): Exiting");
683 }
684
685 return;
686 }
687 else
688 {
689 // If not halting, skip to the next iteration of the loop
690 continue;
691 }
692 }
693
694 server->on_data_in(request);
695 }
696}
697
698void uv_thread(uv_async_t* handle)
699{
700 vrtql_svr* server = (vrtql_svr*)handle->data;
701
702 if (server->state == VS_HALTING)
703 {
704 if (vrtql.tracelevel >= VT_THREAD)
705 {
706 vrtql.trace(VL_INFO, "uv_thread(): stop");
707 }
708
709 // Join worker threads. Worker threads must all exit before we can
710 // shutdown libuv as they must release their mutexes and condition
711 // variables first, otherwise uv_stop() will not actually stop the loop.
712 for (int i = 0; i < server->pool_size; i++)
713 {
714 uv_thread_join(&server->threads[i]);
715 }
716
717 svr_shutdown(server);
718
719 return;
720 }
721
722 while (queue_empty(&server->responses) == false)
723 {
724 vrtql_svr_data* data = queue_pop(&server->responses);
725
726 if ((data == NULL) || (server->responses.state != VS_RUNNING))
727 {
728 return;
729 }
730
731 if (vrtql_is_flag(&data->flags, VM_SVR_DATA_CLOSE))
732 {
733 // Close connection
734 //uv_close((uv_handle_t*)data->cnx->handle, svr_on_close);
735 svr_cnx_close(data->cnx);
736 }
737 else
738 {
739 server->on_data_out(data);
740 }
741 }
742}

748vrtql_svr_data* vrtql_svr_data_new(vrtql_svr_cnx* c, vrtql_buffer* buffer)
749{
750 vrtql_svr_data* item = vrtql_svr_data_own(c, buffer->data, buffer->size);
751
752 // We take ownership of buffer data so we clear the buffer.
753 buffer->data = NULL;
754 buffer->size = 0;
755
756 return item;
757}
758
759vrtql_svr_data* vrtql_svr_data_own(vrtql_svr_cnx* c, ucstr data, size_t size)
760{
761 vrtql_svr_data* item;
762 item = (vrtql_svr_data*)vrtql.malloc(sizeof(vrtql_svr_data));
763
764 item->cnx = c;
765 item->size = size;
766 item->data = data;
767 item->flags = 0;
768
769 return item;
770}

811int vrtql_svr_run(vrtql_svr* server, cstr host, int port)
812{
813 if (vrtql.tracelevel >= VT_SERVICE)
814 {
815 vrtql.trace( VL_INFO,
816 "vrtql_svr_run(%p): Starting worker %i threads",
817 server,
818 server->pool_size );
819 }
820
821 for (int i = 0; i < server->pool_size; i++)
822 {
823 uv_thread_create(&server->threads[i], worker_thread, server);
824 }
825
826 //> Create listening socket
827
828 uv_tcp_t* socket = vrtql.malloc(sizeof(uv_tcp_t));
829
830 uv_tcp_init(server->loop, socket);
831 socket->data = server;
832
833 //> Bind to address
834
835 int rc;
836 struct sockaddr_in addr;
837 uv_ip4_addr(host, port, &addr);
838 rc = uv_tcp_bind(socket, (const struct sockaddr*)&addr, 0);
839
840 if (vrtql.tracelevel >= VT_SERVICE)
841 {
842 vrtql.trace( VL_INFO,
843 "vrtql_svr_run(%p): Bind %s:%lu",
844 server, host, port);
845 }
846
847 if (rc)
848 {
849 vrtql.error(VE_RT, "Bind error %s", uv_strerror(rc));
850 return -1;
851 }
852
853 //> Listen
854
855 rc = uv_listen((uv_stream_t*)socket, server->backlog, svr_on_connect);
856
857 if (rc)
858 {
859 vrtql.error(VE_RT, "Listen error %s", uv_strerror(rc));
860 return -1;
861 }
862
863 if (vrtql.tracelevel >= VT_SERVICE)
864 {
865 vrtql.trace( VL_INFO,
866 "vrtql_svr_run(%s): Listen %s:%lu",
867 server, host, port );
868 }
869
870 //> Start server
871
872 // Set state to running
873 server->state = VS_RUNNING;
874
875 if (vrtql.tracelevel >= VT_SERVICE)
876 {
877 vrtql.trace(VL_INFO, "vrtql_svr_run(%s): Starting uv_run()", server);
878 }
879
880 // Run UV loop. This runs indefinitely, passing network I/O and and out of
881 // system until server is shutdown by vrtql_svr_stop() (by external thread).
882 uv_run(server->loop, UV_RUN_DEFAULT);
883
884 //> Shutdown server
885
886 // Close the listening socket handle
887 uv_close((uv_handle_t*)socket, svr_on_close);
888
889 if (vrtql.tracelevel >= VT_SERVICE)
890 {
891 vrtql.trace(VL_INFO, "vrtql_svr_run(%p): Shutdown complete", server);
892 }
893
894 // Set state to halted
895 server->state = VS_HALTED;
896
897 return 0;
898}
899
900void vrtql_svr_stop(vrtql_svr* server)
901{
902 // Set shutdown flags
903 server->state = VS_HALTING;
904 server->requests.state = VS_HALTING;
905 server->responses.state = VS_HALTING;
906
907 // Wakeup all worker threads
908 if (vrtql.tracelevel >= VT_SERVICE)
909 {
910 vrtql.trace(VL_INFO, "vrtql_svr_stop(): stop worker threads");
911 }
912
913 uv_mutex_lock(&server->requests.mutex);
914 uv_cond_broadcast(&server->requests.cond);
915 uv_mutex_unlock(&server->requests.mutex);
916
917 // Wakeup the main event loop to shutdown main thread
918 if (vrtql.tracelevel >= VT_SERVICE)
919 {
920 vrtql.trace(VL_INFO, "vrtql_svr_stop(): stop main thread");
921 }
922
923 uv_async_send(server->wakeup);
924
925 while (server->state != VS_HALTED)
926 {
927 sleep(1);
928 }
929}

971void on_uv_close(uv_handle_t* handle)
972{
973 if (handle != NULL)
974 {
975 // As a policy we don't put heap data on handle->data. If that is ever
976 // done then it must be freed in the appropriate place. It is ignored
977 // here because it's impossible to tell what it is by the very nature of
978 // uv_handle_t. We will generate a warning however.
979 if (handle->data != NULL)
980 {
981 vrtql.trace( VL_WARN,
982 "on_uv_close(): libuv resource not properly freed: %p",

988void on_uv_walk(uv_handle_t* handle, void* arg)
989{
990 vrtql.trace(VL_INFO, "on_uv_walk(): %p", (void*)handle);
991
992 // If this handle has not been closed, it should have been. Nevertheless we
993 // will make an attempt to close it.
994
995 if (uv_is_closing(handle) == 0)
996 {
997 uv_close(handle, (uv_close_cb)on_uv_close);
998 }

1001void svr_dtor(vrtql_svr* server)
1002{
1003 //> Stop/shutdown server
1004
1005 if (server->state == VS_RUNNING)
1006 {
1007 vrtql_svr_stop(server);
1008 }
1009
1010 svr_shutdown(server);
1011 free(server->threads);
1012
1013 // Close the server async handle
1014 uv_close((uv_handle_t*)server->wakeup, svr_on_close);
1015
1016 //> Shutdown libuv
1017
1018 // Walk the loop to close everything
1019 uv_walk(server->loop, on_uv_walk, NULL);
1020
1021 while (uv_loop_close(server->loop))
1022 {
1023 // Run the loop until there are no more active handles
1024 while (uv_loop_alive(server->loop))
1025 {
1026 uv_run(server->loop, UV_RUN_DEFAULT);
1027 }
1028 }
1029
1030 // Free loop
1031 free(server->loop);
1032
1033 //> Free connection map
1034
1035 svr_cnx_map_clear(&server->cnxs);
1036 sc_map_term_64v(&server->cnxs);
1037}

1213void svr_on_connect(uv_stream_t* socket, int status)
1214{
1215 vrtql_svr* server = (vrtql_svr*) socket->data;
1216
1217 if (status < 0)
1218 {
1219 cstr e = uv_strerror(status);
1220 vrtql.error(VE_RT, "Error in connection callback: %s", e);
1221 return;
1222 }
1223
1224 uv_tcp_t* client = (uv_tcp_t*)vrtql.malloc(sizeof(uv_tcp_t));
1225
1226 if (client == NULL)
1227 {
1228 vrtql.error(VE_RT, "Failed to allocate memory for client");
1229 return;
1230 }
1231
1232 client->data = server;
1233
1234 if (uv_tcp_init(server->loop, client) != 0)
1235 {
1236 vrtql.error(VE_RT, "Failed to initialize client");
1237 return;
1238 }
1239
1240 if (uv_accept(socket, (uv_stream_t*)client) == 0)
1241 {
1242 if (uv_read_start((uv_stream_t*)client, svr_on_realloc, svr_on_read) != 0)
1243 {
1244 vrtql.error(VE_RT, "Failed to start reading from client");
1245 return;
1246 }
1247 }
1248 else
1249 {
1250 uv_close((uv_handle_t*)client, svr_on_close);
1251 }
1252
1253 //> Add connection to registry and initialize
1254
1255 vrtql_svr_cnx* cnx = svr_cnx_new(server, (uv_stream_t*)client);
1256
1257 if (svr_cnx_map_set(&server->cnxs, (uv_stream_t*)client, cnx) == false)
1258 {
1259 vrtql.error(VE_FATAL, "Connection already registered");
1260 }
1261
1262 //> Call svr_on_connect() handler
1263
1264 server->on_connect(cnx);
1265}

1330 queue->size = 0;

1345 {
1346 free(queue->name);
1347 uv_mutex_destroy(&queue->mutex);
1348 uv_cond_destroy(&queue->cond);

1351 queue->state = VS_HALTED;
1352 }

1354
1355void queue_push(vrtql_svr_queue* queue, vrtql_svr_data* data)
1356{
1357 if (queue->state != VS_RUNNING)
1358 {
1359 free(data);
1360 return;
1361 }
1362
1363 uv_mutex_lock(&queue->mutex);
1364
1365 while (queue->size == queue->capacity)
1366 {
1367 uv_cond_wait(&queue->cond, &queue->mutex);
1368 }
1369
1370 if (queue->state != VS_RUNNING)
1371 {
1372 uv_cond_signal(&queue->cond);
1373 uv_mutex_unlock(&queue->mutex);
1374 return;
1375 }
1376
1377 queue->buffer[queue->tail] = data;
1378 queue->tail = (queue->tail + 1) % queue->capacity;
1379 queue->size++;
1380
1381 // Signal condition variable
1382 uv_cond_signal(&queue->cond);
1383 uv_mutex_unlock(&queue->mutex);
1384}
1385
1386vrtql_svr_data* queue_pop(vrtql_svr_queue* queue)
1387{
1388 uv_mutex_lock(&queue->mutex);
1389
1390 while (queue->size == 0 && queue->state == VS_RUNNING)
1391 {
1392 uv_cond_wait(&queue->cond, &queue->mutex);
1393 }
1394
1395 if (queue->state == VS_HALTING)
1396 {
1397 uv_cond_broadcast(&queue->cond);
1398 uv_mutex_unlock(&queue->mutex);
1399
1400 return NULL;
1401 }
1402
1403 vrtql_svr_data* data = queue->buffer[queue->head];
1404 queue->head = (queue->head + 1) % queue->capacity;
1405 queue->size--;
1406
1407 uv_mutex_unlock(&queue->mutex);
1408
1409 return data;
1410}
1411
1412bool queue_empty(vrtql_svr_queue* queue)
1413{
1414 uv_mutex_lock(&queue->mutex);
1415 bool empty = (queue->size == 0);
1416 uv_mutex_unlock(&queue->mutex);
1417
1418 return empty;
1419}

1421//------------------------------------------------------------------------------
1422// Pure WebSocket Server
1423//------------------------------------------------------------------------------
1424
1425void ws_svr_process_frame(vws_cnx* c, vws_frame* f)
1426{
1427 vrtql_svr_cnx* cnx = (vrtql_svr_cnx*)c->data;
1428
1429 switch (f->opcode)
1430 {
1431 case CLOSE_FRAME:
1432 {
1433 // Build the response frame
1434 vrtql_buffer* buffer = vws_generate_close_frame();
1435
1436 // Send back to cliane Send the PONG response
1437 vrtql_svr_data* response;
1438
1439 response = vrtql_svr_data_new(cnx, buffer);
1440 vrtql_svr_send(cnx->server, response);
1441
1442 // Free buffer
1443 vrtql_buffer_free(buffer);
1444
1445 // Free frame
1446 vws_frame_free(f);
1447
1448 break;
1449 }
1450
1451 case TEXT_FRAME:
1452 case BINARY_FRAME:
1453 case CONTINUATION_FRAME:
1454 {
1455 // Add to queue
1456 sc_queue_add_first(&c->queue, f);
1457
1458 break;
1459 }
1460
1461 case PING_FRAME:
1462 {
1463 // Generate the PONG response
1464 vrtql_buffer* buffer = vws_generate_pong_frame(f->data, f->size);
1465
1466 // Send back to cliane Send the PONG response
1467 vrtql_svr_data* response;
1468 response = vrtql_svr_data_new(cnx, buffer);
1469 vrtql_svr_send(cnx->server, response);
1470
1471 // Free buffer
1472 vrtql_buffer_free(buffer);
1473
1474 // Free frame
1475 vws_frame_free(f);
1476
1477 break;
1478 }
1479
1480 case PONG_FRAME:
1481 {
1482 // No need to send a response
1483
1484 vws_frame_free(f);
1485
1486 break;
1487 }
1488
1489 default:
1490 {
1491 // Invalid frame type
1492 vws_frame_free(f);
1493 }
1494 }
1495
1496 vrtql.success();
1497}

1541void ws_svr_client_read(vrtql_svr_cnx* cnx, ssize_t size, const uv_buf_t* buf)
1542{
1543 vrtql_svr* server = cnx->server;
1544 vws_cnx* c = (vws_cnx*)cnx->data;
1545
1546 // Add to client socket buffer
1547 vrtql_buffer_append(c->base.buffer, buf->base, size);
1548
1549 // Free libuv memory
1550 free(buf->base);
1551
1552 // If we are in HTTP mode
1553 if (cnx->upgraded == false)
1554 {
1555 // Parse incoming data as HTTP request.
1556
1557 cstr data = c->base.buffer->data;
1558 size_t size = c->base.buffer->size;
1559 ssize_t n = vrtql_http_msg_parse(cnx->http, data, size);
1560
1561 // Did we get a complete request?
1562 if (cnx->http->complete == true)
1563 {
1564 // Check for parsing errors
1565 enum http_errno err = HTTP_PARSER_ERRNO(cnx->http->parser);
1566
1567 // If there was a parsing error, close connection
1568 if(err != HPE_OK)
1569 {
1570 vrtql.error(VE_RT, "Error: %s (%s)",
1571 http_errno_name(err),
1572 http_errno_description(err) );
1573
1574 // Close connection
1575 svr_cnx_close(cnx);
1576
1577 return;
1578 }
1579
1580 // Drain HTTP request data from cnx->data buffer
1581 vrtql_buffer_drain(c->base.buffer, n);
1582
1583 //> Generate HTTP response and send
1584
1585 vrtql_buffer* http = vrtql_buffer_new();
1586
1587 struct sc_map_str* headers = &cnx->http->headers;
1588 cstr key = vrtql_map_get(headers, "sec-websocket-key");
1589 cstr proto = vrtql_map_get(headers, "sec-websocket-protocol");
1590
1591 vrtql_buffer_printf(http, "HTTP/1.1 101 Switching Protocols\r\n");
1592 vrtql_buffer_printf(http, "Upgrade: websocket\r\n");
1593 vrtql_buffer_printf(http, "Connection: Upgrade\r\n");
1594
1595 cstr ac = vws_accept_key(key);
1596 vrtql_buffer_printf(http, "Sec-WebSocket-Accept: %s\r\n", ac);
1597 free(ac);
1598
1599 vrtql_buffer_printf(http, "Sec-WebSocket-Version: 13\r\n");
1600
1601 vrtql_buffer_printf(http, "Sec-WebSocket-Protocol: ");
1602
1603 if (proto != NULL)
1604 {
1605 vrtql_buffer_printf(http, "%s\r\n", proto);
1606 }
1607 else
1608 {
1609 vrtql_buffer_printf(http, "vrtql\r\n");
1610 }
1611
1612 vrtql_buffer_printf(http, "\r\n");
1613
1614 // Package up response
1615 vrtql_svr_data* reply;
1616 reply = vrtql_svr_data_new(cnx, http);
1617
1618 // Send directly out as we are in uv_thread()
1619 server->on_data_out(reply);
1620
1621 // Cleanup. Buffer data was passed to reply.
1622 vrtql_buffer_free(http);
1623
1624 //> Change state to WebSocket mode
1625
1626 // Set the flag that we are in WebSocket mode
1627 cnx->upgraded = true;
1628
1629 // Free HTTP request as we don't need it anymore
1630 vrtql_http_msg_free(cnx->http);
1631 cnx->http = NULL;
1632
1633 // Do we have any data in the socket after consuming the HTTP
1634 // request? We shouldn't but if so this is WebSocket data.
1635 if (c->base.buffer->size == 0)
1636 {
1637 // No more data in the socket buffer. Done for now.
1638 return;
1639 }
1640 }
1641 else
1642 {
1643 // No complete HTTP request yet
1644 return;
1645 }
1646
1647 // If we get here, we have a complete request and we have incoming
1648 // WebSocket data to process (c->base.buffer->size > 0)
1649 }
1650
1651 if (vws_cnx_ingress(c) > 0)
1652 {
1653 // Process as many messages as possible
1654 while (true)
1655 {
1656 // Check for a complete message
1657 vws_msg* wsm = vws_msg_pop(c);
1658
1659 if (wsm == NULL)
1660 {
1661 return;
1662 }
1663
1664 // Pass message pointer in block
1665 vrtql_svr_data* block;
1666 block = vrtql_svr_data_own(cnx, (ucstr)wsm, sizeof(vws_msg*));
1667 queue_push(&server->requests, block);
1668 }
1669 }
1670}

1672// Runs in worker_thread()
1673void ws_svr_client_data_in(vrtql_svr_data* block)
1674{
1675 //> Append data to connection buffer
1676
1677 vrtql_svr_cnx* cnx = block->cnx;
1678 vrtql_ws_svr* server = (vrtql_ws_svr*)cnx->server;
1679 vws_cnx* c = (vws_cnx*)cnx->data;
1680
1681 // Data simply contains a pointer to a websocket message
1682 vws_msg* wsm = (vws_msg*)block->data;
1683
1684 // Free incoming data
1685 block->data = NULL;
1686 block->size = 0;
1687 vrtql_svr_data_free(block);
1688
1689 //> Process connection buffer data for complete messages
1690 server->on_msg_in(cnx, wsm);
1691}

1693void ws_svr_client_data_out( vrtql_svr_cnx* cnx,
1694 vrtql_buffer* buffer,
1695 unsigned char opcode )
1696{
1697 // Create a binary websocket frame containing message
1698 vws_frame* frame;
1699 cstr data = buffer->data;
1700 size_t size = buffer->size;
1701 frame = vws_frame_new(data, size, opcode);
1702
1703 // This frame is from server to we don't mask it
1704 frame->mask = 0;
1705
1706 // Serialize frame: frame is freed by function
1707 vrtql_buffer* fdata = vws_serialize(frame);
1708
1709 // Pack frame binary into queue data
1710 vrtql_svr_data* response;
1711 response = vrtql_svr_data_new(cnx, fdata);
1712
1713 // Queue the data to uv_thread() to send out on wire
1714 vrtql_svr_send(cnx->server, response);
1715
1716 // Free buffers
1717 vrtql_buffer_free(fdata);
1718}

1792void msg_svr_client_ws_msg_in(vrtql_svr_cnx* cnx, vws_msg* wsm)
1793{

1799
1800 if (vrtql_msg_deserialize(msg, data, size) == false)
1801 {

1803
