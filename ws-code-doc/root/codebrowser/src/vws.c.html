129void vws_trace(vws_log_level_t level, const char* format, ...)
130{
131 if (level < 0 || level >= VL_LEVEL_COUNT)
132 {
133 vws_error_default_submit(VE_WARN, "Invalid log level");
134 return;
135 }
136
137 time_t raw_time;
138 struct tm time_info;
139 char stamp[20];
140
141 time(&raw_time);
142
143#ifdef __windows__
144 // Windows implementation using localtime_s
145 if (localtime_s(&time_info, &raw_time) != 0)
146 {
147 vws_error_default_submit(VE_SYS, "localtime_s failed");
148 return;
149 }
150#else
151 // Non-Windows implementation using localtime_r
152 if (localtime_r(&raw_time, &time_info) == NULL)
153 {
154 vws_error_default_submit(VE_SYS, "localtime_r failed");
155 return;
156 }
157#endif
158
159 if (strftime(stamp, sizeof(stamp), "%Y-%m-%d %H:%M:%S", &time_info) == 0)
160 {
161 vws_error_default_submit(VE_SYS, "strftime returned 0");
162 return;
163 }
164
165 const char* color_code = log_level_infos[level].color;
166 const char* level_name = log_level_infos[level].level;
167
168#if defined(__windows__)
169
170 DWORD tid = GetCurrentThreadId();
171
172 // Windows implementation using Windows API for thread synchronization
173 if (WaitForSingleObject(log_mutex, INFINITE) != WAIT_OBJECT_0)
174 {
175 vws_error_default_submit(VE_SYS, "WaitForSingleObject failed");
176 return;
177 }
178
179#else
180
181 pthread_t tid = pthread_self();
182
183 if (pthread_mutex_lock(&log_mutex) != 0)
184 {
185 vws_error_default_submit(VE_SYS, "pthread_mutex_lock failed");
186 return;
187 }
188
189 pthread_cleanup_push(unlock_mutex, &log_mutex);
190
191#endif
192
193 fprintf(stderr, "%s[%s] [%lu] [%s]%s ",
194 color_code,
195 stamp,
196#if defined(__windows__)
197 tid,
198#else
199 (unsigned long)tid,
200#endif
201 level_name,
202 ANSI_COLOR_RESET);
203
204 va_list args;
205 va_start(args, format);
206 vfprintf(stderr, format, args);
207 va_end(args);
208
209 fprintf(stderr, "\n");
210 fflush(stderr);
211
212#if defined(__windows__)
213
214 // Windows implementation using Windows API for thread synchronization
215 if (!ReleaseMutex(log_mutex))
216 {
217 vws_error_default_submit(VE_SYS, "ReleaseMutex failed");
218 }
219
220#else
221
222 pthread_cleanup_pop(1);
223
224#endif
225}

329int vws_error_default_submit(int code, cstr format, ...)
330{
331 va_list args, args_copy;
332 va_start(args, format);
333
334 va_copy(args_copy, args);
335
336 // Determine the length of the formatted string
337 int length = vsnprintf(NULL, 0, format, args_copy);
338
339 va_end(args_copy); // End args_copy. We're done with it now.
340
341 // Allocate a buffer for the formatted string
342 char* buffer = malloc(length + 1);
343
344 // Format the string into the buffer
345 vsnprintf(buffer, length + 1, format, args);
346
347 // Set
348 vws_set_error(code, buffer);
349
350 // Process
351 vws.process_error(code, buffer);
352
353 // Cleanup
354 vws.free(buffer);
355 va_end(args);
356
357 return 0;
358}

360int vws_error_default_process(int code, cstr message)
361{
362 if (vws.tracelevel >= 1)
363 {
364 switch (code)
365 {
366 case VE_WARN:
367 {
368 vws.trace(VL_WARN, "%s", message);
369 break;
370 }
371
372 case VE_TIMEOUT:
373 {
374 vws.trace(VL_WARN, "timeout: %s", message);
375 break;
376 }
377
378 case VE_SOCKET:
379 {
380 vws.trace(VL_WARN, "disconnect: %s", message);
381 break;
382 }
383
384 case VE_SYS:
385 case VE_RT:
386 {
387 vws.trace(VL_INFO, "error %i: %s", message);
388 break;
389 }
390
391 case VE_MEM:
392 case VE_FATAL:
393 {
394 vws.trace(VL_ERROR, "fatal %i: %s", code, message);
395 break;
396 }
397
398 default:
399 {
400 vws.trace(VL_INFO, "no error");
401 }
402 }
403 }
404
405 switch (code)
406 {
407 case VE_MEM:
408 {
409 break;
410 }
411
412 case VE_FATAL:
413 {
414 exit(1);
415 }
416
417 default:
418 {
419 if (message != NULL)
420 {
421
422 }
423 }
424 }
425
426 return 0;
427}

527void vws_buffer_append(vws_buffer* buffer, ucstr data, size_t size)
528{
529 if (buffer == NULL || data == NULL)
530 {
531 return;
532 }
533
534 size_t total_size = buffer->size + size;
535
536 if (total_size > buffer->allocated)
537 {
538 buffer->allocated = total_size * 1.5;
539
540 ucstr mem;
541 if (buffer->data == NULL)
542 {
543 mem = (ucstr)vws.malloc(buffer->allocated);
544 }
545 else
546 {
547 mem = (ucstr)vws.realloc(buffer->data, buffer->allocated);
548 }
549
550 buffer->data = mem;
551 }
552
553 memcpy(buffer->data + buffer->size, data, size);
554 buffer->size = total_size;
555}
556
557void vws_buffer_drain(vws_buffer* buffer, size_t size)
558{
559 if (buffer == NULL || buffer->data == NULL)
560 {
561 return;
562 }
563
564 if (size >= buffer->size)
565 {
566 // When size >= buffer->size, clear the whole buffer
567 vws_buffer_clear(buffer);
568 }
569 else
570 {
571 memmove(buffer->data, buffer->data + size, buffer->size - size);
572 buffer->size -= size;
573 buffer->data[buffer->size] = 0;
574 }
575}

642char* vws_generate_uuid()
643{
644 unsigned char uuid[16];
645
646 if (RAND_bytes(uuid, sizeof(uuid)) != 1)
647 {
648 return NULL;
649 }
650
651 // Set the version (4) and variant bits
652 uuid[6] = (uuid[6] & 0x0F) | 0x40;
653 uuid[8] = (uuid[8] & 0x3F) | 0x80;
654
655 // Format the UUID as a string
656 char* encoded_uuid = vws_base64_encode(uuid, sizeof(uuid));
657
658 if (encoded_uuid == NULL)
659 {
660 return NULL;
661 }
662
663 // Remove padding characters and dashes
664 size_t encoded_uuid_length = strlen(encoded_uuid);
665 for (size_t i = 0; i < encoded_uuid_length; i++)
666 {
667 if ( encoded_uuid[i] == '=' ||
668 encoded_uuid[i] == '\n' ||
669 encoded_uuid[i] == '\r' ||
670 encoded_uuid[i] == '-' )
671 {
672 encoded_uuid[i] = '_';
673 }
674 }
675
676 return encoded_uuid;
677}

683char* vws_base64_encode(const unsigned char* data, size_t length)
684{
685 BIO* bio = BIO_new(BIO_s_mem());
686 BIO* base64 = BIO_new(BIO_f_base64());
687
688 BIO_set_flags(base64, BIO_FLAGS_BASE64_NO_NL);
689 BIO_push(base64, bio);
690 BIO_write(base64, data, length);
691 BIO_flush(base64);
692
693 BUF_MEM* ptr;
694 BIO_get_mem_ptr(base64, &ptr);
695 size_t output_length = ptr->length;
696 char* encoded_data = (char*)vws.malloc(output_length + 1);
697
698 memcpy(encoded_data, ptr->data, output_length);
699 encoded_data[output_length] = '\0';
700
701 BIO_free_all(base64);
702
703 return encoded_data;
704}
705
706unsigned char* vws_base64_decode(const char* data, size_t* size)
707{
708 BIO* bio = BIO_new_mem_buf(data, -1);
709 BIO* base64 = BIO_new(BIO_f_base64());
710
711 BIO_set_flags(base64, BIO_FLAGS_BASE64_NO_NL);
712 BIO_push(base64, bio);
713
714 // Determine the size of the decoded data
715 size_t encoded_length = strlen(data);
716 size_t decoded_length = (encoded_length * 3) / 4; // Rough estimate
717 unsigned char* decoded_data = (unsigned char*)vws.malloc(decoded_length);
718
719 // Decode the base64 data
720 *size = BIO_read(base64, decoded_data, encoded_length);
721
722 BIO_free_all(base64);
723
724 return decoded_data;
725}

731/**
732 * @brief Sleeps for the specified number of milliseconds.
733 *
734 * This function provides a platform-independent way to sleep for a specific
735 * duration in milliseconds. The behavior is similar to the standard `sleep`
736 * function, but with millisecond precision.
737 *
738 * @param ms The number of milliseconds to sleep.
739 *
740 * @note This function may not be available on all platforms. Make sure to
741 * include the necessary headers and handle any compilation errors or
742 * warnings specific to your environment.
743 */
744void vws_msleep(unsigned int ms)
745{
746#if defined(__windows__) || defined(_WIN64)
747
748 Sleep(ms);
749
750#elif defined(__linux__)
751
752 usleep(ms * 1000);
753
754#elif defined(__bsd__)
755
756 usleep(ms * 1000);
757
758#elif defined(__sunos__)
759
760 struct timespec ts;
761 ts.tv_sec = ms / 1000;
762 ts.tv_nsec = (ms % 1000) * 1000000;
763 nanosleep(&ts, NULL);
764
765#else
766
767#error "Unsupported platform. msleep is not implemented for this platform."
768
769#endif
770}
