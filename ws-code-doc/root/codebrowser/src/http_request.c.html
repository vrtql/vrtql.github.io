75vrtql_http_req* vrtql_http_req_new()
76{
77 vrtql_http_req* req = vrtql.malloc(sizeof(vrtql_http_req));
78 req->parser = vrtql.malloc(sizeof(http_parser));
79 req->settings = vrtql.malloc(sizeof(http_parser_settings));
80 req->parser->data = req;
81 req->url = vrtql_buffer_new();
82 req->body = vrtql_buffer_new();
83 req->field = vrtql_buffer_new();
84 req->value = vrtql_buffer_new();
85 req->complete = false;
86
87 http_parser_init(req->parser, HTTP_REQUEST);
88 http_parser_settings_init(req->settings);
89
90 http_parser_settings* s = req->settings;
91 s->on_message_begin = on_message_begin;
92 s->on_headers_complete = on_headers_complete;
93 s->on_message_complete = on_message_complete;
94 s->on_url = on_url;
95 s->on_header_field = on_header_field;
96 s->on_header_value = on_header_value;
97 s->on_body = on_body;
98
99 sc_map_init_str(&req->headers, 0, 0);
100
101 return req;
102}

104void vrtql_http_req_free(vrtql_http_req* req)
105{
106 if (req == NULL)
107 {
108 return;
109 }
110
111 cstr key; cstr value;
112 sc_map_foreach(&req->headers, key, value)
113 {
114 free(key);
115 free(value);
116 }
117
118 sc_map_term_str(&req->headers);
119 vrtql_buffer_free(req->url);
120 vrtql_buffer_free(req->body);
121 vrtql_buffer_free(req->field);
122 vrtql_buffer_free(req->value);
123 free(req->parser);
124 free(req->settings);
125 free(req);
126}

154/**
155 * @brief Converts a string to lowercase.
156 * @param s The string to convert.
157 * @return The converted lowercase string.

159char* lcase(char* s)
160{
161 for(int i = 0; s[i]; i++)
162 {
163 s[i] = tolower(s[i]);
164 }

169int on_header_field(http_parser* p, cstr at, size_t l, int intr)
170{
171 vrtql_http_req* req = p->data;
172
173 if (req->value->size != 0)
174 {
175 // We've got a complete field-value pair.
176
177 // Null-terminate
178 vrtql_buffer_append(req->field, "\0", 1);
179 vrtql_buffer_append(req->value, "\0", 1);
180
181 cstr field = lcase(req->field->data);
182 cstr data = req->value->data;
183 vrtql_map_set(&req->headers, field, data);
184
185 // Reset for the next field-value pair.
186 vrtql_buffer_clear(req->field);
187 vrtql_buffer_clear(req->value);
188 }
189
190 // Start new field
191 vrtql_buffer_append(req->field, at, l);
192
193 return 0;
194}
