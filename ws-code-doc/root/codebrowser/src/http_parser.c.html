81/* Run the notify callback FOR, returning ER if it fails */
82#define CALLBACK_NOTIFY_(FOR, ER) \
83do { \
84 assert(HTTP_PARSER_ERRNO(parser) == HPE_OK); \
85 \
86 if (LIKELY(settings->on_##FOR)) { \
87 parser->state = CURRENT_STATE(); \
88 if (UNLIKELY(0 != settings->on_##FOR(parser))) { \
89 SET_ERRNO(HPE_CB_##FOR); \
90 } \
91 UPDATE_STATE(parser->state); \
92 \
93 /* We either errored above or got paused; get out */ \
94 if (UNLIKELY(HTTP_PARSER_ERRNO(parser) != HPE_OK)) { \
95 return (ER); \
96 } \
97 } \

207/* Don't allow the total size of the HTTP headers (including the status
208 * line) to exceed max_header_size. This check is here to protect
209 * embedders against denial-of-service attacks where the attacker feeds
210 * us a never-ending header that the embedder keeps buffering.
211 *
212 * This check is arguably the responsibility of embedders but we're doing
213 * it on the embedder's behalf because most won't bother and this way we
214 * make the web a little safer. max_header_size is still far bigger
215 * than any reasonable request or response so this should never affect
216 * day-to-day operation.
217 */
218#define COUNT_HEADER_SIZE(V) \
219do { \
220 nread += (uint32_t)(V); \
221 if (UNLIKELY(nread > max_header_size)) { \
222 SET_ERRNO(HPE_HEADER_OVERFLOW); \
223 goto error; \
224 } \
225} while (0)

546/* Our URL parser.
547 *
548 * This is designed to be shared by http_parser_execute() for URL validation,
549 * hence it has a state transition + byte-for-byte interface. In addition, it
550 * is meant to be embedded in http_parser_parse_url(), which does the dirty
551 * work of turning state transitions URL components for its API.
552 *
553 * This function should only be invoked with non-space characters. It is
554 * assumed that the caller cares about (and can detect) the transition between
555 * URL and non-URL states by looking for these.
556 */
557static enum state
558parse_url_char(enum state s, const char ch)
559{
560 if (ch == ' ' || ch == '\r' || ch == '\n') {
561 return s_dead;
562 }
563
564#if HTTP_PARSER_STRICT
565 if (ch == '\t' || ch == '\f') {
566 return s_dead;
567 }
568#endif
569
570 switch (s) {
571 case s_req_spaces_before_url:
572 /* Proxied requests are followed by scheme of an absolute URI (alpha).
573 * All methods except CONNECT are followed by '/' or '*'.
574 */
575
576 if (ch == '/' || ch == '*') {
577 return s_req_path;
578 }
579
580 if (IS_ALPHA(ch)) {
581 return s_req_schema;
582 }
583
584 break;
585
586 case s_req_schema:
587 if (IS_ALPHA(ch)) {
588 return s;
589 }
590
591 if (ch == ':') {
592 return s_req_schema_slash;
593 }
594
595 break;
596
597 case s_req_schema_slash:
598 if (ch == '/') {
599 return s_req_schema_slash_slash;
600 }
601
602 break;
603
604 case s_req_schema_slash_slash:
605 if (ch == '/') {
606 return s_req_server_start;
607 }
608
609 break;
610
611 case s_req_server_with_at:
612 if (ch == '@') {
613 return s_dead;
614 }
615
616 /* fall through */
617 case s_req_server_start:
618 case s_req_server:
619 if (ch == '/') {
620 return s_req_path;
621 }
622
623 if (ch == '?') {
624 return s_req_query_string_start;
625 }
626
627 if (ch == '@') {
628 return s_req_server_with_at;
629 }
630
631 if (IS_USERINFO_CHAR(ch) || ch == '[' || ch == ']') {
632 return s_req_server;
633 }
634
635 break;
636
637 case s_req_path:
638 if (IS_URL_CHAR(ch)) {
639 return s;
640 }
641
642 switch (ch) {
643 case '?':
644 return s_req_query_string_start;
645
646 case '#':
647 return s_req_fragment_start;
648 }
649
650 break;
651
652 case s_req_query_string_start:
653 case s_req_query_string:
654 if (IS_URL_CHAR(ch)) {
655 return s_req_query_string;
656 }
657
658 switch (ch) {
659 case '?':
660 /* allow extra '?' in query string */
661 return s_req_query_string;
662
663 case '#':
664 return s_req_fragment_start;
665 }
666
667 break;
668
669 case s_req_fragment_start:
670 if (IS_URL_CHAR(ch)) {
671 return s_req_fragment;
672 }
673
674 switch (ch) {
675 case '?':
676 return s_req_fragment;
677
678 case '#':
679 return s;
680 }
681
682 break;
683
684 case s_req_fragment:
685 if (IS_URL_CHAR(ch)) {
686 return s;
687 }
688
689 switch (ch) {
690 case '?':
691 case '#':
692 return s;
693 }
694
695 break;
696
697 default:
698 break;
699 }
700
701 /* We should never fall out of the switch above unless there's an error */
702 return s_dead;
703}
704
705size_t http_parser_execute (http_parser *parser,
706 const http_parser_settings *settings,
707 const char *data,
708 size_t len)
709{
710 char c, ch;
711 int8_t unhex_val;
712 const char *p = data;
713 const char *header_field_mark = 0;
714 const char *header_value_mark = 0;
715 const char *url_mark = 0;
716 const char *body_mark = 0;
717 const char *status_mark = 0;
718 enum state p_state = (enum state) parser->state;
719 const unsigned int lenient = parser->lenient_http_headers;
720 const unsigned int allow_chunked_length = parser->allow_chunked_length;
721
722 uint32_t nread = parser->nread;
723
724 /* We're in an error state. Don't bother doing anything. */
725 if (HTTP_PARSER_ERRNO(parser) != HPE_OK) {
726 return 0;
727 }
728
729 if (len == 0) {
730 switch (CURRENT_STATE()) {
731 case s_body_identity_eof:
732 /* Use of CALLBACK_NOTIFY() here would erroneously return 1 byte read if
733 * we got paused.
734 */
735 CALLBACK_NOTIFY_NOADVANCE(message_complete);
736 return 0;
737
738 case s_dead:
739 case s_start_req_or_res:
740 case s_start_res:
741 case s_start_req:
742 return 0;
743
744 default:
745 SET_ERRNO(HPE_INVALID_EOF_STATE);
746 return 1;
747 }
748 }
749
750
751 if (CURRENT_STATE() == s_header_field)
752 header_field_mark = data;
753 if (CURRENT_STATE() == s_header_value)
754 header_value_mark = data;
755 switch (CURRENT_STATE()) {
756 case s_req_path:
757 case s_req_schema:
758 case s_req_schema_slash:
759 case s_req_schema_slash_slash:
760 case s_req_server_start:
761 case s_req_server:
762 case s_req_server_with_at:
763 case s_req_query_string_start:
764 case s_req_query_string:
765 case s_req_fragment_start:
766 case s_req_fragment:
767 url_mark = data;
768 break;
769 case s_res_status:
770 status_mark = data;
771 break;
772 default:
773 break;
774 }
775
776 for (p=data; p != data + len; p++) {
777 ch = *p;
778
779 if (PARSING_HEADER(CURRENT_STATE()))
780 COUNT_HEADER_SIZE(1);
781
782reexecute:
783 switch (CURRENT_STATE()) {
784
785 case s_dead:
786 /* this state is used after a 'Connection: close' message
787 * the parser will error out if it reads another message
788 */
789 if (LIKELY(ch == CR || ch == LF))
790 break;
791
792 SET_ERRNO(HPE_CLOSED_CONNECTION);
793 goto error;
794
795 case s_start_req_or_res:
796 {
797 if (ch == CR || ch == LF)
798 break;
799 parser->flags = 0;
800 parser->uses_transfer_encoding = 0;
801 parser->content_length = ULLONG_MAX;
802 parser->header_content_length = ULLONG_MAX;
803
804 if (ch == 'H') {
805 UPDATE_STATE(s_res_or_resp_H);
806
807 CALLBACK_NOTIFY(message_begin);
808 } else {
809 parser->type = HTTP_REQUEST;
810 UPDATE_STATE(s_start_req);
811 REEXECUTE();
812 }
813
814 break;
815 }
816
817 case s_res_or_resp_H:
818 if (ch == 'T') {
819 parser->type = HTTP_RESPONSE;
820 UPDATE_STATE(s_res_HT);
821 } else {
822 if (UNLIKELY(ch != 'E')) {
823 SET_ERRNO(HPE_INVALID_CONSTANT);
824 goto error;
825 }
826
827 parser->type = HTTP_REQUEST;
828 parser->method = HTTP_HEAD;
829 parser->index = 2;
830 UPDATE_STATE(s_req_method);
831 }
832 break;
833
834 case s_start_res:
835 {
836 if (ch == CR || ch == LF)
837 break;
838 parser->flags = 0;
839 parser->uses_transfer_encoding = 0;
840 parser->content_length = ULLONG_MAX;
841 parser->header_content_length = ULLONG_MAX;
842
843 if (ch == 'H') {
844 UPDATE_STATE(s_res_H);
845 } else {
846 SET_ERRNO(HPE_INVALID_CONSTANT);
847 goto error;
848 }
849
850 CALLBACK_NOTIFY(message_begin);
851 break;
852 }
853
854 case s_res_H:
855 STRICT_CHECK(ch != 'T');
856 UPDATE_STATE(s_res_HT);
857 break;
858
859 case s_res_HT:
860 STRICT_CHECK(ch != 'T');
861 UPDATE_STATE(s_res_HTT);
862 break;
863
864 case s_res_HTT:
865 STRICT_CHECK(ch != 'P');
866 UPDATE_STATE(s_res_HTTP);
867 break;
868
869 case s_res_HTTP:
870 STRICT_CHECK(ch != '/');
871 UPDATE_STATE(s_res_http_major);
872 break;
873
874 case s_res_http_major:
875 if (UNLIKELY(!IS_NUM(ch))) {
876 SET_ERRNO(HPE_INVALID_VERSION);
877 goto error;
878 }
879
880 parser->http_major = ch - '0';
881 UPDATE_STATE(s_res_http_dot);
882 break;
883
884 case s_res_http_dot:
885 {
886 if (UNLIKELY(ch != '.')) {
887 SET_ERRNO(HPE_INVALID_VERSION);
888 goto error;
889 }
890
891 UPDATE_STATE(s_res_http_minor);
892 break;
893 }
894
895 case s_res_http_minor:
896 if (UNLIKELY(!IS_NUM(ch))) {
897 SET_ERRNO(HPE_INVALID_VERSION);
898 goto error;
899 }
900
901 parser->http_minor = ch - '0';
902 UPDATE_STATE(s_res_http_end);
903 break;
904
905 case s_res_http_end:
906 {
907 if (UNLIKELY(ch != ' ')) {
908 SET_ERRNO(HPE_INVALID_VERSION);
909 goto error;
910 }
911
912 UPDATE_STATE(s_res_first_status_code);
913 break;
914 }
915
916 case s_res_first_status_code:
917 {
918 if (!IS_NUM(ch)) {
919 if (ch == ' ') {
920 break;
921 }
922
923 SET_ERRNO(HPE_INVALID_STATUS);
924 goto error;
925 }
926 parser->status_code = ch - '0';
927 UPDATE_STATE(s_res_status_code);
928 break;
929 }
930
931 case s_res_status_code:
932 {
933 if (!IS_NUM(ch)) {
934 switch (ch) {
935 case ' ':
936 UPDATE_STATE(s_res_status_start);
937 break;
938 case CR:
939 case LF:
940 UPDATE_STATE(s_res_status_start);
941 REEXECUTE();
942 break;
943 default:
944 SET_ERRNO(HPE_INVALID_STATUS);
945 goto error;
946 }
947 break;
948 }
949
950 parser->status_code *= 10;
951 parser->status_code += ch - '0';
952
953 if (UNLIKELY(parser->status_code > 999)) {
954 SET_ERRNO(HPE_INVALID_STATUS);
955 goto error;
956 }
957
958 break;
959 }
960
961 case s_res_status_start:
962 {
963 MARK(status);
964 UPDATE_STATE(s_res_status);
965 parser->index = 0;
966
967 if (ch == CR || ch == LF)
968 REEXECUTE();
969
970 break;
971 }
972
973 case s_res_status:
974 if (ch == CR) {
975 UPDATE_STATE(s_res_line_almost_done);
976 CALLBACK_DATA(status);
977 break;
978 }
979
980 if (ch == LF) {
981 UPDATE_STATE(s_header_field_start);
982 CALLBACK_DATA(status);
983 break;
984 }
985
986 break;
987
988 case s_res_line_almost_done:
989 STRICT_CHECK(ch != LF);
990 UPDATE_STATE(s_header_field_start);
991 break;
992
993 case s_start_req:
994 {
995 if (ch == CR || ch == LF)
996 break;
997 parser->flags = 0;
998 parser->uses_transfer_encoding = 0;
999 parser->content_length = ULLONG_MAX;
1000 parser->header_content_length = ULLONG_MAX;
1001
1002 if (UNLIKELY(!IS_ALPHA(ch))) {
1003 SET_ERRNO(HPE_INVALID_METHOD);
1004 goto error;
1005 }
1006
1007 parser->method = (enum http_method) 0;
1008 parser->index = 1;
1009 switch (ch) {
1010 case 'A': parser->method = HTTP_ACL; break;
1011 case 'B': parser->method = HTTP_BIND; break;
1012 case 'C': parser->method = HTTP_CONNECT; /* or COPY, CHECKOUT */ break;
1013 case 'D': parser->method = HTTP_DELETE; break;
1014 case 'G': parser->method = HTTP_GET; break;
1015 case 'H': parser->method = HTTP_HEAD; break;
1016 case 'L': parser->method = HTTP_LOCK; /* or LINK */ break;
1017 case 'M': parser->method = HTTP_MKCOL; /* or MOVE, MKACTIVITY, MERGE, M-SEARCH, MKCALENDAR */ break;
1018 case 'N': parser->method = HTTP_NOTIFY; break;
1019 case 'O': parser->method = HTTP_OPTIONS; break;
1020 case 'P': parser->method = HTTP_POST;
1021 /* or PROPFIND|PROPPATCH|PUT|PATCH|PURGE */
1022 break;
1023 case 'R': parser->method = HTTP_REPORT; /* or REBIND */ break;
1024 case 'S': parser->method = HTTP_SUBSCRIBE; /* or SEARCH, SOURCE */ break;
1025 case 'T': parser->method = HTTP_TRACE; break;
1026 case 'U': parser->method = HTTP_UNLOCK; /* or UNSUBSCRIBE, UNBIND, UNLINK */ break;
1027 default:
1028 SET_ERRNO(HPE_INVALID_METHOD);
1029 goto error;
1030 }
1031 UPDATE_STATE(s_req_method);
1032
1033 CALLBACK_NOTIFY(message_begin);
1034
1035 break;
1036 }
1037
1038 case s_req_method:
1039 {
1040 const char *matcher;
1041 if (UNLIKELY(ch == '\0')) {
1042 SET_ERRNO(HPE_INVALID_METHOD);
1043 goto error;
1044 }
1045
1046 matcher = method_strings[parser->method];
1047 if (ch == ' ' && matcher[parser->index] == '\0') {
1048 UPDATE_STATE(s_req_spaces_before_url);
1049 } else if (ch == matcher[parser->index]) {
1050 ; /* nada */
1051 } else if ((ch >= 'A' && ch <= 'Z') || ch == '-') {
1052
1053 switch (parser->method << 16 | parser->index << 8 | ch) {
1054#define XX(meth, pos, ch, new_meth) \
1055 case (HTTP_##meth << 16 | pos << 8 | ch): \
1056 parser->method = HTTP_##new_meth; break;
1057
1058 XX(POST, 1, 'U', PUT)
1059 XX(POST, 1, 'A', PATCH)
1060 XX(POST, 1, 'R', PROPFIND)
1061 XX(PUT, 2, 'R', PURGE)
1062 XX(CONNECT, 1, 'H', CHECKOUT)
1063 XX(CONNECT, 2, 'P', COPY)
1064 XX(MKCOL, 1, 'O', MOVE)
1065 XX(MKCOL, 1, 'E', MERGE)
1066 XX(MKCOL, 1, '-', MSEARCH)
1067 XX(MKCOL, 2, 'A', MKACTIVITY)
1068 XX(MKCOL, 3, 'A', MKCALENDAR)
1069 XX(SUBSCRIBE, 1, 'E', SEARCH)
1070 XX(SUBSCRIBE, 1, 'O', SOURCE)
1071 XX(REPORT, 2, 'B', REBIND)
1072 XX(PROPFIND, 4, 'P', PROPPATCH)
1073 XX(LOCK, 1, 'I', LINK)
1074 XX(UNLOCK, 2, 'S', UNSUBSCRIBE)
1075 XX(UNLOCK, 2, 'B', UNBIND)
1076 XX(UNLOCK, 3, 'I', UNLINK)
1077#undef XX
1078 default:
1079 SET_ERRNO(HPE_INVALID_METHOD);
1080 goto error;
1081 }
1082 } else {
1083 SET_ERRNO(HPE_INVALID_METHOD);
1084 goto error;
1085 }
1086
1087 ++parser->index;
1088 break;
1089 }
1090
1091 case s_req_spaces_before_url:
1092 {
1093 if (ch == ' ') break;
1094
1095 MARK(url);
1096 if (parser->method == HTTP_CONNECT) {
1097 UPDATE_STATE(s_req_server_start);
1098 }
1099
1100 UPDATE_STATE(parse_url_char(CURRENT_STATE(), ch));
1101 if (UNLIKELY(CURRENT_STATE() == s_dead)) {
1102 SET_ERRNO(HPE_INVALID_URL);
1103 goto error;
1104 }
1105
1106 break;
1107 }
1108
1109 case s_req_schema:
1110 case s_req_schema_slash:
1111 case s_req_schema_slash_slash:
1112 case s_req_server_start:
1113 {
1114 switch (ch) {
1115 /* No whitespace allowed here */
1116 case ' ':
1117 case CR:
1118 case LF:
1119 SET_ERRNO(HPE_INVALID_URL);
1120 goto error;
1121 default:
1122 UPDATE_STATE(parse_url_char(CURRENT_STATE(), ch));
1123 if (UNLIKELY(CURRENT_STATE() == s_dead)) {
1124 SET_ERRNO(HPE_INVALID_URL);
1125 goto error;
1126 }
1127 }
1128
1129 break;
1130 }
1131
1132 case s_req_server:
1133 case s_req_server_with_at:
1134 case s_req_path:
1135 case s_req_query_string_start:
1136 case s_req_query_string:
1137 case s_req_fragment_start:
1138 case s_req_fragment:
1139 {
1140 switch (ch) {
1141 case ' ':
1142 UPDATE_STATE(s_req_http_start);
1143 CALLBACK_DATA(url);
1144 break;
1145 case CR:
1146 case LF:
1147 parser->http_major = 0;
1148 parser->http_minor = 9;
1149 UPDATE_STATE((ch == CR) ?
1150 s_req_line_almost_done :
1151 s_header_field_start);
1152 CALLBACK_DATA(url);
1153 break;
1154 default:
1155 UPDATE_STATE(parse_url_char(CURRENT_STATE(), ch));
1156 if (UNLIKELY(CURRENT_STATE() == s_dead)) {
1157 SET_ERRNO(HPE_INVALID_URL);
1158 goto error;
1159 }
1160 }
1161 break;
1162 }
1163
1164 case s_req_http_start:
1165 switch (ch) {
1166 case ' ':
1167 break;
1168 case 'H':
1169 UPDATE_STATE(s_req_http_H);
1170 break;
1171 case 'I':
1172 if (parser->method == HTTP_SOURCE) {
1173 UPDATE_STATE(s_req_http_I);
1174 break;
1175 }
1176 /* fall through */
1177 default:
1178 SET_ERRNO(HPE_INVALID_CONSTANT);
1179 goto error;
1180 }
1181 break;
1182
1183 case s_req_http_H:
1184 STRICT_CHECK(ch != 'T');
1185 UPDATE_STATE(s_req_http_HT);
1186 break;
1187
1188 case s_req_http_HT:
1189 STRICT_CHECK(ch != 'T');
1190 UPDATE_STATE(s_req_http_HTT);
1191 break;
1192
1193 case s_req_http_HTT:
1194 STRICT_CHECK(ch != 'P');
1195 UPDATE_STATE(s_req_http_HTTP);
1196 break;
1197
1198 case s_req_http_I:
1199 STRICT_CHECK(ch != 'C');
1200 UPDATE_STATE(s_req_http_IC);
1201 break;
1202
1203 case s_req_http_IC:
1204 STRICT_CHECK(ch != 'E');
1205 UPDATE_STATE(s_req_http_HTTP); /* Treat "ICE" as "HTTP". */
1206 break;
1207
1208 case s_req_http_HTTP:
1209 STRICT_CHECK(ch != '/');
1210 UPDATE_STATE(s_req_http_major);
1211 break;
1212
1213 case s_req_http_major:
1214 if (UNLIKELY(!IS_NUM(ch))) {
1215 SET_ERRNO(HPE_INVALID_VERSION);
1216 goto error;
1217 }
1218
1219 parser->http_major = ch - '0';
1220 UPDATE_STATE(s_req_http_dot);
1221 break;
1222
1223 case s_req_http_dot:
1224 {
1225 if (UNLIKELY(ch != '.')) {
1226 SET_ERRNO(HPE_INVALID_VERSION);
1227 goto error;
1228 }
1229
1230 UPDATE_STATE(s_req_http_minor);
1231 break;
1232 }
1233
1234 case s_req_http_minor:
1235 if (UNLIKELY(!IS_NUM(ch))) {
1236 SET_ERRNO(HPE_INVALID_VERSION);
1237 goto error;
1238 }
1239
1240 parser->http_minor = ch - '0';
1241 UPDATE_STATE(s_req_http_end);
1242 break;
1243
1244 case s_req_http_end:
1245 {
1246 if (ch == CR) {
1247 UPDATE_STATE(s_req_line_almost_done);
1248 break;
1249 }
1250
1251 if (ch == LF) {
1252 UPDATE_STATE(s_header_field_start);
1253 break;
1254 }
1255
1256 SET_ERRNO(HPE_INVALID_VERSION);
1257 goto error;
1258 break;
1259 }
1260
1261 /* end of request line */
1262 case s_req_line_almost_done:
1263 {
1264 if (UNLIKELY(ch != LF)) {
1265 SET_ERRNO(HPE_LF_EXPECTED);
1266 goto error;
1267 }
1268
1269 UPDATE_STATE(s_header_field_start);
1270 break;
1271 }
1272
1273 case s_header_field_start:
1274 {
1275 if (ch == CR) {
1276 UPDATE_STATE(s_headers_almost_done);
1277 break;
1278 }
1279
1280 if (ch == LF) {
1281 /* they might be just sending \n instead of \r\n so this would be
1282 * the second \n to denote the end of headers*/
1283 UPDATE_STATE(s_headers_almost_done);
1284 REEXECUTE();
1285 }
1286
1287 c = TOKEN(ch);
1288
1289 if (UNLIKELY(!c)) {
1290 SET_ERRNO(HPE_INVALID_HEADER_TOKEN);
1291 goto error;
1292 }
1293
1294 MARK(header_field);
1295
1296 parser->index = 0;
1297 UPDATE_STATE(s_header_field);
1298
1299 switch (c) {
1300 case 'c':
1301 parser->header_state = h_C;
1302 break;
1303
1304 case 'p':
1305 parser->header_state = h_matching_proxy_connection;
1306 break;
1307
1308 case 't':
1309 parser->header_state = h_matching_transfer_encoding;
1310 break;
1311
1312 case 'u':
1313 parser->header_state = h_matching_upgrade;
1314 break;
1315
1316 default:
1317 parser->header_state = h_general;
1318 break;
1319 }
1320 break;
1321 }
1322
1323 case s_header_field:
1324 {
1325 const char* start = p;
1326 for (; p != data + len; p++) {
1327 ch = *p;
1328 c = TOKEN(ch);
1329
1330 if (!c)
1331 break;
1332
1333 switch (parser->header_state) {
1334 case h_general: {
1335 size_t left = data + len - p;
1336 const char* pe = p + MIN(left, max_header_size);
1337 while (p+1 < pe && TOKEN(p[1])) {
1338 p++;
1339 }
1340 break;
1341 }
1342
1343 case h_C:
1344 parser->index++;
1345 parser->header_state = (c == 'o' ? h_CO : h_general);
1346 break;
1347
1348 case h_CO:
1349 parser->index++;
1350 parser->header_state = (c == 'n' ? h_CON : h_general);
1351 break;
1352
1353 case h_CON:
1354 parser->index++;
1355 switch (c) {
1356 case 'n':
1357 parser->header_state = h_matching_connection;
1358 break;
1359 case 't':
1360 parser->header_state = h_matching_content_length;
1361 break;
1362 default:
1363 parser->header_state = h_general;
1364 break;
1365 }
1366 break;
1367
1368 /* connection */
1369
1370 case h_matching_connection:
1371 parser->index++;
1372 if (parser->index > sizeof(CONNECTION)-1
1373 || c != CONNECTION[parser->index]) {
1374 parser->header_state = h_general;
1375 } else if (parser->index == sizeof(CONNECTION)-2) {
1376 parser->header_state = h_connection;
1377 }
1378 break;
1379
1380 /* proxy-connection */
1381
1382 case h_matching_proxy_connection:
1383 parser->index++;
1384 if (parser->index > sizeof(PROXY_CONNECTION)-1
1385 || c != PROXY_CONNECTION[parser->index]) {
1386 parser->header_state = h_general;
1387 } else if (parser->index == sizeof(PROXY_CONNECTION)-2) {
1388 parser->header_state = h_connection;
1389 }
1390 break;
1391
1392 /* content-length */
1393
1394 case h_matching_content_length:
1395 parser->index++;
1396 if (parser->index > sizeof(CONTENT_LENGTH)-1
1397 || c != CONTENT_LENGTH[parser->index]) {
1398 parser->header_state = h_general;
1399 } else if (parser->index == sizeof(CONTENT_LENGTH)-2) {
1400 parser->header_state = h_content_length;
1401 }
1402 break;
1403
1404 /* transfer-encoding */
1405
1406 case h_matching_transfer_encoding:
1407 parser->index++;
1408 if (parser->index > sizeof(TRANSFER_ENCODING)-1
1409 || c != TRANSFER_ENCODING[parser->index]) {
1410 parser->header_state = h_general;
1411 } else if (parser->index == sizeof(TRANSFER_ENCODING)-2) {
1412 parser->header_state = h_transfer_encoding;
1413 parser->uses_transfer_encoding = 1;
1414 }
1415 break;
1416
1417 /* upgrade */
1418
1419 case h_matching_upgrade:
1420 parser->index++;
1421 if (parser->index > sizeof(UPGRADE)-1
1422 || c != UPGRADE[parser->index]) {
1423 parser->header_state = h_general;
1424 } else if (parser->index == sizeof(UPGRADE)-2) {
1425 parser->header_state = h_upgrade;
1426 }
1427 break;
1428
1429 case h_connection:
1430 case h_content_length:
1431 case h_transfer_encoding:
1432 case h_upgrade:
1433 if (ch != ' ') parser->header_state = h_general;
1434 break;
1435
1436 default:
1437 assert(0 && "Unknown header_state");
1438 break;
1439 }
1440 }
1441
1442 if (p == data + len) {
1443 --p;
1444 COUNT_HEADER_SIZE(p - start);
1445 break;
1446 }
1447
1448 COUNT_HEADER_SIZE(p - start);
1449
1450 if (ch == ':') {
1451 UPDATE_STATE(s_header_value_discard_ws);
1452 CALLBACK_DATA(header_field);
1453 break;
1454 }
1455
1456 SET_ERRNO(HPE_INVALID_HEADER_TOKEN);
1457 goto error;
1458 }
1459
1460 case s_header_value_discard_ws:
1461 if (ch == ' ' || ch == '\t') break;
1462
1463 if (ch == CR) {
1464 UPDATE_STATE(s_header_value_discard_ws_almost_done);
1465 break;
1466 }
1467
1468 if (ch == LF) {
1469 UPDATE_STATE(s_header_value_discard_lws);
1470 break;
1471 }
1472
1473 /* fall through */
1474
1475 case s_header_value_start:
1476 {
1477 MARK(header_value);
1478
1479 UPDATE_STATE(s_header_value);
1480 parser->index = 0;
1481
1482 c = LOWER(ch);
1483
1484 switch (parser->header_state) {
1485 case h_upgrade:
1486 parser->flags |= F_UPGRADE;
1487 parser->header_state = h_general;
1488 break;
1489
1490 case h_transfer_encoding:
1491 /* looking for 'Transfer-Encoding: chunked' */
1492 if ('c' == c) {
1493 parser->header_state = h_matching_transfer_encoding_chunked;
1494 } else {
1495 parser->header_state = h_matching_transfer_encoding_token;
1496 }
1497 break;
1498
1499 /* Multi-value `Transfer-Encoding` header */
1500 case h_matching_transfer_encoding_token_start:
1501 break;
1502
1503 case h_content_length:
1504 if (UNLIKELY(!IS_NUM(ch))) {
1505 SET_ERRNO(HPE_INVALID_CONTENT_LENGTH);
1506 goto error;
1507 }
1508
1509 if (parser->flags & F_CONTENTLENGTH) {
1510 SET_ERRNO(HPE_UNEXPECTED_CONTENT_LENGTH);
1511 goto error;
1512 }
1513
1514 parser->flags |= F_CONTENTLENGTH;
1515 parser->content_length = ch - '0';
1516 parser->header_state = h_content_length_num;
1517 break;
1518
1519 /* when obsolete line folding is encountered for content length
1520 * continue to the s_header_value state */
1521 case h_content_length_ws:
1522 break;
1523
1524 case h_connection:
1525 /* looking for 'Connection: keep-alive' */
1526 if (c == 'k') {
1527 parser->header_state = h_matching_connection_keep_alive;
1528 /* looking for 'Connection: close' */
1529 } else if (c == 'c') {
1530 parser->header_state = h_matching_connection_close;
1531 } else if (c == 'u') {
1532 parser->header_state = h_matching_connection_upgrade;
1533 } else {
1534 parser->header_state = h_matching_connection_token;
1535 }
1536 break;
1537
1538 /* Multi-value `Connection` header */
1539 case h_matching_connection_token_start:
1540 break;
1541
1542 default:
1543 parser->header_state = h_general;
1544 break;
1545 }
1546 break;
1547 }
1548
1549 case s_header_value:
1550 {
1551 const char* start = p;
1552 enum header_states h_state = (enum header_states) parser->header_state;
1553 for (; p != data + len; p++) {
1554 ch = *p;
1555 if (ch == CR) {
1556 UPDATE_STATE(s_header_almost_done);
1557 parser->header_state = h_state;
1558 CALLBACK_DATA(header_value);
1559 break;
1560 }
1561
1562 if (ch == LF) {
1563 UPDATE_STATE(s_header_almost_done);
1564 COUNT_HEADER_SIZE(p - start);
1565 parser->header_state = h_state;
1566 CALLBACK_DATA_NOADVANCE(header_value);
1567 REEXECUTE();
1568 }
1569
1570 if (!lenient && !IS_HEADER_CHAR(ch)) {
1571 SET_ERRNO(HPE_INVALID_HEADER_TOKEN);
1572 goto error;
1573 }
1574
1575 c = LOWER(ch);
1576
1577 switch (h_state) {
1578 case h_general:
1579 {
1580 size_t left = data + len - p;
1581 const char* pe = p + MIN(left, max_header_size);
1582
1583 for (; p != pe; p++) {
1584 ch = *p;
1585 if (ch == CR || ch == LF) {
1586 --p;
1587 break;
1588 }
1589 if (!lenient && !IS_HEADER_CHAR(ch)) {
1590 SET_ERRNO(HPE_INVALID_HEADER_TOKEN);
1591 goto error;
1592 }
1593 }
1594 if (p == data + len)
1595 --p;
1596 break;
1597 }
1598
1599 case h_connection:
1600 case h_transfer_encoding:
1601 assert(0 && "Shouldn't get here.");
1602 break;
1603
1604 case h_content_length:
1605 if (ch == ' ') break;
1606 h_state = h_content_length_num;
1607 /* fall through */
1608
1609 case h_content_length_num:
1610 {
1611 uint64_t t;
1612
1613 if (ch == ' ') {
1614 h_state = h_content_length_ws;
1615 break;
1616 }
1617
1618 if (UNLIKELY(!IS_NUM(ch))) {
1619 SET_ERRNO(HPE_INVALID_CONTENT_LENGTH);
1620 parser->header_state = h_state;
1621 goto error;
1622 }
1623
1624 t = parser->content_length;
1625 t *= 10;
1626 t += ch - '0';
1627
1628 /* Overflow? Test against a conservative limit for simplicity. */
1629 if (UNLIKELY((ULLONG_MAX - 10) / 10 < parser->content_length)) {
1630 SET_ERRNO(HPE_INVALID_CONTENT_LENGTH);
1631 parser->header_state = h_state;
1632 goto error;
1633 }
1634
1635 parser->content_length = t;
1636 parser->header_content_length = t;
1637 break;
1638 }
1639
1640 case h_content_length_ws:
1641 if (ch == ' ') break;
1642 SET_ERRNO(HPE_INVALID_CONTENT_LENGTH);
1643 parser->header_state = h_state;
1644 goto error;
1645
1646 /* Transfer-Encoding: chunked */
1647 case h_matching_transfer_encoding_token_start:
1648 /* looking for 'Transfer-Encoding: chunked' */
1649 if ('c' == c) {
1650 h_state = h_matching_transfer_encoding_chunked;
1651 } else if (STRICT_TOKEN(c)) {
1652 /* TODO(indutny): similar code below does this, but why?
1653 * At the very least it seems to be inconsistent given that
1654 * h_matching_transfer_encoding_token does not check for
1655 * `STRICT_TOKEN`
1656 */
1657 h_state = h_matching_transfer_encoding_token;
1658 } else if (c == ' ' || c == '\t') {
1659 /* Skip lws */
1660 } else {
1661 h_state = h_general;
1662 }
1663 break;
1664
1665 case h_matching_transfer_encoding_chunked:
1666 parser->index++;
1667 if (parser->index > sizeof(CHUNKED)-1
1668 || c != CHUNKED[parser->index]) {
1669 h_state = h_matching_transfer_encoding_token;
1670 } else if (parser->index == sizeof(CHUNKED)-2) {
1671 h_state = h_transfer_encoding_chunked;
1672 }
1673 break;
1674
1675 case h_matching_transfer_encoding_token:
1676 if (ch == ',') {
1677 h_state = h_matching_transfer_encoding_token_start;
1678 parser->index = 0;
1679 }
1680 break;
1681
1682 case h_matching_connection_token_start:
1683 /* looking for 'Connection: keep-alive' */
1684 if (c == 'k') {
1685 h_state = h_matching_connection_keep_alive;
1686 /* looking for 'Connection: close' */
1687 } else if (c == 'c') {
1688 h_state = h_matching_connection_close;
1689 } else if (c == 'u') {
1690 h_state = h_matching_connection_upgrade;
1691 } else if (STRICT_TOKEN(c)) {
1692 h_state = h_matching_connection_token;
1693 } else if (c == ' ' || c == '\t') {
1694 /* Skip lws */
1695 } else {
1696 h_state = h_general;
1697 }
1698 break;
1699
1700 /* looking for 'Connection: keep-alive' */
1701 case h_matching_connection_keep_alive:
1702 parser->index++;
1703 if (parser->index > sizeof(KEEP_ALIVE)-1
1704 || c != KEEP_ALIVE[parser->index]) {
1705 h_state = h_matching_connection_token;
1706 } else if (parser->index == sizeof(KEEP_ALIVE)-2) {
1707 h_state = h_connection_keep_alive;
1708 }
1709 break;
1710
1711 /* looking for 'Connection: close' */
1712 case h_matching_connection_close:
1713 parser->index++;
1714 if (parser->index > sizeof(CLOSE)-1 || c != CLOSE[parser->index]) {
1715 h_state = h_matching_connection_token;
1716 } else if (parser->index == sizeof(CLOSE)-2) {
1717 h_state = h_connection_close;
1718 }
1719 break;
1720
1721 /* looking for 'Connection: upgrade' */
1722 case h_matching_connection_upgrade:
1723 parser->index++;
1724 if (parser->index > sizeof(UPGRADE) - 1 ||
1725 c != UPGRADE[parser->index]) {
1726 h_state = h_matching_connection_token;
1727 } else if (parser->index == sizeof(UPGRADE)-2) {
1728 h_state = h_connection_upgrade;
1729 }
1730 break;
1731
1732 case h_matching_connection_token:
1733 if (ch == ',') {
1734 h_state = h_matching_connection_token_start;
1735 parser->index = 0;
1736 }
1737 break;
1738
1739 case h_transfer_encoding_chunked:
1740 if (ch != ' ') h_state = h_matching_transfer_encoding_token;
1741 break;
1742
1743 case h_connection_keep_alive:
1744 case h_connection_close:
1745 case h_connection_upgrade:
1746 if (ch == ',') {
1747 if (h_state == h_connection_keep_alive) {
1748 parser->flags |= F_CONNECTION_KEEP_ALIVE;
1749 } else if (h_state == h_connection_close) {
1750 parser->flags |= F_CONNECTION_CLOSE;
1751 } else if (h_state == h_connection_upgrade) {
1752 parser->flags |= F_CONNECTION_UPGRADE;
1753 }
1754 h_state = h_matching_connection_token_start;
1755 parser->index = 0;
1756 } else if (ch != ' ') {
1757 h_state = h_matching_connection_token;
1758 }
1759 break;
1760
1761 default:
1762 UPDATE_STATE(s_header_value);
1763 h_state = h_general;
1764 break;
1765 }
1766 }
1767 parser->header_state = h_state;
1768
1769 if (p == data + len)
1770 --p;
1771
1772 COUNT_HEADER_SIZE(p - start);
1773 break;
1774 }
1775
1776 case s_header_almost_done:
1777 {
1778 if (UNLIKELY(ch != LF)) {
1779 SET_ERRNO(HPE_LF_EXPECTED);
1780 goto error;
1781 }
1782
1783 UPDATE_STATE(s_header_value_lws);
1784 break;
1785 }
1786
1787 case s_header_value_lws:
1788 {
1789 if (ch == ' ' || ch == '\t') {
1790 if (parser->header_state == h_content_length_num) {
1791 /* treat obsolete line folding as space */
1792 parser->header_state = h_content_length_ws;
1793 }
1794 UPDATE_STATE(s_header_value_start);
1795 REEXECUTE();
1796 }
1797
1798 /* finished the header */
1799 switch (parser->header_state) {
1800 case h_connection_keep_alive:
1801 parser->flags |= F_CONNECTION_KEEP_ALIVE;
1802 break;
1803 case h_connection_close:
1804 parser->flags |= F_CONNECTION_CLOSE;
1805 break;
1806 case h_transfer_encoding_chunked:
1807 parser->flags |= F_CHUNKED;
1808 break;
1809 case h_connection_upgrade:
1810 parser->flags |= F_CONNECTION_UPGRADE;
1811 break;
1812 default:
1813 break;
1814 }
1815
1816 UPDATE_STATE(s_header_field_start);
1817 REEXECUTE();
1818 }
1819
1820 case s_header_value_discard_ws_almost_done:
1821 {
1822 STRICT_CHECK(ch != LF);
1823 UPDATE_STATE(s_header_value_discard_lws);
1824 break;
1825 }
1826
1827 case s_header_value_discard_lws:
1828 {
1829 if (ch == ' ' || ch == '\t') {
1830 UPDATE_STATE(s_header_value_discard_ws);
1831 break;
1832 } else {
1833 switch (parser->header_state) {
1834 case h_connection_keep_alive:
1835 parser->flags |= F_CONNECTION_KEEP_ALIVE;
1836 break;
1837 case h_connection_close:
1838 parser->flags |= F_CONNECTION_CLOSE;
1839 break;
1840 case h_connection_upgrade:
1841 parser->flags |= F_CONNECTION_UPGRADE;
1842 break;
1843 case h_transfer_encoding_chunked:
1844 parser->flags |= F_CHUNKED;
1845 break;
1846 case h_content_length:
1847 /* do not allow empty content length */
1848 SET_ERRNO(HPE_INVALID_CONTENT_LENGTH);
1849 goto error;
1850 break;
1851 default:
1852 break;
1853 }
1854
1855 /* header value was empty */
1856 MARK(header_value);
1857 UPDATE_STATE(s_header_field_start);
1858 CALLBACK_DATA_NOADVANCE(header_value);
1859 REEXECUTE();
1860 }
1861 }
1862
1863 case s_headers_almost_done:
1864 {
1865 STRICT_CHECK(ch != LF);
1866
1867 if (parser->flags & F_TRAILING) {
1868 /* End of a chunked request */
1869 UPDATE_STATE(s_message_done);
1870 CALLBACK_NOTIFY_NOADVANCE(chunk_complete);
1871 REEXECUTE();
1872 }
1873
1874 /* Cannot use transfer-encoding and a content-length header together
1875 per the HTTP specification. (RFC 7230 Section 3.3.3) */
1876 if ((parser->uses_transfer_encoding == 1) &&
1877 (parser->flags & F_CONTENTLENGTH)) {
1878 /* Allow it for lenient parsing as long as `Transfer-Encoding` is
1879 * not `chunked` or allow_length_with_encoding is set
1880 */
1881 if (parser->flags & F_CHUNKED) {
1882 if (!allow_chunked_length) {
1883 SET_ERRNO(HPE_UNEXPECTED_CONTENT_LENGTH);
1884 goto error;
1885 }
1886 } else if (!lenient) {
1887 SET_ERRNO(HPE_UNEXPECTED_CONTENT_LENGTH);
1888 goto error;
1889 }
1890 }
1891
1892 UPDATE_STATE(s_headers_done);
1893
1894 /* Set this here so that on_headers_complete() callbacks can see it */
1895 if ((parser->flags & F_UPGRADE) &&
1896 (parser->flags & F_CONNECTION_UPGRADE)) {
1897 /* For responses, "Upgrade: foo" and "Connection: upgrade" are
1898 * mandatory only when it is a 101 Switching Protocols response,
1899 * otherwise it is purely informational, to announce support.
1900 */
1901 parser->upgrade =
1902 (parser->type == HTTP_REQUEST || parser->status_code == 101);
1903 } else {
1904 parser->upgrade = (parser->method == HTTP_CONNECT);
1905 }
1906
1907 /* Here we call the headers_complete callback. This is somewhat
1908 * different than other callbacks because if the user returns 1, we
1909 * will interpret that as saying that this message has no body. This
1910 * is needed for the annoying case of recieving a response to a HEAD
1911 * request.
1912 *
1913 * We'd like to use CALLBACK_NOTIFY_NOADVANCE() here but we cannot, so
1914 * we have to simulate it by handling a change in errno below.
1915 */
1916 if (settings->on_headers_complete) {
1917 switch (settings->on_headers_complete(parser)) {
1918 case 0:
1919 break;
1920
1921 case 2:
1922 parser->upgrade = 1;
1923
1924 /* fall through */
1925 case 1:
1926 parser->flags |= F_SKIPBODY;
1927 break;
1928
1929 default:
1930 SET_ERRNO(HPE_CB_headers_complete);
1931 RETURN(p - data); /* Error */
1932 }
1933 }
1934
1935 if (HTTP_PARSER_ERRNO(parser) != HPE_OK) {
1936 RETURN(p - data);
1937 }
1938
1939 REEXECUTE();
1940 }
1941
1942 case s_headers_done:
1943 {
1944 int hasBody;
1945 STRICT_CHECK(ch != LF);
1946
1947 parser->nread = 0;
1948 nread = 0;
1949
1950 hasBody = parser->flags & F_CHUNKED ||
1951 (parser->content_length > 0 && parser->content_length != ULLONG_MAX);
1952
1953 if (parser->upgrade && (parser->method == HTTP_CONNECT ||
1954 (parser->flags & F_SKIPBODY) || !hasBody)) {
1955 /* Exit, the rest of the message is in a different protocol. */
1956 UPDATE_STATE(NEW_MESSAGE());
1957 CALLBACK_NOTIFY(message_complete);
1958 RETURN((p - data) + 1);
1959 }
1960
1961 if (parser->flags & F_SKIPBODY) {
1962 UPDATE_STATE(NEW_MESSAGE());
1963 CALLBACK_NOTIFY(message_complete);
1964 } else if (parser->flags & F_CHUNKED) {
1965 /* chunked encoding - ignore Content-Length header,
1966 * prepare for a chunk */
1967 UPDATE_STATE(s_chunk_size_start);
1968 } else if (parser->uses_transfer_encoding == 1) {
1969 if (parser->type == HTTP_REQUEST && !lenient) {
1970 /* RFC 7230 3.3.3 */
1971
1972 /* If a Transfer-Encoding header field
1973 * is present in a request and the chunked transfer coding is not
1974 * the final encoding, the message body length cannot be determined
1975 * reliably; the server MUST respond with the 400 (Bad Request)
1976 * status code and then close the connection.
1977 */
1978 SET_ERRNO(HPE_INVALID_TRANSFER_ENCODING);
1979 RETURN(p - data); /* Error */
1980 } else {
1981 /* RFC 7230 3.3.3 */
1982
1983 /* If a Transfer-Encoding header field is present in a response and
1984 * the chunked transfer coding is not the final encoding, the
1985 * message body length is determined by reading the connection until
1986 * it is closed by the server.
1987 */
1988 UPDATE_STATE(s_body_identity_eof);
1989 }
1990 } else {
1991 if (parser->content_length == 0) {
1992 /* Content-Length header given but zero: Content-Length: 0\r\n */
1993 UPDATE_STATE(NEW_MESSAGE());
1994 CALLBACK_NOTIFY(message_complete);
1995 } else if (parser->content_length != ULLONG_MAX) {
1996 /* Content-Length header given and non-zero */
1997 UPDATE_STATE(s_body_identity);
1998 } else {
1999 if (!http_message_needs_eof(parser)) {
2000 /* Assume content-length 0 - read the next */
2001 UPDATE_STATE(NEW_MESSAGE());
2002 CALLBACK_NOTIFY(message_complete);
2003 } else {
2004 /* Read body until EOF */
2005 UPDATE_STATE(s_body_identity_eof);
2006 }
2007 }
2008 }
2009
2010 break;
2011 }
2012
2013 case s_body_identity:
2014 {
2015 uint64_t to_read = MIN(parser->content_length,
2016 (uint64_t) ((data + len) - p));
2017
2018 assert(parser->content_length != 0
2019 && parser->content_length != ULLONG_MAX);
2020
2021 /* The difference between advancing content_length and p is because
2022 * the latter will automaticaly advance on the next loop iteration.
2023 * Further, if content_length ends up at 0, we want to see the last
2024 * byte again for our message complete callback.
2025 */
2026 MARK(body);
2027 parser->content_length -= to_read;
2028 p += to_read - 1;
2029
2030 if (parser->content_length == 0) {
2031 UPDATE_STATE(s_message_done);
2032
2033 /* Mimic CALLBACK_DATA_NOADVANCE() but with one extra byte.
2034 *
2035 * The alternative to doing this is to wait for the next byte to
2036 * trigger the data callback, just as in every other case. The
2037 * problem with this is that this makes it difficult for the test
2038 * harness to distinguish between complete-on-EOF and
2039 * complete-on-length. It's not clear that this distinction is
2040 * important for applications, but let's keep it for now.
2041 */
2042 CALLBACK_DATA_(body, p - body_mark + 1, p - data);
2043 REEXECUTE();
2044 }
2045
2046 break;
2047 }
2048
2049 /* read until EOF */
2050 case s_body_identity_eof:
2051 MARK(body);
2052 p = data + len - 1;
2053
2054 break;
2055
2056 case s_message_done:
2057 UPDATE_STATE(NEW_MESSAGE());
2058 CALLBACK_NOTIFY(message_complete);
2059 if (parser->upgrade) {
2060 /* Exit, the rest of the message is in a different protocol. */
2061 RETURN((p - data) + 1);
2062 }
2063 break;
2064
2065 case s_chunk_size_start:
2066 {
2067 assert(nread == 1);
2068 assert(parser->flags & F_CHUNKED);
2069
2070 unhex_val = unhex[(unsigned char)ch];
2071 if (UNLIKELY(unhex_val == -1)) {
2072 SET_ERRNO(HPE_INVALID_CHUNK_SIZE);
2073 goto error;
2074 }
2075
2076 parser->content_length = unhex_val;
2077 UPDATE_STATE(s_chunk_size);
2078 break;
2079 }
2080
2081 case s_chunk_size:
2082 {
2083 uint64_t t;
2084
2085 assert(parser->flags & F_CHUNKED);
2086
2087 if (ch == CR) {
2088 UPDATE_STATE(s_chunk_size_almost_done);
2089 break;
2090 }
2091
2092 unhex_val = unhex[(unsigned char)ch];
2093
2094 if (unhex_val == -1) {
2095 if (ch == ';' || ch == ' ') {
2096 UPDATE_STATE(s_chunk_parameters);
2097 break;
2098 }
2099
2100 SET_ERRNO(HPE_INVALID_CHUNK_SIZE);
2101 goto error;
2102 }
2103
2104 t = parser->content_length;
2105 t *= 16;
2106 t += unhex_val;
2107
2108 /* Overflow? Test against a conservative limit for simplicity. */
2109 if (UNLIKELY((ULLONG_MAX - 16) / 16 < parser->content_length)) {
2110 SET_ERRNO(HPE_INVALID_CONTENT_LENGTH);
2111 goto error;
2112 }
2113
2114 parser->content_length = t;
2115 break;
2116 }
2117
2118 case s_chunk_parameters:
2119 {
2120 assert(parser->flags & F_CHUNKED);
2121 /* just ignore this shit. TODO check for overflow */
2122 if (ch == CR) {
2123 UPDATE_STATE(s_chunk_size_almost_done);
2124 break;
2125 }
2126 break;
2127 }
2128
2129 case s_chunk_size_almost_done:
2130 {
2131 assert(parser->flags & F_CHUNKED);
2132 STRICT_CHECK(ch != LF);
2133
2134 parser->nread = 0;
2135 nread = 0;
2136
2137 if (parser->content_length == 0) {
2138 parser->flags |= F_TRAILING;
2139 UPDATE_STATE(s_header_field_start);
2140 } else {
2141 UPDATE_STATE(s_chunk_data);
2142 }
2143 CALLBACK_NOTIFY(chunk_header);
2144 break;
2145 }
2146
2147 case s_chunk_data:
2148 {
2149 uint64_t to_read = MIN(parser->content_length,
2150 (uint64_t) ((data + len) - p));
2151
2152 assert(parser->flags & F_CHUNKED);
2153 assert(parser->content_length != 0
2154 && parser->content_length != ULLONG_MAX);
2155
2156 /* See the explanation in s_body_identity for why the content
2157 * length and data pointers are managed this way.
2158 */
2159 MARK(body);
2160 parser->content_length -= to_read;
2161 p += to_read - 1;
2162
2163 if (parser->content_length == 0) {
2164 UPDATE_STATE(s_chunk_data_almost_done);
2165 }
2166
2167 break;
2168 }
2169
2170 case s_chunk_data_almost_done:
2171 assert(parser->flags & F_CHUNKED);
2172 assert(parser->content_length == 0);
2173 STRICT_CHECK(ch != CR);
2174 UPDATE_STATE(s_chunk_data_done);
2175 CALLBACK_DATA(body);
2176 break;
2177
2178 case s_chunk_data_done:
2179 assert(parser->flags & F_CHUNKED);
2180 STRICT_CHECK(ch != LF);
2181 parser->nread = 0;
2182 nread = 0;
2183 UPDATE_STATE(s_chunk_size_start);
2184 CALLBACK_NOTIFY(chunk_complete);
2185 break;
2186
2187 default:
2188 assert(0 && "unhandled state");
2189 SET_ERRNO(HPE_INVALID_INTERNAL_STATE);
2190 goto error;
2191 }
2192 }
2193
2194 /* Run callbacks for any marks that we have leftover after we ran out of
2195 * bytes. There should be at most one of these set, so it's OK to invoke
2196 * them in series (unset marks will not result in callbacks).
2197 *
2198 * We use the NOADVANCE() variety of callbacks here because 'p' has already
2199 * overflowed 'data' and this allows us to correct for the off-by-one that
2200 * we'd otherwise have (since CALLBACK_DATA() is meant to be run with a 'p'
2201 * value that's in-bounds).
2202 */
2203
2204 assert(((header_field_mark ? 1 : 0) +
2205 (header_value_mark ? 1 : 0) +
2206 (url_mark ? 1 : 0) +
2207 (body_mark ? 1 : 0) +
2208 (status_mark ? 1 : 0)) <= 1);
2209
2210 CALLBACK_DATA_NOADVANCE_INTR(header_field);
2211 CALLBACK_DATA_NOADVANCE_INTR(header_value);
2212 CALLBACK_DATA_NOADVANCE_INTR(url);
2213 CALLBACK_DATA_NOADVANCE_INTR(body);
2214 CALLBACK_DATA_NOADVANCE_INTR(status);
2215
2216 RETURN(len);
2217
2218error:
2219 if (HTTP_PARSER_ERRNO(parser) == HPE_OK) {
2220 SET_ERRNO(HPE_UNKNOWN);
2221 }
2222
2223 RETURN(p - data);
2224}

2226int http_header_only_message(const http_parser *parser)
2227{
2228 int hasBody;
2229
2230 hasBody = parser->flags & F_CHUNKED ||
2231 (parser->header_content_length > 0 && parser->header_content_length != ULLONG_MAX);
2232
2233 if (parser->upgrade && (parser->method == HTTP_CONNECT ||
2234 (parser->flags & F_SKIPBODY) || !hasBody)) {
2235 return 1;
2236 }
2237
2238 if (parser->flags & F_SKIPBODY) {
2239 return 1;
2240 } else if (parser->flags & F_CHUNKED) {
2241 /* chunked encoding - ignore Content-Length header */
2242 return 0;
2243 } else {

2245 /* Content-Length header given but zero: Content-Length: 0\r\n */
2246 return 1;
2247 } else if (parser->header_content_length != ULLONG_MAX) {

2262/* Does the parser need to see an EOF to find the end of the message? */
2263int http_message_needs_eof (const http_parser *parser)
2264{

2266 return 0;
2267 }
2268
2269 /* See RFC 2616 section 4.4 */
2270 if (parser->status_code / 100 == 1 || /* 1xx e.g. Continue */
2271 parser->status_code == 204 || /* No Content */
2272 parser->status_code == 304 || /* Not Modified */
2273 parser->flags & F_SKIPBODY) { /* response to a HEAD request */
2274 return 0;
2275 }
2276
2277 /* RFC 7230 3.3.3, see `s_headers_almost_done` */
2278 if ((parser->uses_transfer_encoding == 1) &&
2279 (parser->flags & F_CHUNKED) == 0) {
2280 return 1;
2281 }
2282
2283 if ((parser->flags & F_CHUNKED) || parser->content_length != ULLONG_MAX) {
2284 return 0;
2285 }
2286
2287 return 1;

2441{

2443
2444 const char *p;
2445 size_t buflen = u->field_data[UF_HOST].off + u->field_data[UF_HOST].len;
2446
2447 assert(u->field_set & (1 << UF_HOST));
2448
2449 u->field_data[UF_HOST].len = 0;
2450
2451 s = found_at ? s_http_userinfo_start : s_http_host_start;

2470 u->field_data[UF_HOST].off = (uint16_t)(p - buf);
2471 }
2472 u->field_data[UF_HOST].len++;
2473 break;
2474
2475 case s_http_host_v6_zone_start:
2476 case s_http_host_v6_zone:
2477 u->field_data[UF_HOST].len++;
2478 break;
2479

2483 u->field_data[UF_PORT].len = 0;
2484 u->field_set |= (1 << UF_PORT);
2485 }

2487 break;
2488
2489 case s_http_userinfo:
2490 if (s != s_http_userinfo) {
2491 u->field_data[UF_USERINFO].off = (uint16_t)(p - buf);
2492 u->field_data[UF_USERINFO].len = 0;

2527int
2528http_parser_parse_url(const char *buf, size_t buflen, int is_connect,
2529 struct http_parser_url *u)
2530{
2531 enum state s;
2532 const char *p;
2533 enum http_parser_url_fields uf, old_uf;
2534 int found_at = 0;
2535
2536 if (buflen == 0) {
2537 return 1;
2538 }
2539
2540 u->port = u->field_set = 0;
2541 s = is_connect ? s_req_server_start : s_req_spaces_before_url;
2542 old_uf = UF_MAX;
2543
2544 for (p = buf; p < buf + buflen; p++) {
2545 s = parse_url_char(s, *p);
2546
2547 /* Figure out the next field that we're operating on */
2548 switch (s) {
2549 case s_dead:
2550 return 1;
2551
2552 /* Skip delimeters */
2553 case s_req_schema_slash:
2554 case s_req_schema_slash_slash:
2555 case s_req_server_start:
2556 case s_req_query_string_start:
2557 case s_req_fragment_start:
2558 continue;
2559
2560 case s_req_schema:
2561 uf = UF_SCHEMA;
2562 break;
2563
2564 case s_req_server_with_at:
2565 found_at = 1;
2566
2567 /* fall through */
2568 case s_req_server:
2569 uf = UF_HOST;
2570 break;
2571
2572 case s_req_path:
2573 uf = UF_PATH;
2574 break;
2575
2576 case s_req_query_string:
2577 uf = UF_QUERY;
2578 break;
2579
2580 case s_req_fragment:
2581 uf = UF_FRAGMENT;
2582 break;
2583
2584 default:
2585 assert(!"Unexpected state");
2586 return 1;
2587 }
2588
2589 /* Nothing's changed; soldier on */
2590 if (uf == old_uf) {
2591 u->field_data[uf].len++;
2592 continue;
2593 }
2594
2595 u->field_data[uf].off = (uint16_t)(p - buf);
2596 u->field_data[uf].len = 1;
2597
2598 u->field_set |= (1 << uf);
2599 old_uf = uf;
2600 }
2601
2602 /* host must be present if there is a schema */
2603 /* parsing http:///toto will fail */
2604 if ((u->field_set & (1 << UF_SCHEMA)) &&
2605 (u->field_set & (1 << UF_HOST)) == 0) {
2606 return 1;
2607 }
2608
2609 if (u->field_set & (1 << UF_HOST)) {
2610 if (http_parse_host(buf, u, found_at) != 0) {
2611 return 1;
2612 }
2613 }
2614
2615 /* CONNECT requests can only contain "hostname:port" */
2616 if (is_connect && u->field_set != ((1 << UF_HOST)|(1 << UF_PORT))) {
2617 return 1;
2618 }
2619
2620 if (u->field_set & (1 << UF_PORT)) {
2621 uint16_t off;
2622 uint16_t len;
2623 const char* p;
2624 const char* end;
2625 unsigned long v;
2626
2627 off = u->field_data[UF_PORT].off;
2628 len = u->field_data[UF_PORT].len;
2629 end = buf + off + len;
2630
2631 /* NOTE: The characters are already validated and are in the [0-9] range */
2632 assert((size_t) (off + len) <= buflen && "Port number overflow");
2633 v = 0;
2634 for (p = buf + off; p < end; p++) {
2635 v *= 10;
2636 v += *p - '0';
2637
2638 /* Ports have a max value of 2^16 */
2639 if (v > 0xffff) {
2640 return 1;
2641 }
2642 }
2643
2644 u->port = (uint16_t) v;
2645 }
2646
2647 return 0;
2648}
