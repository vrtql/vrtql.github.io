83CTEST2(test, buffer)
84{
85 vws_buffer* buffer = vws_buffer_new(&buffer);
86
87 // Append data to the buffer
88 cstr data1 = "Hello, ";
89 vws_buffer_append(buffer, (ucstr)data1, strlen(data1));
90
91 cstr data2 = "world!";
92 vws_buffer_append(buffer, (ucstr)data2, strlen(data2));
93
94 // Drain the first 7 bytes from the buffer
95 vws_buffer_drain(buffer, 7);
96
97 // Print the remaining data in the buffer
98 printf("%s\n", buffer->data);
99 ASSERT_STR((cstr)buffer->data, "world!");
100
101 // Free the buffer
102 vws_buffer_free(buffer);
