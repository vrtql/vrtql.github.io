28CTEST2(test, read)
29{
30 const char *json = "{\"name\":\"Mash\",\"star\":4,\"hits\":[2,2,1,3]}";
31
32 // Read JSON and get root
33 yyjson_doc* doc = yyjson_read(json, strlen(json), 0);
34 yyjson_val* root = yyjson_doc_get_root(doc);
35
36 // Get root["name"]
37 yyjson_val* name = yyjson_obj_get(root, "name");
38
39 ASSERT_TRUE(yyjson_get_type(root) == YYJSON_TYPE_OBJ);
40
41 printf("name: %s\n", yyjson_get_str(name));
42 printf("name length:%d\n", (int)yyjson_get_len(name));
43
44 // Get root["star"]
45 yyjson_val *star = yyjson_obj_get(root, "star");
46 printf("star: %d\n", (int)yyjson_get_int(star));
47
48 // Get root["hits"], iterate over the array
49 yyjson_val *hits = yyjson_obj_get(root, "hits");
50 size_t idx, max;
51 yyjson_val *hit;
52 yyjson_arr_foreach(hits, idx, max, hit)
53 {
54 printf("hit%d: %d\n", (int)idx, (int)yyjson_get_int(hit));
55 }
56
57 // Free the doc
58 yyjson_doc_free(doc);
59}

61CTEST2(test, write)
62{
63 // Create a mutable doc
64 yyjson_mut_doc* doc = yyjson_mut_doc_new(NULL);
65 yyjson_mut_val* root = yyjson_mut_obj(doc);
66 yyjson_mut_doc_set_root(doc, root);
67
68 // Set root["name"] and root["star"]
69 yyjson_mut_obj_add_str(doc, root, "name", "Mash");
70 yyjson_mut_obj_add_int(doc, root, "star", 4);
71
72 // Set root["hits"] with an array
73 int hits_arr[] = {2, 2, 1, 3};
74 yyjson_mut_val* hits = yyjson_mut_arr_with_sint32(doc, hits_arr, 4);
75 yyjson_mut_obj_add_val(doc, root, "hits", hits);
76
77 // To string, minified
78 cstr json = yyjson_mut_write(doc, 0, NULL);
79
80 if (json)
81 {
82 printf("json: %s\n", json); // {"name":"Mash","star":4,"hits":[2,2,1,3]}
83 free((void *)json);
84 }
85
86 // Free the doc
87 yyjson_mut_doc_free(doc);
88}
