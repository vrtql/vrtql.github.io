17void process_data(vws_svr_data* req)
18{
19 vws_tcp_svr* server = req->cnx->server;
20
21 vws.trace(VL_INFO, "process_data (%p)", req);
22
23 //> Prepare the response: echo the data back
24
25 // Allocate memory for the data to be sent in response
26 char* data = (char*)vws.malloc(req->size);
27
28 // Copy the request's data to the response data
29 strncpy(data, req->data, req->size);
30
31 // Create response
32 vws_svr_data* reply = vws_svr_data_own(req->cnx, (ucstr)data, req->size);
33
34 // Free request
35 vws_svr_data_free(req);
36
37 if (vws.tracelevel >= VT_APPLICATION)
38 {
39 vws.trace( VL_INFO,
40 "process_data(%p): %i bytes",
41 reply->cnx,
42 reply->size);
43 }
44
45 // Send reply. This will wakeup network thread.
46 vws_tcp_svr_send(server, reply);
47}

49void server_thread(void* arg)
50{
51 // Create a listening socket
52 int listenfd = socket(AF_INET, SOCK_STREAM, 0);
53
54 ASSERT_TRUE(listenfd >= 0);
55
56 // setsockopt: Handy debugging trick that lets us rerun the server
57 // immediately after we kill it; otherwise we have to wait about 20 secs.
58 // Eliminates "ERROR on binding: Address already in use" error.
59 int optval = 1;
60 setsockopt( listenfd, SOL_SOCKET, SO_REUSEADDR,
61 (const void*)&optval, sizeof(int) );
62
63 struct sockaddr_in serveraddr;
64 memset((char*)&serveraddr, 0, sizeof(serveraddr));
65 serveraddr.sin_family = AF_INET;
66 serveraddr.sin_addr.s_addr = htonl(INADDR_ANY);
67 serveraddr.sin_port = htons((unsigned short)server_port);
68
69 struct sockaddr* bindaddr = (struct sockaddr*)&serveraddr;
70 ASSERT_TRUE(bind(listenfd, bindaddr, sizeof(serveraddr)) >= 0);
71
72 vws_tcp_svr* server = (vws_tcp_svr*)arg;
73 server->state = VS_RUNNING;
74 server->on_data_in = process_data;
75 vws.tracelevel = VT_ALL;
76 server->trace = vws.tracelevel;
77
78 ASSERT_TRUE(listen(listenfd, 10) >= 0);
79
80 struct sockaddr_in clientaddr;
81 socklen_t clientlen = sizeof(clientaddr);
82 int connfd = accept(listenfd, (struct sockaddr*)&clientaddr, &clientlen);
83 ASSERT_TRUE(connfd >= 0);
84
85 vws_tcp_svr_inetd_run(server, connfd);
86}

88void client_thread(void* arg)
89{
90 // Connect
91 vws.trace(VL_INFO, "[CLIENT] Connecting");
92 vws_socket* s = vws_socket_new();
93 ASSERT_TRUE(vws_socket_connect(s, server_host, server_port, false));
94 vws.trace(VL_INFO, "[CLIENT] Connected");
95
96 // Send request
97 vws.trace(VL_INFO, "[CLIENT] Send: %s", content);
98 vws_socket_write(s, (ucstr)content, strlen(content));
99
100 // Get reply
101 ssize_t n = vws_socket_read(s);
102 ASSERT_TRUE(n > 0);
103 vws.trace(VL_INFO, "[CLIENT] Receive: %s", s->buffer->data);
104
105 // Disconnect and cleanup.
106 vws_socket_free(s);
107}

109CTEST(test_server, echo)
110{
111 vws_tcp_svr* server = vws_tcp_svr_new(10, 0, 0);
112 vws.tracelevel = VT_THREAD;
113 server->on_data_in = process_data;
114
115 vws.trace(VL_INFO, "[CLIENT] Starting server");
116
117 uv_thread_t server_tid;
118 uv_thread_create(&server_tid, server_thread, server);
119
120 // Wait for server to start up
121 while (server->state != VS_RUNNING)
122 {
123 vws_msleep(100);
124 }
125
126 int nc = 1;
127 uv_thread_t* threads = vws.malloc(sizeof(uv_thread_t) * nc);
128
129 for (int i = 0; i < nc; i++)
130 {
131 uv_thread_create(&threads[i], client_thread, NULL);
132 vws.trace(VL_INFO, "started client thread %p", threads[i]);
133 }
134
135 for (int i = 0; i < nc; i++)
136 {
137 uv_thread_join(&threads[i]);
138 vws.trace(VL_INFO, "stopped client thread %p", threads[i]);
139 }
140
141 free(threads);
142
143 // Server auto-shutdown on client disconnect. Wait for thread.
144 uv_thread_join(&server_tid);
145
146 // Cleanup
147 vws_tcp_svr_free(server);
148}
