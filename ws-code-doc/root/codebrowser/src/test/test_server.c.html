10void process_data(vws_svr_data* req)
11{
12 vrtql_svr* server = req->cnx->server;
13
14 vws.trace(VL_INFO, "process_data (%p)", req);
15
16 //> Prepare the response: echo the data back
17
18 // Allocate memory for the data to be sent in response
19 char* data = (char*)vws.malloc(req->size);
20
21 // Copy the request's data to the response data
22 strncpy(data, req->data, req->size);
23
24 // Create response
25 vws_svr_data* reply = vws_svr_data_own(req->cnx, data, req->size);
26
27 // Free request
28 vws_svr_data_free(req);
29
30 if (vws.tracelevel >= VT_APPLICATION)
31 {
32 vws.trace( VL_INFO,
33 "process_data(%p): %i bytes",
34 reply->cnx,
35 reply->size);
36 }
37
38 // Send reply. This will wakeup network thread.
39 vrtql_svr_send(server, reply);
40}

42void server_thread(void* arg)
43{
44 vrtql_svr* server = (vrtql_svr*)arg;
45
46 vws.tracelevel = VT_THREAD;
47 server->trace = vws.tracelevel;
48
49 vrtql_svr_run(server, server_host, server_port);

52void client_thread(void* arg)
53{
54 // Connect
55 vws.trace(VL_INFO, "[CLIENT] Connecting");
56 vws_socket* s = vws_socket_new();
57 ASSERT_TRUE(vws_socket_connect(s, server_host, server_port, false));
58 vws.trace(VL_INFO, "[CLIENT] Connected");
59
60 // Send request
61 vws.trace(VL_INFO, "[CLIENT] Send: %s", content);
62 vws_socket_write(s, content, strlen(content));
63
64 // Get reply
65 ssize_t n = vws_socket_read(s);
66 ASSERT_TRUE(n > 0);
67 vws.trace(VL_INFO, "[CLIENT] Receive: %s", s->buffer->data);
68
69 // Disconnect and cleanup.
70 vws_socket_free(s);
71}

73CTEST(test_server, echo)
74{
75 vrtql_svr* server = vrtql_svr_new(10, 0, 0);
76 vws.tracelevel = VT_THREAD;
77 server->on_data_in = process_data;
78
79 vws.trace(VL_INFO, "[CLIENT] Starting server");
80
81 uv_thread_t server_tid;
82 uv_thread_create(&server_tid, server_thread, server);
83
84 // Wait for server to start up
85 while (server->state != VS_RUNNING)
86 {
87 vws_msleep(100);
88 }
89
90 int nc = 10;
91 uv_thread_t* threads = vws.malloc(sizeof(uv_thread_t) * nc);
92
93 for (int i = 0; i < nc; i++)
94 {
95 uv_thread_create(&threads[i], client_thread, NULL);
96 vws.trace(VL_INFO, "started client thread %p", threads[i]);
97 }
98
99 for (int i = 0; i < nc; i++)
100 {
101 uv_thread_join(&threads[i]);
102 vws.trace(VL_INFO, "stopped client thread %p", threads[i]);
103 }
104
105 free(threads);
106
107 // Shutdown server
108 vws.trace(VL_INFO, "[CLIENT] Stopping server");
109 vrtql_svr_stop(server);
110 uv_thread_join(&server_tid);
111 vrtql_svr_free(server);
112
113 vws.trace(VL_INFO, "[CLIENT] Done");
114}
