13// Server function to process messages. Runs in context of worker thread.
14void process_message(vws_svr_cnx* cnx, vrtql_msg* m)
15{
16 vrtql_msg_svr* server = (vrtql_msg_svr*)cnx->server;
17
18 vws.trace(VL_INFO, "process_message (%p) %p", cnx, m);
19
20 // Echo back. Note: You should always set reply messages format to the
21 // format of the connection.
22
23 // Create reply message
24 vrtql_msg* reply = vrtql_msg_new();
25 reply->format = cnx->format;
26
27 // Copy content
28 ucstr data = m->content->data;
29 size_t size = m->content->size;
30 vws_buffer_append(reply->content, data, size);
31
32 // Send. We don't free message as send() does it for us.
33 server->send(cnx, reply);
34
35 // Clean up request
36 vrtql_msg_free(m);
37}

48void client_thread(void* arg)
49{
50 vws_cnx* cnx = vws_cnx_new();
51
52 while (vws_connect(cnx, uri) == false)
53 {
54 vws.trace(VL_ERROR, "[client]: connecting %s", uri);
55 }
56
57 cstr payload = "payload";
58
59 int i = 0;
60 while (true)
61 {
62 if (i++ > 10)
63 {
64 break;
65 }
66
67 // Create
68 vrtql_msg* request = vrtql_msg_new();
69 vrtql_msg_set_content(request, payload);
70
71 // Send
72 while (vrtql_msg_send(cnx, request) < 0)
73 {
74 if (vws_socket_is_connected((vws_socket*)cnx) == false)
75 {
76 goto restart;
77 }
78 }
79
80 // Receive
81 vrtql_msg* reply = NULL;
82
83 while (reply == NULL)
84 {
85 printf("vrtql_msg_recv(cnx)\n");
86 reply = vrtql_msg_recv(cnx);
87 if (vws_socket_is_connected((vws_socket*)cnx) == false)
88 {
89 goto restart;
90 }
91 }
92
93 // Check
94 ucstr content = reply->content->data;
95 size_t size = reply->content->size;
96 ASSERT_TRUE(strncmp(payload, (cstr)content, size) == 0);
97 vrtql_msg_free(reply);
98
99restart:
100
101 // Cleanup
102 vrtql_msg_free(request);
103 }
104
105 // Disconnect
106 vws_disconnect(cnx);
107 vws_cnx_free(cnx);
108}

110void client_test(int iterations, int nt)
111{
112 for (int i = 0; i < iterations; i++)
113 {
114 uv_thread_t* threads = vws.malloc(sizeof(uv_thread_t) * nt);
115
116 for (int i = 0; i < nt; i++)
117 {
118 uv_thread_create(&threads[i], client_thread, NULL);
119 vws.trace(VL_INFO, "started client thread %p", threads[i]);
120 }
121
122 for (int i = 0; i < nt; i++)
123 {
124 uv_thread_join(&threads[i]);
125 vws.trace(VL_INFO, "stopped client thread %p", threads[i]);
126 }
127
128 free(threads);
129 }
130}

132CTEST(test_msg_server, echo)
133{
134 vrtql_msg_svr* server = vrtql_msg_svr_new(10, 0, 0);
135 server->process = process_message;
136
137 uv_thread_t server_tid;
138 uv_thread_create(&server_tid, server_thread, server);
139
140 // Wait for server to start up
141 while (vws_tcp_svr_state((vws_tcp_svr*)server) != VS_RUNNING)
142 {
143 vws_msleep(100);
144 }
145
146 client_test(5, 50);
147
148 // Shutdown
149 vws_tcp_svr_stop((vws_tcp_svr*)server);
150 uv_thread_join(&server_tid);
151 vrtql_msg_svr_free(server);
152}
