11// Server function to process messages. Runs in context of worker thread.
12void process_message(vrtql_svr_cnx* cnx, vrtql_msg* m)
13{
14 vrtql_msg_svr* server = (vrtql_msg_svr*)cnx->server;
15
16 vrtql_trace(VL_INFO, "process_message (%p) %p", cnx, m);
17
18 // Echo back. Note: You should always set reply messages format to the
19 // format of the connection.
20
21 // Create reply message
22 vrtql_msg* reply = vrtql_msg_new();
23 reply->format = cnx->format;
24
25 // Copy content
26 cstr data = m->content->data;
27 size_t size = m->content->size;
28 vrtql_buffer_append(reply->content, data, size);
29
30 // Send. We don't free message as send() does it for us.
31 server->send(cnx, reply);
32
33 // Clean up request
34 vrtql_msg_free(m);
35}

37void server_thread(void* arg)
38{
39 vrtql_svr* server = (vrtql_svr*)arg;
40 vrtql_svr_trace(server, 1);
41 vrtql_svr_run(server, server_host, server_port);
42}

44CTEST(test_msg_server, echo)
45{
46 vrtql_msg_svr* server = vrtql_msg_svr_new(10, 0, 0);
47 server->process = process_message;
48
49 uv_thread_t server_tid;
50 uv_thread_create(&server_tid, server_thread, server);
51
52 // Wait for server to start up
53 while (vrtql_svr_state((vrtql_svr*)server) != VS_RUNNING)
54 {
55 vrtql_msleep(100);
56 }
57
58 vws_cnx* cnx = vws_cnx_new();
59 ASSERT_TRUE(vws_connect(cnx, uri));
60
61 cstr payload = "payload";
62
63 int i = 0;
64 while (true)
65 {
66 if (i++ > 10)
67 {
68 break;
69 }
70
71 // Create
72 vrtql_msg* request = vrtql_msg_new();
73 vrtql_msg_set_content(request, payload);
74
75 // Send
76 ASSERT_TRUE(vrtql_msg_send(cnx, request) > 0);
77
78 // Receive
79 vrtql_msg* reply = vrtql_msg_receive(cnx);
80
81 // Check
82 cstr content = reply->content->data;
83 size_t size = reply->content->size;
84 ASSERT_TRUE(strncmp(payload, content, size) == 0);
85
86 // Cleanup
87 vrtql_msg_free(request);
88 vrtql_msg_free(reply);
89 }
90
91 // Disconnect
92 vws_disconnect(cnx);
93 vws_cnx_free(cnx);
94
95 // Shutdown
96 vrtql_svr_stop((vrtql_svr*)server);
97 uv_thread_join(&server_tid);
98 vrtql_msg_svr_free(server);
99}
