37void dump_msg(vrtql_msg* m)
38{
39 m->format = VM_JSON_FORMAT;
40 vrtql_buffer* text = vrtql_msg_serialize(m);

45CTEST2(test, message)
46{
47 // Create request message
48 vrtql_msg* req = vrtql_msg_new();
49 vrtql_msg_set_routing(req, "tag", "xVx1");
50 vrtql_msg_set_header(req, "id", "vql");
51
52 // Headers for authentication
53 vrtql_msg_set_header(req, "username", "vrtql");
54 vrtql_msg_set_header(req, "password", "vrtql");
55 vrtql_msg_set_header(req, "domain", "global");
56
57 // VQL command to execute
58 vrtql_msg_set_content(req, "sql select * from solari");
59
60 // Process request
61 ASSERT_TRUE(vrtql_msg_send(data->c, req) > 0);
62 vrtql_msg* reply = vrtql_msg_recv(data->c);
63 ASSERT_TRUE(reply != NULL);
64 dump_msg(reply);
65
66 // Remove uneeded headers as the websocket state is established.
67 vrtql_msg_clear_header(req, "username");
68 vrtql_msg_clear_header(req, "password");
69 vrtql_msg_clear_header(req, "domain");
70
71 // Process another request
72 vrtql_msg_set_content(req, "get system info");
73 ASSERT_TRUE(vrtql_msg_send(data->c, req) > 0);
74 reply = vrtql_msg_recv(data->c);
75 content = (cstr)reply->content->data;
76 vrtql.trace(VL_INFO, "%s", content);
77
78 // Cleanup
79 vrtql_msg_free(req);
80 vrtql_msg_free(reply);
81}
