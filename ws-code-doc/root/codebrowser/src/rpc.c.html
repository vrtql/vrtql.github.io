144bool parse_rpc_string(const char* input, char** module, char** function)
145{
146 // Find the first occurrence of the period
147 const char* delimiter = strchr(input, '.');
148
149 if (delimiter != NULL)
150 {
151 // Calculate the lengths of the module and function substrings
152 size_t m_len = delimiter - input;
153 size_t f_len = strlen(input) - m_len - 1;
154
155 // Allocate memory for the module and function strings
156 *module = (char*)vrtql.malloc((m_len + 1) * sizeof(char));
157 *function = (char*)vrtql.malloc((f_len + 1) * sizeof(char));
158
159 // Copy the module substring
160 strncpy(*module, input, m_len);
161 (*module)[m_len] = '\0';
162
163 // Copy the function substring
164 strncpy(*function, delimiter + 1, f_len);
165 (*function)[f_len] = '\0';
166
167 return true;
168 }
169
170 // Invalid input format
171 return false;
172}

174vrtql_msg* vrtql_rpc(vrtql_rpc_system* s, vrtql_rpc_env* e, vrtql_msg* m)
175{
176 vrtql.success();
177
178 cstr id = vrtql_msg_get_header(m, "id");
179
180 if (id == NULL)
181 {
182 vrtql.error(VE_RT, "ID not specified");
183
184 vrtql_msg_free(m);
185
186 return NULL;
187 }
188
189 // Parse ID into module and function
190 char* mn; char* fn;
191 if (parse_rpc_string(id, &mn, &fn) == false)
192 {
193 vrtql.error(VE_RT, "Invalid ID format");
194
195 vrtql_msg_free(m);
196
197 return NULL;
198 }
199
200 // Lookup module in system
201 vrtql_rpc_module* module = vrtql_rpc_system_get(s, mn);
202
203 if (module == NULL)
204 {
205 vrtql.error(VE_RT, "RPC does not exist");
206
207 vrtql_msg_free(m);
208 free(mn);
209 free(fn);
210
211 return NULL;
212 }
213
214 // Lookup RPC in module
215 vrtql_rpc_call rpc = vrtql_rpc_module_get(module, fn);
216
217 if (rpc == NULL)
218 {
219 vrtql.error(VE_RT, "RPC does not exist");
220
221 vrtql_msg_free(m);
222 free(mn);
223 free(fn);
224
225 return NULL;
226 }
227
228 free(mn);
229 free(fn);
230
231 // Invoke RPC
232 vrtql_msg* reply = rpc(e, m);
233
234 // Free request
235 vrtql_msg_free(m);
236
237 // Return reply
238 return reply;
239}

245void* sys_map_get(vrtql_rpc_map* map, cstr key)
246{
247 // See if we have an existing entry
248 cstr v = sc_map_get_sv(map, key);
249
250 if (sc_map_found(map) == false)
251 {
252 return NULL;
253 }
254
255 return v;
256}

258void sys_map_set(vrtql_rpc_map* map, cstr key, void* value)
259{
260 // See if we have an existing entry
261 sc_map_get_sv(map, key);
262
263 if (sc_map_found(map) == false)
264 {
265 // We don't. Therefore we need to allocate new key.
266 key = strdup(key);
267 }
268
269 sc_map_put_sv(map, key, value);
270}
271
272void sys_map_clear(vrtql_rpc_map* map, cstr key, vrtql_rpc_map_free cb)
273{
274 // See if we have an existing entry
275 cstr v = sc_map_get_sv(map, key);
276
277 if (sc_map_found(map) == true)
278 {
279 // Call callback function to cleanup
280 cb(v);
281 }
282
283 sc_map_del_sv(map, key);
284}
