37char* vrtql_rpc_tag(uint16_t length)
38{
39 char valid_chars[] = "abcdefghijklmnopqrstuvwxyz0123456789";
40 unsigned char* data = (unsigned char*)malloc(length);
41 unsigned char* tag = (unsigned char*)malloc(length);
42
43 if (RAND_bytes(data, length) != 1)
44 {
45 free(data);
46 free(tag);
47
48 return NULL;
49 }
50
51 uint16_t vc_len = strlen(valid_chars);
52 for (uint16_t cnt = 0; cnt < length; cnt++)
53 {
54 uint16_t c = data[cnt] % vc_len;
55 tag[cnt] = valid_chars[c];
56 }
57
58 free(data);
59
60 return (char*)tag;
61}

63bool vrtql_rpc_invoke(vrtql_rpc* rpc, vrtql_msg* req)
64{
65 // Clear previous response
66 vws_buffer_clear(rpc->val);
67
68 // Make the call
69 vrtql_msg* reply = vrtql_rpc_exec(rpc, req);
70
71 // If no response
72 if (reply == NULL)
73 {
74 // Error already set
75 return false;
76 }
77
78 // If we received content
79 if (req->content->size > 0)
80 {
81 // Copy it
82 vws_buffer_append(rpc->val, reply->content->data, reply->content->size);
83 }
84
85 // Translate response code and message
86 cstr rc = vrtql_msg_get_header(reply, "rc");
87 cstr msg = vrtql_msg_get_header(reply, "msg");
88
89 if ((rc != NULL) && (msg != NULL))
90 {
91 vws.error(atoi(rc), msg);
92 }
93 else
94 {
95 vws.e.code = atoi(rc);
96 }
97
98 // Cleanup
99 vrtql_msg_free(req);
100 vrtql_msg_free(reply);
101
102 return true;
103}

105vrtql_msg* vrtql_rpc_exec(vrtql_rpc* rpc, vrtql_msg* req)
106{
107 //> Send request
108
109 // Assign a tag to verify response
110 cstr tag = vrtql_rpc_tag(7);
111 vrtql_msg_set_routing(req, "tag", tag);
112
113 if (vws.tracelevel >= VT_SERVICE)
114 {
115 vws_trace_lock();
116 printf("\n\n");
117 printf("+----------------------------------------------------+\n");
118 printf("| Message Sent |\n");
119 printf("+----------------------------------------------------+\n");
120 vrtql_msg_dump(req);
121 printf("------------------------------------------------------\n");
122 vws_trace_unlock();
123 }
124
125 // Loop until message sent or fatal error. Timeouts are ignored: keeps
126 // grinding until message is fully sent or error.
127 while (true)
128 {
129 if (vrtql_msg_send(rpc->cnx, req) > 0)
130 {
131 // Message successfully sent
132 break;
133 }
134
135 // If connection dropped
136 if (vws.e.code == VE_SOCKET)
137 {
138 // Try to reconnect
139 if (reconnect(rpc) == true)
140 {
141 // Reconnect worked. Try again.
142 continue;
143 }
144
145 // Failed to reconnect. Modify error to indicate the failure was on
146 // send so the caller knows the message was not sent. Error will
147 // have both bits set: VE_SOCKET and VE_SEND.
148 vws_set_flag(&vws.e.code, VE_SEND);
149 }
150
151 // Hand error back to caller.
152 free(tag);
153 return NULL;
154 }
155
156 //> Wait for response
157
158 int retries = 0;
159 vrtql_msg* reply = NULL;
160
161 while (retries < rpc->retries)
162 {
163 reply = vrtql_msg_recv(rpc->cnx);
164
165 // If we have a message
166 if (reply != NULL)
167 {
168 // Get message tag
169 cstr t = vrtql_msg_get_routing(reply, "tag");
170
171 // If tags do not match
172 if (strncmp(tag, t, strlen(tag)) != 0)
173 {
174 // This is not response message. Send to handler.
175 rpc->out_of_band(rpc, reply);
176
177 // Keep waiting for response.
178 continue;
179 }
180
181 vws.success();
182
183 break;
184 }
185
186 if (vws.e.code == VE_TIMEOUT)
187 {
188 retries++;
189 continue;
190 }
191 else
192 {
193 if (vws.e.code == VE_SOCKET)
194 {
195 // Modify error to indicate the failure was on recv, so the
196 // caller knows the message was sent. Error will have both bits
197 // set: VE_SOCKET and VE_RECV.
198 vws_set_flag(&vws.e.code, VE_RECV);
199 }
200
201 // Something unexpected happend. We expect vrtql_msg_recv() to set
202 // appropriate error. Even if disconenct (VE_SOCKET) there is no
203 // point in reconnecting because we have lost our response no matter
204 // what. We will leave error as it is and hand back to caller.
205 break;
206 }
207 }
208
209 free(tag);
210
211 if ((vws.tracelevel >= VT_SERVICE) && (reply != NULL))
212 {
213 vws_trace_lock();
214 printf("\n\n");
215 printf("+----------------------------------------------------+\n");
216 printf("| Message Received |\n");
217 printf("+----------------------------------------------------+\n");
218 vrtql_msg_dump(reply);
219 printf("------------------------------------------------------\n");
220 vws_trace_unlock();
221 }
222
223 return reply;
224}

396bool parse_rpc_string(const char* input, char** module, char** function)
397{
398 // Find the first occurrence of the period
399 const char* delimiter = strchr(input, '.');
400
401 if (delimiter != NULL)
402 {
403 // Calculate the lengths of the module and function substrings
404 size_t m_len = delimiter - input;
405 size_t f_len = strlen(input) - m_len - 1;
406
407 // Allocate memory for the module and function strings
408 *module = (char*)vws.malloc((m_len + 1) * sizeof(char));
409 *function = (char*)vws.malloc((f_len + 1) * sizeof(char));
410
411 // Copy the module substring
412 strncpy(*module, input, m_len);
413 (*module)[m_len] = '\0';
414
415 // Copy the function substring
416 strncpy(*function, delimiter + 1, f_len);
417 (*function)[f_len] = '\0';
418
419 return true;
420 }
421
422 // Invalid input format
423 return false;
424}

426vrtql_msg* vrtql_rpc_service(vrtql_rpc_system* s, vrtql_rpc_env* e, vrtql_msg* m)
427{
428 vws.success();
429
430 cstr id = vrtql_msg_get_header(m, "id");
431
432 if (id == NULL)
433 {
434 vws.error(VE_RT, "ID not specified");
435
436 vrtql_msg_free(m);
437
438 return NULL;
439 }
440
441 // Parse ID into module and function
442 char* mn; char* fn;
443 if (parse_rpc_string(id, &mn, &fn) == false)
444 {
445 vws.error(VE_RT, "Invalid ID format");
446
447 vrtql_msg_free(m);
448
449 return NULL;
450 }
451
452 // Lookup module in system
453 vrtql_rpc_module* module = vrtql_rpc_system_get(s, mn);
454
455 if (module == NULL)
456 {
457 vws.error(VE_RT, "RPC does not exist");
458
459 vrtql_msg_free(m);
460 vws.free(mn);
461 vws.free(fn);
462
463 return NULL;
464 }
465
466 // Lookup RPC in module
467 vrtql_rpc_call rpc = vrtql_rpc_module_get(module, fn);
468
469 if (rpc == NULL)
470 {
471 vws.error(VE_RT, "RPC does not exist");
472
473 vrtql_msg_free(m);
474 vws.free(mn);
475 vws.free(fn);
476
477 return NULL;
478 }
479
480 vws.free(mn);
481 vws.free(fn);
482
483 // Invoke RPC
484 vrtql_msg* reply = rpc(e, m);
485
486 // Free request
487 vrtql_msg_free(m);
488
489 // Return reply
490 return reply;
491}

510void sys_map_set(vrtql_rpc_map* map, cstr key, void* value)
511{
512 // See if we have an existing entry
513 sc_map_get_sv(map, key);
514
515 if (sc_map_found(map) == false)
516 {
517 // We don't. Therefore we need to allocate new key.
518 key = strdup(key);
519 }
520
521 sc_map_put_sv(map, key, value);
522}
523
524void sys_map_clear(vrtql_rpc_map* map, cstr key, vrtql_rpc_map_free cb)
525{
526 // See if we have an existing entry
527 cstr v = sc_map_get_sv(map, key);
528
529 if (sc_map_found(map) == true)
530 {
531 // Call callback function to cleanup
532 cb(v);
533 }
534
535 sc_map_del_sv(map, key);
536}
