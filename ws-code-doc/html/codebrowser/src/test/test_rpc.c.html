44CTEST(test_rpc, call)
45{
46 // Create RPC system
47 vrtql_rpc_system* system = vrtql_rpc_system_new();
48
49 // Create module
50 vrtql_rpc_module* module = vrtql_rpc_module_new("session");
51 vrtql_rpc_module_set(module, "login", session_login);
52 vrtql_rpc_module_set(module, "logout", session_logout);
53 vrtql_rpc_module_set(module, "info", session_info);
54
55 // Register module in system
56 vrtql_rpc_system_set(system, module);
57
58 // Setup RPC request
59 vrtql_rpc_env env;
60 vrtql_msg* req = vrtql_msg_new();
61 vrtql_msg_set_header(req, "id", "session.login");
62
63 // Invoke RPC
64 vrtql_msg* reply = vrtql_rpc(system, &env, req);
65 ASSERT_TRUE(reply != NULL);
66
67 // Verify reply
68 cstr rc = vrtql_msg_get_header(reply, "rc");
69 ASSERT_TRUE(strncmp(rc, "0", 1) == 0);
70
71 // Cleanup
72 vrtql_msg_free(reply);
73 vrtql_rpc_system_free(system);
74}
