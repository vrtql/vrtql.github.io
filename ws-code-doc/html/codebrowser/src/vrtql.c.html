83void vrtql_trace(vrtql_log_level level, const char* format, ...)
84{
85 if (level < 0 || level >= VL_LEVEL_COUNT)
86 {
87 vrtql_error_default_submit(VE_WARN, "Invalid log level");
88 return;
89 }
90
91 time_t raw_time;
92 struct tm time_info;
93 char stamp[20];
94
95 time(&raw_time);
96
97#ifdef __windows__
98 // Windows implementation using localtime_s
99 if (localtime_s(&time_info, &raw_time) != 0)
100 {
101 vrtql_error_default_submit(VE_SYS, "localtime_s failed");
102 return;
103 }
104#else
105 // Non-Windows implementation using localtime_r
106 if (localtime_r(&raw_time, &time_info) == NULL)
107 {
108 vrtql_error_default_submit(VE_SYS, "localtime_r failed");
109 return;
110 }
111#endif
112
113 if (strftime(stamp, sizeof(stamp), "%Y-%m-%d %H:%M:%S", &time_info) == 0)
114 {
115 vrtql_error_default_submit(VE_SYS, "strftime returned 0");
116 return;
117 }
118
119 const char* color_code = log_level_infos[level].color;
120 const char* level_name = log_level_infos[level].level;
121
122#if defined(__windows__)
123
124 DWORD tid = GetCurrentThreadId();
125
126 // Windows implementation using Windows API for thread synchronization
127 if (WaitForSingleObject(log_mutex, INFINITE) != WAIT_OBJECT_0)
128 {
129 vrtql_error_default_submit(VE_SYS, "WaitForSingleObject failed");
130 return;
131 }
132
133#else
134
135 pthread_t tid = pthread_self();
136
137 if (pthread_mutex_lock(&log_mutex) != 0)
138 {
139 vrtql_error_default_submit(VE_SYS, "pthread_mutex_lock failed");
140 return;
141 }
142
143 pthread_cleanup_push(unlock_mutex, &log_mutex);
144
145#endif
146
147 fprintf(stderr, "%s[%s] [%lu] [%s]%s ",
148 color_code,
149 stamp,
150#if defined(__windows__)
151 tid,
152#else
153 (unsigned long)tid,
154#endif
155 level_name,
156 ANSI_COLOR_RESET);
157
158 va_list args;
159 va_start(args, format);
160 vfprintf(stderr, format, args);
161 va_end(args);
162
163 fprintf(stderr, "\n");
164 fflush(stderr);
165
166#if defined(__windows__)
167
168 // Windows implementation using Windows API for thread synchronization
169 if (!ReleaseMutex(log_mutex))
170 {
171 vrtql_error_default_submit(VE_SYS, "ReleaseMutex failed");
172 }
173
174#else
175
176 pthread_cleanup_pop(1);
177
178#endif
179}

278int vrtql_error_default_submit(int code, cstr format, ...)
279{
280 va_list args, args_copy;
281 va_start(args, format);
282
283 va_copy(args_copy, args);
284
285 // Determine the length of the formatted string
286 int length = vsnprintf(NULL, 0, format, args_copy);
287
288 va_end(args_copy); // End args_copy. We're done with it now.
289
290 // Allocate a buffer for the formatted string
291 char* buffer = malloc(length + 1);
292
293 // Format the string into the buffer
294 vsnprintf(buffer, length + 1, format, args);
295
296 // Set
297 vrtql_set_error(code, buffer);
298
299 // Process
300 vrtql.process_error(code, buffer);
301
302 // Print the formatted string to stderr
303 fprintf(stderr, "%s", buffer);
304
305 // Cleanup
306 free(buffer);
307 va_end(args);
308
309 return 0;
310}

312int vrtql_error_default_process(int code, cstr message)
313{
314 if (vrtql.trace == 1)
315 {
316 switch (code)
317 {
318 case VE_TIMEOUT:
319 case VE_WARN:
320 {
321 vrtql_trace(VL_WARN, "Error %i: %s", code, message);
322 break;
323 }
324
325 case VE_SYS:
326 case VE_RT:
327 {
328 vrtql_trace(VL_INFO, "Error %i: %s", code, message);
329 break;
330 }
331
332 case VE_MEM:
333 case VE_FATAL:
334 {
335 vrtql_trace(VL_ERROR, "Error %i: %s", code, message);
336 break;
337 }
338
339 default:
340 {
341 vrtql_trace(VL_INFO, "No error");
342 }
343 }
344 }
345
346 switch (code)
347 {
348 case VE_MEM:
349 {
350 fprintf(stderr, "Out of memory error\n");
351 break;
352 }
353
354 case VE_FATAL:
355 {
356 fprintf(stderr, "Fatal error\n");
357 exit(1);
358 }
359
360 default:
361 {
362 if (message != NULL)
363 {
364 fprintf(stderr, "Error %i: %s\n", code, message);
365 }
366 }
367 }
368
369 return 0;
370}

465void vrtql_buffer_append(vrtql_buffer* buffer, ucstr data, size_t size)
466{
467 if (buffer == NULL || data == NULL)
468 {
469 return;
470 }
471
472 size_t total_size = buffer->size + size;
473
474 if (total_size > buffer->allocated)
475 {
476 buffer->allocated = total_size * 1.5;
477
478 ucstr mem;
479 if (buffer->data == NULL)
480 {
481 mem = (ucstr)vrtql.malloc(buffer->allocated);
482 }
483 else
484 {
485 mem = (ucstr)vrtql.realloc(buffer->data, buffer->allocated);
486 }
487
488 buffer->data = mem;
489 }
490
491 memcpy(buffer->data + buffer->size, data, size);
492 buffer->size = total_size;
493}
494
495void vrtql_buffer_drain(vrtql_buffer* buffer, size_t size)
496{
497 if (buffer == NULL || buffer->data == NULL)
498 {
499 return;
500 }
501
502 if (size >= buffer->size)
503 {
504 // When size >= buffer->size, clear the whole buffer
505 vrtql_buffer_clear(buffer);
506 }
507 else
508 {
509 memmove(buffer->data, buffer->data + size, buffer->size - size);
510 buffer->size -= size;
511 buffer->data[buffer->size] = 0;
512 }
513}

579
580char* vrtql_generate_uuid()
581{
582 unsigned char uuid[16];
583
584 if (RAND_bytes(uuid, sizeof(uuid)) != 1)
585 {
586 return NULL;
587 }
588
589 // Set the version (4) and variant bits
590 uuid[6] = (uuid[6] & 0x0F) | 0x40;
591 uuid[8] = (uuid[8] & 0x3F) | 0x80;
592
593 // Format the UUID as a string
594 char* encoded_uuid = vrtql_base64_encode(uuid, sizeof(uuid));
595
596 if (encoded_uuid == NULL)
597 {
598 return NULL;
599 }
600
601 // Remove padding characters and dashes
602 size_t encoded_uuid_length = strlen(encoded_uuid);
603 for (size_t i = 0; i < encoded_uuid_length; i++)
604 {
605 if ( encoded_uuid[i] == '=' ||
606 encoded_uuid[i] == '\n' ||
607 encoded_uuid[i] == '\r' ||
608 encoded_uuid[i] == '-' )
609 {
610 encoded_uuid[i] = '_';
611 }
612 }
613
614 return encoded_uuid;
615}

621char* vrtql_base64_encode(const unsigned char* data, size_t length)
622{
623 BIO* bio = BIO_new(BIO_s_mem());
624 BIO* base64 = BIO_new(BIO_f_base64());
625
626 BIO_set_flags(base64, BIO_FLAGS_BASE64_NO_NL);
627 BIO_push(base64, bio);
628 BIO_write(base64, data, length);
629 BIO_flush(base64);
630
631 BUF_MEM* ptr;
632 BIO_get_mem_ptr(base64, &ptr);
633 size_t output_length = ptr->length;
634 char* encoded_data = (char*)vrtql.malloc(output_length + 1);
635
636 memcpy(encoded_data, ptr->data, output_length);
637 encoded_data[output_length] = '\0';
638
639 BIO_free_all(base64);
640
641 return encoded_data;
642}

644unsigned char* vrtql_base64_decode(const char* data, size_t* size)
645{
646 BIO* bio = BIO_new_mem_buf(data, -1);
647 BIO* base64 = BIO_new(BIO_f_base64());
648
649 BIO_set_flags(base64, BIO_FLAGS_BASE64_NO_NL);
650 BIO_push(base64, bio);
651
652 // Determine the size of the decoded data
653 size_t encoded_length = strlen(data);
654 size_t decoded_length = (encoded_length * 3) / 4; // Rough estimate
655 unsigned char* decoded_data = (unsigned char*)vrtql.malloc(decoded_length);
656
657 // Decode the base64 data
658 *size = BIO_read(base64, decoded_data, encoded_length);
659
660 BIO_free_all(base64);
661
662 return decoded_data;
663}

670 * @brief Sleeps for the specified number of milliseconds.
671 *
672 * This function provides a platform-independent way to sleep for a specific
673 * duration in milliseconds. The behavior is similar to the standard `sleep`
674 * function, but with millisecond precision.
675 *
676 * @param ms The number of milliseconds to sleep.
677 *
678 * @note This function may not be available on all platforms. Make sure to

680 * warnings specific to your environment.

685

691

694

698 ts.tv_sec = ms / 1000;
699 ts.tv_nsec = (ms % 1000) * 1000000;
700 nanosleep(&ts, NULL);
701
702#else

705
706#endif

728static char* url_extract_part(const char* url, const char* sep, char** rest)
729{
730 const char* part_end = strstr(url, sep);
731 if (part_end == NULL)
732 {
733 return NULL;
734 }
735
736 size_t part_length = part_end - url;
737 char* part = vrtql.malloc(part_length + 1);
738
739 strncpy(part, url, part_length);
740 part[part_length] = '\0';
741
742 *rest = strdup(part_end + strlen(sep));
743 if (*rest == NULL)
744 {
745 vrtql.error(VE_MEM, "strdup()");
746 free(part);
747 return NULL;
748 }
749
750 return part;
751}

774static char* vrtql_url_extract_port(char** host)
775{
776 char* port_start = strchr(*host, ':');
777 if (port_start == NULL)
778 {
779 return NULL;
780 }
781
782 char* port = strdup(port_start + 1);
783 if (port == NULL)
784 {
785 vrtql.error(VE_MEM, "strdup()");
786 return NULL;
787 }
788
789 char* truncated_host = strndup(*host, port_start - *host);
790 if (truncated_host == NULL)
791 {
792 vrtql.error(VE_MEM, "strndup()");
793 free(port);
794 return NULL;
795 }
796
797 free(*host);
798 *host = truncated_host;
799
800 return port;
801}

803vrtql_url vrtql_url_parse(const char* url)
804{
805 vrtql_url parts = { NULL, NULL, NULL, NULL, NULL, NULL };
806
807 // Extract the scheme
808
809 parts.scheme = url_extract_part(url, "://", (char**)&url);
810
811 if (parts.scheme == NULL)
812 {
813 url = strdup(url);
814
815 if (url == NULL)
816 {
817 vrtql.error(VE_MEM, "strdup()");
818 }
819 }
820
821 // Extract the fragment
822
823 char* rest = NULL;
824 parts.fragment = url_extract_part(url, "#", &rest);
825
826 if (parts.fragment != NULL)
827 {
828 free(url);
829 url = rest;
830 }
831 else if (rest != NULL)
832 {
833 free(rest);
834 }
835
836 // Extract the query
837
838 rest = NULL;
839 parts.query = url_extract_part(url, "?", &rest);
840
841 if (parts.query != NULL)
842 {
843 free(url);
844 url = rest;
845 }
846 else if (rest != NULL)
847 {
848 free(rest);
849 }
850
851 // The remaining URL is the host and path
852
853 rest = NULL;
854 parts.host = url_extract_part(url, "/", &rest);
855
856 if (parts.host != NULL)
857 {
858 free(url);
859 url = rest;
860 rest = NULL;
861 }
862 else
863 {
864 if (parts.host == NULL)
865 {
866 vrtql.error(VE_MEM, "strdup()");
867 }
868
869 parts.host = strdup(url);
870 free(url);
871
872 if (rest != NULL)
873 {
874 free(rest);
875 }
876 }
877
878 // The rest of the URL is the path. Add leading slash manually.
879 size_t path_len = strlen(url);
880
881 // Allocate memory for '/' + url + '\0'
882 parts.path = (char*)vrtql.malloc((path_len + 2) * sizeof(char));
883
884 if (url[0] != '/')
885 {
886 parts.path[0] = '/';
887 strcpy(parts.path + 1, url);
888 }
889 else
890 {
891 strcpy(parts.path, url);
892 }
893
894 free(url);
895
896 // Extract the port from the host
897 parts.port = vrtql_url_extract_port(&parts.host);
898
899 return parts;
900}

955char* vrtql_url_build(const vrtql_url* parts)
956{
957 // Calculate the total length needed for the URL
958 size_t total_length = 1; // for null-terminating character '\0'
959
960 if (parts->scheme != NULL)
961 {
962 total_length += strlen(parts->scheme) + 3; // "://"
963 }
964
965 if (parts->host != NULL)
966 {
967 total_length += strlen(parts->host);
968 }
969
970 if (parts->port != NULL)
971 {
972 total_length += strlen(parts->port) + 1; // ":"
973 }
974
975 if (parts->path != NULL)
976 {
977 total_length += strlen(parts->path);
978 }
979
980 if (parts->query != NULL)
981 {
982 total_length += strlen(parts->query) + 1; // "?"
983 }
984
985 if (parts->fragment != NULL)
986 {
987 total_length += strlen(parts->fragment) + 1; // "#"
988 }
989
990 // Allocate memory for the URL string
991 char* url = (char*)vrtql.malloc(total_length);
992
993 url[0] = '\0'; // Start with an empty string
994
995 // Use snprintf to append to the string, which is safer and potentially
996 // faster than strcat
997
998 if (parts->scheme != NULL)
999 {
1000 snprintf(url + strlen(url), total_length, "%s://", parts->scheme);
1001 }
1002
1003 if (parts->host != NULL)
1004 {
1005 snprintf(url + strlen(url), total_length, "%s", parts->host);
1006 }
1007
1008 if (parts->port != NULL)
1009 {
1010 snprintf(url + strlen(url), total_length, ":%s", parts->port);
1011 }
1012
1013 if (parts->path != NULL)
1014 {
1015 snprintf(url + strlen(url), total_length, "%s", parts->path);
1016 }
1017
1018 if (parts->query != NULL)
1019 {
1020 snprintf(url + strlen(url), total_length, "?%s", parts->query);
1021 }
1022
1023 if (parts->fragment != NULL)
1024 {
1025 snprintf(url + strlen(url), total_length, "#%s", parts->fragment);
1026 }
1027
1028 return url;
1029}
