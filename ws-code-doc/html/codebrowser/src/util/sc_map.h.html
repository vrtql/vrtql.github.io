67#define sc_map_of(name, K, V) \
68 struct sc_map_##name { \
69 struct sc_map_item_##name *mem; \
70 uint32_t cap; \
71 uint32_t size; \
72 uint32_t load_fac; \
73 uint32_t remap; \
74 bool used; \
75 bool oom; \
76 bool found; \
77 }; \

79 /** \
80 * Create map \
81 * \
82 * @param map map \
83 * @param cap initial capacity, zero is accepted \
84 * @param load_factor must be >25 and <95. Pass 0 for default value. \
85 * @return 'true' on success, \
86 * 'false' on out of memory or if 'load_factor' value is \
87 * invalid. \
88 */ \
89 bool sc_map_init_##name(struct sc_map_##name *map, uint32_t cap, \
90 uint32_t load_factor); \
91 \
92 /** \
93 * Destroy map. \
94 * \
95 * @param map map \
96 */ \
97 void sc_map_term_##name(struct sc_map_##name *map); \
98 \
99 /** \
100 * Get map element count \
101 * \
102 * @param map map \
103 * @return element count \
104 */ \
105 uint32_t sc_map_size_##name(struct sc_map_##name *map); \
106 \
107 /** \
108 * Clear map \
109 * \
110 * @param map map \
111 */ \
112 void sc_map_clear_##name(struct sc_map_##name *map); \
113 \
114 /** \
115 * Put element to the map \
116 * \
117 * struct sc_map_str map; \
118 * sc_map_put_str(&map, "key", "value"); \
119 * \
120 * @param map map \
121 * @param K key \
122 * @param V value \
123 * @return previous value if exists \
124 * call sc_map_found() to see if the returned value is valid. \
125 */ \
126 V sc_map_put_##name(struct sc_map_##name *map, K key, V val); \
127 \
128 /** \
129 * Get element \
130 * \
131 * @param map map \
132 * @param K key \
133 * @return current value if exists. \
134 * call sc_map_found() to see if the returned value is valid. \
135 */ \
136 /** NOLINTNEXTLINE */ \
137 V sc_map_get_##name(struct sc_map_##name *map, K key); \
138 \
139 /** \
140 * Delete element \
141 * \
142 * @param map map \
143 * @param K key \
144 * @return current value if exists. \
145 * call sc_map_found() to see if the returned value is valid. \
146 */ \
147 /** NOLINTNEXTLINE */ \
148 V sc_map_del_##name(struct sc_map_##name *map, K key);
149
150/**
151 * @param map map
152 * @return - if put operation overrides a value, returns true
153 * - if get operation finds the key, returns true
154 * - if del operation deletes a key, returns true
155 */
156#define sc_map_found(map) ((map)->found)
157
158/**
159 * @param map map
160 * @return true if put operation failed with out of memory

162#define sc_map_oom(map) ((map)->oom)

166/**
167 * Foreach loop
168 *
169 * char *key, *value;
170 * struct sc_map_str map;
171 *
172 * sc_map_foreach(&map, key, value) {
173 * printf("key = %s, value = %s \n");
174 * }
175 */
176#define sc_map_foreach(map, K, V) \
177 for (int64_t _i = -1, _b = 0; !_b && _i < (map)->cap; _i++) \
178 for ((V) = (map)->mem[_i].value, (K) = (map)->mem[_i].key, _b = 1; \
179 _b && ((_i == -1 && (map)->used) || (K) != 0) ? 1 : (_b = 0); \
180 _b = 0)
181
182/**
183 * Foreach loop for keys
184 *
185 * char *key;
186 * struct sc_map_str map;
187 *
188 * sc_map_foreach_key(&map, key) {
189 * printf("key = %s \n");
190 * }
191 */
192#define sc_map_foreach_key(map, K) \
193 for (int64_t _i = -1, _b = 0; !_b && _i < (map)->cap; _i++) \
194 for ((K) = (map)->mem[_i].key, _b = 1; \
195 _b && ((_i == -1 && (map)->used) || (K) != 0) ? 1 : (_b = 0); \
196 _b = 0)
197
198/**
199 * Foreach loop for values
200 *
201 * char *value;
202 * struct sc_map_str map;
203 *
204 * sc_map_foreach_value(&map, value) {
205 * printf("value = %s \n");
206 * }
207 */
208#define sc_map_foreach_value(map, V) \
209 for (int64_t _i = -1, _b = 0; !_b && _i < (map)->cap; _i++) \
210 for ((V) = (map)->mem[_i].value, _b = 1; \
211 _b && ((_i == -1 && (map)->used) || (map)->mem[_i].key != 0) ? 1 : (_b = 0); \
212 _b = 0)
