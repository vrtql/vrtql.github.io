219 * SSL_export_keying_material exports a value derived from the master secret,
220 * as specified in RFC 5705. It writes |olen| bytes to |out| given a label and

222 * flag controls whether a context is included.) It returns 1 on success and
223 * 0 or -1 otherwise.
224 */

227 const unsigned char *context,
